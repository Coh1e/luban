21ComponentRecord parse_record(const std::string& name, const json& j) {
22 ComponentRecord r;
23 r.name = name;
24 auto get_str = [&](const char* k) -> std::string {
25 if (!j.contains(k) || j[k].is_null()) return "";
26 if (j[k].is_string()) return j[k].get<std::string>();
27 return j[k].dump();
28 };
29 r.version = get_str("version");
30 r.source = get_str("source");
31 r.url = get_str("url");
32 r.hash_spec = get_str("hash");
33 r.toolchain_dir = get_str("toolchain_dir");
34 r.architecture = j.value("architecture", std::string{"x86_64"});
35 r.installed_at = get_str("installed_at");
36 if (j.contains("store_keys") && j["store_keys"].is_array()) {
37 for (auto& v : j["store_keys"]) r.store_keys.emplace_back(v.get<std::string>());
38 }
39 if (j.contains("bins") && j["bins"].is_array()) {
40 for (auto& pair : j["bins"]) {
41 if (pair.is_array() && pair.size() == 2) {
42 r.bins.emplace_back(pair[0].get<std::string>(), pair[1].get<std::string>());
43 }
44 }
45 }
46 return r;
47}

67std::map<std::string, ComponentRecord> load_installed() {
68 std::map<std::string, ComponentRecord> out;
69 auto p = paths::installed_json_path();
70 std::error_code ec;
71 if (!fs::exists(p, ec)) return out;
72 // installed.json is luban-written so no BOM concern, but routing
73 // through file_util keeps every JSON reader in the codebase using
74 // the same I/O path and immune to user-introduced BOMs (someone
75 // hand-edits to fix something).
76 std::string text = file_util::read_text_no_bom(p);
77 if (text.empty()) return out;
78 json doc;
79 try { doc = json::parse(text); } catch (...) { return out; }
80 if (!doc.contains("components") || !doc["components"].is_object()) return out;
81 for (auto& [name, info] : doc["components"].items()) {
82 if (info.is_object()) out.emplace(name, parse_record(name, info));
83 }
84 return out;
85}

87std::optional<AliasHit> resolve_alias(const std::string& alias) {
88 auto recs = load_installed();
89 AliasHit hit;
90 bool found = false;
91 for (auto& [comp_name, rec] : recs) {
92 for (auto& [a, rel] : rec.bins) {
93 if (a != alias) continue;
94 std::string norm = rel;
95 for (auto& c : norm) if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
96 fs::path exe = paths::toolchain_dir(rec.toolchain_dir) / norm;
97 if (!found) {
98 hit.component = comp_name;
99 hit.alias = a;
100 hit.exe = exe;
101 found = true;
102 }
103 hit.all_components.push_back(comp_name);
104 break; // 同 component 内 alias 不会重，进下一个 component
105 }
106 }
107 if (!found) return std::nullopt;
108 return hit;
109}
