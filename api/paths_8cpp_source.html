27std::string trim(std::string s) {
28 auto not_space = [](unsigned char c) { return c != ' ' && c != '\t' && c != '\r' && c != '\n'; };
29 s.erase(s.begin(), std::find_if(s.begin(), s.end(), not_space));
30 s.erase(std::find_if(s.rbegin(), s.rend(), not_space).base(), s.end());
31 return s;
32}

34std::optional<std::string> raw_env(std::string_view var) {
35#ifdef _WIN32
36 // GetEnvironmentVariableW: handles unicode env values, no MSVC _dupenv_s warning.
37 std::wstring wname = win::from_utf8(var);
38 DWORD n = GetEnvironmentVariableW(wname.c_str(), nullptr, 0);
39 if (n == 0) return std::nullopt;
40 std::wstring buf(n, L'\0');
41 DWORD got = GetEnvironmentVariableW(wname.c_str(), buf.data(), n);
42 if (got == 0) return std::nullopt;
43 buf.resize(got);
44 return win::to_utf8(buf);
45#else
46 std::string name(var);
47 const char* v = std::getenv(name.c_str());
48 if (!v) return std::nullopt;
49 return std::string(v);
50#endif
51}

53// Expand ~ at the start. Mirrors Path.expanduser() — only literal "~" or "~/...".
54fs::path expanduser(const std::string& s) {
55 if (s.empty() || s[0] != '~') return fs::path(s);
56 if (s.size() == 1) return home();
57 if (s[1] == '/' || s[1] == '\\') return home() / fs::path(s.substr(2));
58 return fs::path(s);
59}

104std::optional<fs::path> from_env(std::string_view var) {
105 auto raw = raw_env(var);
106 if (!raw) return std::nullopt;
107 std::string trimmed = trim(*raw);
108 if (trimmed.empty()) return std::nullopt;
109 return expanduser(trimmed);

112fs::path home() {
113#ifdef _WIN32
114 auto userprofile = raw_env("USERPROFILE");
115 if (userprofile && !userprofile->empty()) return fs::path(*userprofile);
116 auto drive = raw_env("HOMEDRIVE");
117 auto path = raw_env("HOMEPATH");
118 if (drive && path) return fs::path(*drive + *path);
119 return known_folder(FOLDERID_Profile);
120#else
121 auto h = raw_env("HOME");
122 if (h && !h->empty()) return fs::path(*h);
123 return fs::path("/");
124#endif
125}

164// luban-managed shim dir (cargo's ~/.cargo/bin/ pattern). One toolchain
165// install can produce hundreds of shims (LLVM-MinGW alone yields ~280 cross-
166// compiler aliases), so dropping them into the shared XDG ~/.local/bin would
167// clobber the volume of files there and stress Defender real-time scanning

221bool same_volume(const fs::path& a, const fs::path& b) {
222#ifdef _WIN32
223 // Plan §7: don't canonicalize — junction may resolve to a different drive
224 // than the user-typed path. Compare root_name() of the literal paths.
225 auto root = [](const fs::path& p) {
226 auto s = p.root_name().string();
227 std::transform(s.begin(), s.end(), s.begin(),
228 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
229 return s;
230 };
231 return root(a) == root(b);
232#else
233 std::error_code ec;
234 auto sa = fs::status(a, ec);
235 auto sb = fs::status(b, ec);
236 if (ec) return false;
237 return sa.permissions() == sa.permissions(); // POSIX dev compare omitted; M2 covers it.
238#endif
239}
