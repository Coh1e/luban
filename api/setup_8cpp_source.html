1// `luban setup` — install toolchain components per selection.json.
2//
3// Orchestration:
4// 1. ensure_dirs
5// 2. deploy seed manifests + selection.json on first run
6// 3. (optional) refresh bucket mirrors — currently per-manifest fetch only
7// 4. install each enabled component
8// 5. summarize failures + exit code
9//
10// Note: pre-v0.2 also wrote activate.{cmd,ps1,sh} into <data>/env. Removed —
11// `luban env --user` (rustup-style) is the canonical PATH integration; CI
12// uses `LUBAN_PREFIX=...` + the same. Activate scripts had no documented
13// users by v0.2.

37const char* error_kind_name(component::ErrorKind k) {
38 switch (k) {
39 case component::ErrorKind::ManifestNotFound: return "manifest-not-found";
40 case component::ErrorKind::UnsafeManifest: return "unsafe-manifest";
41 case component::ErrorKind::IncompleteManifest: return "incomplete-manifest";
42 case component::ErrorKind::DownloadFailed: return "download-failed";
43 case component::ErrorKind::HashMismatch: return "hash-mismatch";
44 case component::ErrorKind::ExtractFailed: return "extract-failed";
45 case component::ErrorKind::Filesystem: return "filesystem";
46 }
47 return "?";
48}

50std::vector<std::string> parse_csv(const std::string& s) {
51 std::vector<std::string> out;
52 size_t start = 0;
53 while (start < s.size()) {
54 size_t end = s.find(',', start);
55 if (end == std::string::npos) end = s.size();
56 std::string name = s.substr(start, end - start);
57 while (!name.empty() && std::isspace(static_cast<unsigned char>(name.front()))) name.erase(name.begin());
58 while (!name.empty() && std::isspace(static_cast<unsigned char>(name.back()))) name.pop_back();
59 if (!name.empty()) out.push_back(name);
60 if (end == s.size()) break;
61 start = end + 1;
62 }
63 return out;
64}

66// Recursively expand `depends:` from each component's manifest, and
67// topologically sort so deps appear before dependents in the install order.
68// Manifests not findable are silently skipped (the install step surfaces a
69// clearer error).
70void expand_depends(std::vector<selection::Entry>& chosen) {
71 // Pass 1: discover transitive dep graph.
72 std::map<std::string, std::vector<std::string>> deps_of;
73 std::vector<std::string> queue;
74 for (auto& e : chosen) queue.push_back(e.name);
75
76 while (!queue.empty()) {
77 std::string name = std::move(queue.back());
78 queue.pop_back();
79 if (deps_of.count(name)) continue;
80 deps_of[name] = {};
81 auto fetched = manifest_source::load(name);
82 if (!fetched) continue;
83 scoop_manifest::ResolvedManifest parsed;
84 try {
85 parsed = scoop_manifest::parse(fetched->manifest, name);
86 } catch (...) { continue; }
87 for (auto& dep : parsed.depends) {
88 deps_of[name].push_back(dep);
89 if (!deps_of.count(dep)) queue.push_back(dep);
90 }
91 }
92
93 // Pass 2: DFS post-order — deps emit before their dependents.
94 std::vector<std::string> sorted;
95 std::set<std::string> visited;
96 auto visit = [&](auto& self, const std::string& n) -> void {
97 if (!visited.insert(n).second) return;
98 auto it = deps_of.find(n);
99 if (it != deps_of.end()) for (auto& d : it->second) self(self, d);
100 sorted.push_back(n);
101 };
102 std::vector<std::string> initial;
103 for (auto& e : chosen) initial.push_back(e.name);
104 for (auto& n : initial) visit(visit, n);
105
106 // Pass 3: rebuild chosen[] in sorted order; preserve enabled/note for
107 // already-listed entries; fabricate enabled=true entries for new deps.
108 std::map<std::string, selection::Entry> by_name;
109 for (auto& e : chosen) by_name[e.name] = e;
110 chosen.clear();
111 for (auto& n : sorted) {
112 auto it = by_name.find(n);
113 if (it != by_name.end()) {
114 chosen.push_back(it->second);
115 } else {
116 log::infof(" pulling in dep '{}' (transitive)", n);
117 selection::Entry e;
118 e.name = n;
119 e.enabled = true;
120 chosen.push_back(e);
121 }
122 }
123}

125int run_setup(const cli::ParsedArgs& a) {
126 log::step("Ensuring canonical directories exist");
127 paths::ensure_dirs();
128 log::ok("directories ready");
129
130 log::step("Deploying seed manifests + selection");
131 auto overlays = selection::deploy_overlays();
132 log::okf("overlays on disk: {}", overlays.size());
133 auto sel = selection::load();
134
135 bool dry_run = false;
136 auto it_dry = a.flags.find("dry-run");
137 if (it_dry != a.flags.end() && it_dry->second) dry_run = true;
138
139 // --with / --without persist into <config>/selection.json BEFORE we build
140 // the chosen set, so future `luban setup` (no flags) keeps the same set.
141 std::vector<std::string> with_names;
142 {
143 auto it = a.opts.find("with");
144 if (it != a.opts.end() && !it->second.empty()) with_names = parse_csv(it->second);
145 }
146 std::vector<std::string> without_names;
147 {
148 auto it = a.opts.find("without");
149 if (it != a.opts.end() && !it->second.empty()) without_names = parse_csv(it->second);
150 }
151 bool selection_changed = false;
152 for (auto& n : with_names) {
153 if (selection::enable(sel, n)) {
154 log::okf("{}enabled '{}' in selection.json",
155 dry_run ? "would have " : "", n);
156 selection_changed = true;
157 }
158 }
159 for (auto& n : without_names) {
160 if (selection::disable(sel, n)) {
161 log::okf("{}disabled '{}' in selection.json",
162 dry_run ? "would have " : "", n);
163 selection_changed = true;
164 }
165 // Symmetric uninstall: --without removes from disk too. Idempotent.
166 if (!dry_run && component::uninstall(n)) {
167 log::okf("uninstalled '{}' (toolchain dir + shims removed)", n);
168 }
169 }
170 if (!dry_run && selection_changed) selection::save(sel);
171
172 // --only filter（可重复使用 -- ）
173 std::vector<std::string> only;
174 auto it_only = a.opts.find("only");
175 if (it_only != a.opts.end() && !it_only->second.empty()) {
176 only = parse_csv(it_only->second);
177 }
178
179 bool force = false;
180 auto it_force = a.flags.find("force");
181 if (it_force != a.flags.end() && it_force->second) force = true;
182
183 auto in_only = [&](const std::string& n) {
184 if (only.empty()) return true;
185 return std::find(only.begin(), only.end(), n) != only.end();
186 };
187
188 std::vector<selection::Entry> chosen;
189 for (auto& list : {std::cref(sel.components), std::cref(sel.extras)}) {
190 for (auto& e : list.get()) {
191 // `--only` 显式选项覆盖 enabled——用户说要装就装。
192 // 没传 --only 时仍尊重 enabled。
193 if (only.empty()) {
194 if (e.enabled) chosen.push_back(e);
195 } else {
196 if (in_only(e.name)) chosen.push_back(e);
197 }
198 }
199 }
200 // 若 --only 指定了 selection 里没有的名字，也尝试装（让 overlay-only 组件能 setup）。
201 if (!only.empty()) {
202 for (auto& n : only) {
203 bool found = false;
204 for (auto& e : chosen) if (e.name == n) { found = true; break; }
205 if (!found) {
206 selection::Entry synthetic;
207 synthetic.name = n;
208 synthetic.enabled = true;
209 chosen.push_back(synthetic);
210 }
211 }
212 }
213 if (chosen.empty()) {
214 log::warn("nothing to install \xe2\x80\x94 selection.json has no enabled entries.");
215 return 0;
216 }
217
218 // Dependency closure: pull in any `depends:` entries declared in each
219 // chosen component's manifest. Common case: emscripten depends on node.
220 expand_depends(chosen);
221
222 log::stepf("Installing {} component(s)", chosen.size());
223 std::vector<std::pair<std::string, std::string>> failures;
224 for (auto& entry : chosen) {
225 log::stepf("\xe2\x86\x92 {}", entry.name);
226 if (dry_run) {
227 log::infof(" would install {} (dry-run; skipping fetch+download+install)", entry.name);
228 continue;
229 }
230 auto rc = component::install(entry.name, force);
231 if (!rc) {
232 log::errf(" install of {} failed: {}: {}",
233 entry.name, error_kind_name(rc.error().kind), rc.error().message);
234 failures.emplace_back(entry.name, rc.error().message);
235 }
236 }
237
238 if (!failures.empty()) {
239 log::warnf("{} component(s) failed:", failures.size());
240 for (auto& [n, msg] : failures) log::warnf(" - {}: {}", n, msg);
241 return 1;
242 }
243
244 log::ok("setup complete");
245 log::infof("shims written to {}", paths::bin_dir().string());
246 log::info("run `luban env --user` to register on HKCU PATH (rustup-style).");
247 return 0;
248}

252void register_setup() {
253 cli::Subcommand c;
254 c.name = "setup";
255 c.help = "install toolchain (LLVM-MinGW + cmake + ninja + mingit + vcpkg)";
256 c.group = "setup";
257 // Manual-grade --help: `docs/src/commands/setup.md` is embedded at build
258 // time (cmake/embed_text.cmake). Single source of truth for the mdBook
259 // chapter and the in-binary --help.
260 c.long_help = embedded_help::setup_help;
261 c.opts = {{"only", ""}, {"with", ""}, {"without", ""}};
262 // --refresh-buckets removed in v0.2 (bucket_sync gone — manifests come
263 // from manifests_seed/ overlay only; nothing to refresh from network).
264 c.flags = {"dry-run", "force"};
265 c.examples = {
266 "luban setup\tInstall all enabled components",
267 "luban setup --only vcpkg\tInstall just vcpkg (overrides selection.enabled)",
268 "luban setup --only ninja --force\tForce reinstall of ninja",
269 "luban setup --dry-run\tShow what would be installed",
270 "luban setup --with emscripten\tInstall emscripten + deps; persist in selection.json",
271 "luban setup --without zig\tDisable + uninstall zig",
272 };
273 c.run = run_setup;
274 cli::register_subcommand(std::move(c));
275}
