28// 拒绝列表：这些是系统层 toolchain，不该进 vcpkg.json。
29constexpr std::array<const char*, 14> kSystemTools = {
30 "cmake", "ninja", "clang", "clang++", "clangd", "clang-format", "clang-tidy",
31 "lld", "llvm-mingw", "mingit", "git", "vcpkg", "make", "gcc",
32};

34bool is_system_tool(const std::string& name) {
35 for (auto* t : kSystemTools) if (name == t) return true;
36 return false;
37}

39// 找项目根：cwd 向上一直到根，第一个含 vcpkg.json 的目录。
40// 找不到 = cwd 本身（让用户在 cwd 起一个新项目）。
41fs::path find_project_root() {
42 std::error_code ec;
43 fs::path d = fs::current_path(ec);
44 while (!d.empty()) {
45 if (fs::exists(d / "vcpkg.json", ec)) return d;
46 fs::path parent = d.parent_path();
47 if (parent == d) break;
48 d = parent;
49 }
50 return fs::current_path(ec);
51}

53void regenerate(const fs::path& project_dir) {
54 auto targets = luban_cmake_gen::read_targets_from_cmake(project_dir);
55 if (targets.empty()) {
56 // 兜底：从项目名推一个
57 targets.push_back(project_dir.filename().string());
58 }
59 luban_cmake_gen::regenerate_in_project(project_dir, targets);
60}

62int run_add(const cli::ParsedArgs& a) {
63 if (a.positional.empty()) {
64 log::err("usage: luban add <pkg>[@version]");
65 return 2;
66 }
67 auto [pkg, ver] = vcpkg_manifest::parse_pkg_spec(a.positional[0]);
68 if (pkg.empty()) {
69 log::err("empty package name");
70 return 2;
71 }
72 if (is_system_tool(pkg)) {
73 log::errf("'{}' is a system-level toolchain, not a vcpkg library.", pkg);
74 log::infof("System tools live in <data>/toolchains/. Use:");
75 log::infof(" luban setup --only {}", pkg);
76 return 2;
77 }
78
79 fs::path proj = find_project_root();
80 fs::path vcpkg_path = proj / "vcpkg.json";
81
82 auto m = vcpkg_manifest::load(vcpkg_path, proj.filename().string());
83 vcpkg_manifest::add(m, pkg, ver);
84 vcpkg_manifest::save(vcpkg_path, m);
85 log::okf("added {} to vcpkg.json{}", pkg, ver.empty() ? "" : " (>= " + ver + ")");
86
87 regenerate(proj);
88 log::okf("regenerated luban.cmake");
89
90 log::info("next: luban build (cmake will fetch the dep via vcpkg manifest mode)");
91 log::info("note: ensure VCPKG_ROOT is set + CMakePresets has CMAKE_TOOLCHAIN_FILE");
92 return 0;
93}

95int run_remove(const cli::ParsedArgs& a) {
96 if (a.positional.empty()) {
97 log::err("usage: luban remove <pkg>");
98 return 2;
99 }
100 const std::string& pkg = a.positional[0];
101
102 fs::path proj = find_project_root();
103 fs::path vcpkg_path = proj / "vcpkg.json";
104 auto m = vcpkg_manifest::load(vcpkg_path, proj.filename().string());
105 if (!vcpkg_manifest::remove(m, pkg)) {
106 log::warnf("'{}' was not in vcpkg.json (no change)", pkg);
107 return 0;
108 }
109 vcpkg_manifest::save(vcpkg_path, m);
110 log::okf("removed {} from vcpkg.json", pkg);
111
112 regenerate(proj);
113 log::okf("regenerated luban.cmake");
114 return 0;
115}

117int run_sync(const cli::ParsedArgs&) {
118 fs::path proj = find_project_root();
119 log::stepf("syncing luban.cmake in {}", proj.string());
120 regenerate(proj);
121 log::ok("regenerated luban.cmake from vcpkg.json + luban.toml");
122 return 0;
123}

127void register_add() {
128 cli::Subcommand c;
129 c.name = "add";
130 c.help = "add a vcpkg library (edits vcpkg.json + luban.cmake)";
131 c.group = "dep";
132 c.long_help =
133 " Add a vcpkg port to the current project's vcpkg.json and regenerate\n"
134 " luban.cmake (find_package + target_link_libraries auto-wired).\n"
135 "\n"
136 " Version constraint: `luban add fmt@10` → vcpkg `version>=: 10.0.0`.\n"
137 "\n"
138 " System tools (cmake, ninja, clang, ...) are rejected — those are\n"
139 " managed via `luban setup`, not vcpkg.";
140 c.n_positional = 1;
141 c.positional_names = {"pkg[@version]"};
142 c.examples = {
143 "luban add fmt\tAdd fmt at baseline version",
144 "luban add spdlog@1.13\tAdd spdlog with version>=1.13.0",
145 "luban add boost-asio\tAdd Boost.Asio (Boost::asio target)",
146 };
147 c.run = run_add;
148 cli::register_subcommand(std::move(c));
149}
