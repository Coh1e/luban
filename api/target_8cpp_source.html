27fs::path find_project_root() {
28 std::error_code ec;
29 fs::path d = fs::current_path(ec);
30 while (!d.empty()) {
31 if (fs::exists(d / "luban.cmake", ec)) return d;
32 fs::path parent = d.parent_path();
33 if (parent == d) break;
34 d = parent;
35 }
36 return fs::current_path(ec);
37}

46// 生成 src/<name>/CMakeLists.txt + 源文件骨架。
47void scaffold_target_dir(const fs::path& project, const std::string& name, const std::string& kind) {
48 fs::path tdir = project / "src" / name;
49 std::error_code ec;
50 fs::create_directories(tdir, ec);
51
52 if (kind == "lib") {
53 std::ostringstream cml;
54 cml << "add_library(" << name << " STATIC " << name << ".cpp)\n"
55 << "target_include_directories(" << name << " PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})\n"
56 << "luban_apply(" << name << ")\n";
57 write_text(tdir / "CMakeLists.txt", cml.str());
58
59 std::ostringstream h;
60 h << "#pragma once\n"
61 << "namespace " << name << " { int hello(); }\n";
62 write_text(tdir / (name + ".h"), h.str());
63
64 std::ostringstream cpp;
65 cpp << "#include \"" << name << ".h\"\n"
66 << "namespace " << name << " { int hello() { return 42; } }\n";
67 write_text(tdir / (name + ".cpp"), cpp.str());
68 } else {
69 // exe
70 std::ostringstream cml;
71 cml << "add_executable(" << name << " main.cpp)\n"
72 << "luban_apply(" << name << ")\n";
73 write_text(tdir / "CMakeLists.txt", cml.str());
74
75 std::ostringstream main;
76 main << "#include <print>\n\n"
77 << "int main() {\n"
78 << " std::println(\"hello from " << name << "!\");\n"
79 << " return 0;\n"
80 << "}\n";
81 write_text(tdir / "main.cpp", main.str());
82 }
83}

85int run_target(const cli::ParsedArgs& a) {
86 if (a.positional.empty()) {
87 log::err("usage: luban target add {lib|exe} <name> | luban target remove <name>");
88 return 2;
89 }
90 const std::string& sub = a.positional[0];
91
92 fs::path proj = find_project_root();
93 if (!fs::exists(proj / "luban.cmake")) {
94 log::errf("not in a luban project (no luban.cmake found from {})", fs::current_path().string());
95 return 2;
96 }
97
98 auto targets = luban_cmake_gen::read_targets_from_cmake(proj);
99
100 if (sub == "add") {
101 if (a.positional.size() < 3) {
102 log::err("usage: luban target add {lib|exe} <name>");
103 return 2;
104 }
105 const std::string& kind = a.positional[1];
106 const std::string& name = a.positional[2];
107 if (kind != "lib" && kind != "exe") {
108 log::errf("unsupported kind '{}' (use lib or exe)", kind);
109 return 2;
110 }
111 if (!valid_name(name)) {
112 log::errf("invalid target name '{}'", name);
113 return 2;
114 }
115 if (std::find(targets.begin(), targets.end(), name) != targets.end()) {
116 log::errf("target '{}' already exists in LUBAN_TARGETS", name);
117 return 2;
118 }
119 std::error_code ec;
120 if (fs::exists(proj / "src" / name, ec)) {
121 log::errf("src/{} already exists; refusing to overwrite", name);
122 return 2;
123 }
124 scaffold_target_dir(proj, name, kind);
125 targets.push_back(name);
126 luban_cmake_gen::regenerate_in_project(proj, targets);
127 log::okf("created {} target '{}' at src/{}/", kind, name, name);
128 log::okf("regenerated luban.cmake (LUBAN_TARGETS now: {})",
129 [&]{
130 std::string s;
131 for (size_t i = 0; i < targets.size(); ++i) { if (i) s += ", "; s += targets[i]; }
132 return s;
133 }());
134 return 0;
135 }
136
137 if (sub == "remove") {
138 if (a.positional.size() < 2) {
139 log::err("usage: luban target remove <name>");
140 return 2;
141 }
142 const std::string& name = a.positional[1];
143 auto it = std::find(targets.begin(), targets.end(), name);
144 if (it == targets.end()) {
145 log::warnf("target '{}' not in LUBAN_TARGETS", name);
146 return 0;
147 }
148 targets.erase(it);
149 luban_cmake_gen::regenerate_in_project(proj, targets);
150 log::okf("removed '{}' from LUBAN_TARGETS", name);
151 std::error_code ec;
152 if (fs::exists(proj / "src" / name, ec)) {
153 log::warnf("src/{}/ left in place; rm manually if you really want it gone", name);
154 }
155 return 0;
156 }
157
158 log::errf("unknown target subcommand '{}' (use add or remove)", sub);
159 return 2;
160}

164void register_target() {
165 cli::Subcommand c;
166 c.name = "target";
167 c.help = "add/remove a build target (lib or exe) within the current project";
168 c.group = "project";
169 c.long_help =
170 " `luban target add lib <name>` creates src/<name>/{<name>.h, <name>.cpp,\n"
171 " CMakeLists.txt} and registers it.\n"
172 " `luban target add exe <name>` creates src/<name>/{main.cpp, CMakeLists.txt}\n"
173 " `luban target remove <name>` unregisters; src/<name>/ is left in place.\n"
174 "\n"
175 " Inter-target linking is plain cmake — write\n"
176 " target_link_libraries(<exe> PRIVATE <lib>)\n"
177 " in src/<exe>/CMakeLists.txt yourself. Luban does not abstract this.";
178 c.n_positional = 1;
179 c.positional_names = {"sub", "kind?", "name?"};
180 c.examples = {
181 "luban target add lib mathlib\tNew static lib with mathlib::hello() stub",
182 "luban target add exe bench\tNew executable target",
183 "luban target remove temp\tUnregister (does NOT delete src/temp/)",
184 };
185 c.run = run_target;
186 cli::register_subcommand(std::move(c));
187}
