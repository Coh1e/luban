21// Accepted C++ standards. We deliberately keep this narrow:
22// 17 — current floor, most vcpkg ports compile against it
23// 20 — modules / coroutines / concepts territory
24// 23 — luban's own default; std::format / std::expected / ranges-v3 baked in
25// 14/11 are too old to be worth supporting from `luban new`. 26 isn't yet
26// universally available across MinGW/clang releases — revisit when the
27// compiler we ship lands -std=c++26 by default.
28bool cpp_supported(int n) {
29 return n == 17 || n == 20 || n == 23;
30}

32int validate_cpp(int requested, std::string_view source_label) {
33 if (cpp_supported(requested)) return requested;
34 log::warnf("luban.toml [project] cpp = {} is not supported (accepted: 17, 20, 23).",
35 requested);
36 log::infof("falling back to cpp = 23 (was: {} from {}).", requested, source_label);
37 return 23;
38}

40void load_from_table(const ::toml::table& tbl, Config& out) {
41 if (auto* proj = tbl["project"].as_table()) {
42 if (auto v = (*proj)["default_preset"].value<std::string>())
43 out.project.default_preset = *v;
44 if (auto v = (*proj)["triplet"].value<std::string>())
45 out.project.triplet = *v;
46 // Accept both `cpp = 23` (integer) and `cpp = "23"` (string). The
47 // string form is forgiving for users who copy-paste from the docs.
48 if (auto v = (*proj)["cpp"].value<int64_t>()) {
49 out.project.cpp = validate_cpp(static_cast<int>(*v), "[project] cpp (integer)");
50 }
51 if (auto v = (*proj)["cpp"].value<std::string>()) {
52 try {
53 out.project.cpp = validate_cpp(std::stoi(*v), "[project] cpp (string)");
54 } catch (...) {
55 log::warnf("luban.toml [project] cpp = \"{}\" is not a valid integer; "
56 "falling back to cpp = 23.", *v);
57 out.project.cpp = 23;
58 }
59 }
60 }
61 if (auto* scaf = tbl["scaffold"].as_table()) {
62 if (auto v = (*scaf)["warnings"].value<std::string>())
63 out.scaffold.warnings = parse_warnings(*v);
64 if (auto* arr = (*scaf)["sanitizers"].as_array()) {
65 for (auto& el : *arr) {
66 if (auto s = el.value<std::string>()) {
67 out.scaffold.sanitizers.push_back(*s);
68 }
69 }
70 }
71 }

73 // [toolchain] — flat string→string map of component → required version.
74 // Per OQ-2, this is opt-in: missing section = no pins. Each key parses
75 // as a string (we accept integer values too, for cases like cmake = 423
76 // typo — coerce to "423"). Unknown component names are accepted silently;
77 // the runtime check (commands/build_project.cpp) maps them against the
78 // installed registry and warns there.
79 if (auto* tc = tbl["toolchain"].as_table()) {
80 for (auto&& [key, node] : *tc) {
81 // Explicit is_string / is_integer checks — toml++'s
82 // value<int64_t>() will silently coerce a boolean to 0/1, which
83 // is not a sensible version pin. Same for arrays/tables/dates.
84 std::string version_str;
85 if (node.is_string()) {
86 version_str = *node.value<std::string>();
87 } else if (node.is_integer()) {
88 version_str = std::to_string(*node.value<int64_t>());
89 } else {
90 continue; // silently skip bool/array/table/date/float
91 }
92 if (version_str.empty()) continue;
93 out.toolchain[std::string(key.str())] = std::move(version_str);
94 }
95 }
96}

100Config load(const fs::path& path) {
101 Config out;
102 std::error_code ec;
103 if (!fs::exists(path, ec)) return out;
104
105 try {
106 ::toml::table tbl = ::toml::parse_file(path.string());
107 load_from_table(tbl, out);
108 } catch (const ::toml::parse_error& e) {
109 log::warnf("luban.toml parse error at {}: {}", path.string(), e.what());
110 }
111 return out;
112}
