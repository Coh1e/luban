17// "10.2" → "10.2.0" — vcpkg 的 version>= 期望 X.Y.Z 三段；不全则补 0。
18// 已是三段或更多段（含 - 后缀）原样返回。
19std::string normalize_version_ge(const std::string& v) {
20 if (v.empty()) return v;
21 int dots = 0;
22 for (char c : v) if (c == '.') ++dots;
23 std::string out = v;
24 while (dots < 2) { out += ".0"; ++dots; }
25 return out;
26}

30Manifest load(const fs::path& path, const std::string& fallback_name) {
31 Manifest m;
32 m.name = fallback_name;
33
34 std::error_code ec;
35 if (!fs::exists(path, ec)) return m;
36
37 // BOM-strip via file_util — VS Code saves vcpkg.json with UTF-8 BOM
38 // by default, and nlohmann::json's stream operator rejects raw BOM.
39 // Without this, a user's first VS Code edit silently breaks `luban add`.
40 std::string text = file_util::read_text_no_bom(path);
41 if (text.empty()) return m;
42
43 json doc;
44 try { doc = json::parse(text); } catch (...) { return m; }
45 if (!doc.is_object()) return m;
46
47 if (doc.contains("name") && doc["name"].is_string()) {
48 m.name = doc["name"].get<std::string>();
49 }
50 if (doc.contains("version") && doc["version"].is_string()) {
51 m.version = doc["version"].get<std::string>();
52 } else if (doc.contains("version-string") && doc["version-string"].is_string()) {
53 // 旧 manifest 用 version-string；统一折成 version
54 m.version = doc["version-string"].get<std::string>();
55 }
56 if (doc.contains("dependencies") && doc["dependencies"].is_array()) {
57 for (auto& d : doc["dependencies"]) {
58 Dependency dep;
59 if (d.is_string()) {
60 dep.name = d.get<std::string>();
61 } else if (d.is_object() && d.contains("name") && d["name"].is_string()) {
62 dep.name = d["name"].get<std::string>();
63 if (d.contains("version>=") && d["version>="].is_string()) {
64 dep.version_ge = d["version>="].get<std::string>();
65 }
66 } else {
67 continue;
68 }
69 m.dependencies.push_back(std::move(dep));
70 }
71 }
72
73 // 保留所有未识别字段（features / overrides / supports / builtin-baseline / 等）
74 json extras = json::object();
75 for (auto& [k, v] : doc.items()) {
76 if (k == "name" || k == "version" || k == "version-string" || k == "dependencies") continue;
77 extras[k] = v;
78 }
79 m.extras = std::move(extras);
80 return m;
81}

83void save(const fs::path& path, const Manifest& m) {
84 json doc = json::object();
85 doc["name"] = m.name;
86 doc["version"] = m.version;
87
88 json deps = json::array();
89 for (auto& d : m.dependencies) {
90 if (d.version_ge && !d.version_ge->empty()) {
91 json obj = json::object();
92 obj["name"] = d.name;
93 obj["version>="] = normalize_version_ge(*d.version_ge);
94 deps.push_back(obj);
95 } else {
96 deps.push_back(d.name);
97 }
98 }
99 doc["dependencies"] = deps;
100
101 if (m.extras.is_object()) {
102 for (auto& [k, v] : m.extras.items()) doc[k] = v;
103 }
104
105 // Atomic write via file_util — same tmp+rename + cross-volume fallback
106 // logic as the previous inline code, just shared with other manifest
107 // writers (selection.cpp, registry.cpp soon to follow).
108 std::string text = doc.dump(2);
109 text.push_back('\n'); // trailing newline matches prior behaviour
110 file_util::write_text_atomic(path, text);
111}

113void add(Manifest& m, const std::string& pkg, const std::string& version_ge) {
114 auto it = std::find_if(m.dependencies.begin(), m.dependencies.end(),
115 [&](const Dependency& d) { return d.name == pkg; });
116 Dependency dep;
117 dep.name = pkg;
118 if (!version_ge.empty()) dep.version_ge = version_ge;
119 if (it != m.dependencies.end()) {
120 *it = std::move(dep);
121 } else {
122 m.dependencies.push_back(std::move(dep));
123 }
124}

126bool remove(Manifest& m, const std::string& pkg) {
127 auto it = std::find_if(m.dependencies.begin(), m.dependencies.end(),
128 [&](const Dependency& d) { return d.name == pkg; });
129 if (it == m.dependencies.end()) return false;
130 m.dependencies.erase(it);
131 return true;
132}
