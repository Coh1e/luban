47std::string to_hex(const unsigned char* data, size_t n) {
48 static constexpr char kHex[] = "0123456789abcdef";
49 std::string out(n * 2, '\0');
50 for (size_t i = 0; i < n; ++i) {
51 out[2 * i] = kHex[(data[i] >> 4) & 0xF];
52 out[2 * i + 1] = kHex[data[i] & 0xF];
53 }
54 return out;

69std::optional<Algorithm> parse_algo(std::string_view name) {
70 std::string s(name);
71 std::transform(s.begin(), s.end(), s.begin(),
72 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
73 if (s == "sha256") return Algorithm::Sha256;
74 if (s == "sha512") return Algorithm::Sha512;

80std::optional<HashSpec> parse(std::string_view raw) {
81 // 去首尾 ws
82 size_t b = 0;
83 size_t e = raw.size();
84 while (b < e && std::isspace(static_cast<unsigned char>(raw[b]))) ++b;
85 while (e > b && std::isspace(static_cast<unsigned char>(raw[e-1]))) --e;
86 if (b == e) return std::nullopt;
87 std::string trimmed(raw.substr(b, e - b));
88
89 auto colon = trimmed.find(':');
90 Algorithm algo = Algorithm::Sha256;
91 std::string hex;
92 if (colon != std::string::npos) {
93 auto a = parse_algo(std::string_view(trimmed).substr(0, colon));
94 if (!a) return std::nullopt;
95 algo = *a;
96 hex = trimmed.substr(colon + 1);
97 } else {
98 hex = trimmed;
99 }
100 // 验证全 hex
101 if (hex.empty()) return std::nullopt;
102 for (char c : hex) {
103 if (!std::isxdigit(static_cast<unsigned char>(c))) return std::nullopt;
104 }
105 std::transform(hex.begin(), hex.end(), hex.begin(),
106 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
107 return HashSpec{algo, hex};
108}

114std::optional<HashSpec> hash_file(const fs::path& path, Algorithm algo) {
115#ifdef _WIN32
116 AlgorithmHandle alg;
117 if (BCryptOpenAlgorithmProvider(&alg.h, bcrypt_algo_id(algo), nullptr, 0) != 0) {
118 return std::nullopt;
119 }
120
121 DWORD digest_size = 0;
122 DWORD got = 0;
123 if (BCryptGetProperty(alg.h, BCRYPT_HASH_LENGTH,
124 reinterpret_cast<PUCHAR>(&digest_size), sizeof(digest_size),
125 &got, 0) != 0) {
126 return std::nullopt;
127 }
128
129 HashHandle hash;
130 if (BCryptCreateHash(alg.h, &hash.h, nullptr, 0, nullptr, 0, 0) != 0) {
131 return std::nullopt;
132 }
133
134 std::FILE* fp = nullptr;
135#ifdef _MSC_VER
136 if (_wfopen_s(&fp, path.wstring().c_str(), L"rb") != 0) fp = nullptr;
137#else
138 fp = _wfopen(path.wstring().c_str(), L"rb");
139#endif
140 if (!fp) return std::nullopt;
141
142 std::vector<unsigned char> buf(kChunk);
143 while (true) {
144 size_t n = std::fread(buf.data(), 1, buf.size(), fp);
145 if (n == 0) break;
146 if (BCryptHashData(hash.h, buf.data(), static_cast<ULONG>(n), 0) != 0) {
147 std::fclose(fp);
148 return std::nullopt;
149 }
150 if (n < buf.size()) break;
151 }
152 std::fclose(fp);
153
154 std::vector<unsigned char> digest(digest_size);
155 if (BCryptFinishHash(hash.h, digest.data(), digest_size, 0) != 0) {
156 return std::nullopt;
157 }
158 return HashSpec{algo, to_hex(digest.data(), digest.size())};
159#else
160 // POSIX backend: OpenSSL EVP. ADR-0006 Phase B — system libssl is the
161 // pragmatic choice (Ubuntu/Debian/Fedora/Homebrew all ship it; pure-
162 // header alternatives like mbedtls would mean vendoring ~5k lines).
163 const EVP_MD* md = nullptr;
164 switch (algo) {
165 case Algorithm::Sha256: md = EVP_sha256(); break;
166 case Algorithm::Sha512: md = EVP_sha512(); break;
167 case Algorithm::Sha1: md = EVP_sha1(); break;
168 case Algorithm::Md5: md = EVP_md5(); break;
169 }
170 if (!md) return std::nullopt;
171
172 EVP_MD_CTX* ctx = EVP_MD_CTX_new();
173 if (!ctx) return std::nullopt;
174 if (EVP_DigestInit_ex(ctx, md, nullptr) != 1) {
175 EVP_MD_CTX_free(ctx);
176 return std::nullopt;
177 }
178
179 std::FILE* fp = std::fopen(path.string().c_str(), "rb");
180 if (!fp) {
181 EVP_MD_CTX_free(ctx);
182 return std::nullopt;
183 }
184
185 std::vector<unsigned char> buf(kChunk);
186 while (true) {
187 size_t n = std::fread(buf.data(), 1, buf.size(), fp);
188 if (n == 0) break;
189 if (EVP_DigestUpdate(ctx, buf.data(), n) != 1) {
190 std::fclose(fp);
191 EVP_MD_CTX_free(ctx);
192 return std::nullopt;
193 }
194 if (n < buf.size()) break;
195 }
196 std::fclose(fp);
197
198 unsigned char digest[EVP_MAX_MD_SIZE];
199 unsigned int digest_len = 0;
200 if (EVP_DigestFinal_ex(ctx, digest, &digest_len) != 1) {
201 EVP_MD_CTX_free(ctx);
202 return std::nullopt;
203 }
204 EVP_MD_CTX_free(ctx);
205 return HashSpec{algo, to_hex(digest, digest_len)};
206#endif
207}
