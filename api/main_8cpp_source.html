1// luban-shim.exe — rustup-style native exe proxy.
2//
3// This is a separate executable (not part of luban.exe) that gets hard-linked
4// into <data>/bin/<alias>.exe for every alias in installed.json. At runtime it:
5//
6// 1. GetModuleFileNameW → resolves its own absolute path (whichever name
7// the user invoked it as)
8// 2. Strips dir + .exe → "alias"
9// 3. Reads <self_dir>/.shim-table.json (a flat alias→absolute-path map
10// written by `luban shim`)
11// 4. CreateProcessW to that target with the rest of argv preserved
12// 5. Waits for child + propagates exit code
13//
14// Why this exists: `cmake.cmd` shims work for shells but cmake's compiler
15// probe iterates PATH looking for `clang.exe` etc. (literally a .exe). With
16// hard-linked native shims, we look like a real `clang.exe` and probes succeed.
17//
18// Design constraints:
19// - tiny binary (~200 KB static-linked); we don't link luban's own .cpp
20// modules so the shim stays compact
21// - no stdout chatter; errors go to stderr only on failure
22// - no allocations on the happy path beyond what nlohmann/json forces

42std::string utf8(const std::wstring& w) {
43 if (w.empty()) return {};
44 int n = WideCharToMultiByte(CP_UTF8, 0, w.data(), static_cast<int>(w.size()),
45 nullptr, 0, nullptr, nullptr);
46 std::string s(static_cast<size_t>(n), '\0');
47 WideCharToMultiByte(CP_UTF8, 0, w.data(), static_cast<int>(w.size()),
48 s.data(), n, nullptr, nullptr);
49 return s;
50}

52std::wstring wide(std::string_view s) {
53 if (s.empty()) return {};
54 int n = MultiByteToWideChar(CP_UTF8, 0, s.data(), static_cast<int>(s.size()),
55 nullptr, 0);
56 std::wstring w(static_cast<size_t>(n), L'\0');
57 MultiByteToWideChar(CP_UTF8, 0, s.data(), static_cast<int>(s.size()),
58 w.data(), n);
59 return w;
60}

62// Quote argv element per CommandLineToArgvW rules. Same as luban's proc.cpp.
63std::wstring quote_arg(const std::wstring& a) {
64 if (!a.empty() && a.find_first_of(L" \t\"") == std::wstring::npos) return a;
65 std::wstring out = L"\"";
66 int backslashes = 0;
67 for (wchar_t c : a) {
68 if (c == L'\\') { ++backslashes; continue; }
69 if (c == L'"') {
70 out.append(static_cast<size_t>(backslashes) * 2 + 1, L'\\');
71 backslashes = 0;
72 out.push_back(L'"');
73 continue;
74 }
75 out.append(static_cast<size_t>(backslashes), L'\\');
76 backslashes = 0;
77 out.push_back(c);
78 }
79 out.append(static_cast<size_t>(backslashes) * 2, L'\\');
80 out.push_back(L'"');
81 return out;
82}

91// MinGW provides wmain via -municode link, but luban-shim isn't expected to
92// face Unicode-named arguments often. Use the standard main() and convert via
93// GetCommandLineW for consistency with luban.exe itself.
94int main(int /*argc*/, char** /*argv*/) {
95 // 1. self path
96 wchar_t self_buf[MAX_PATH * 8];
97 DWORD got = GetModuleFileNameW(nullptr, self_buf, static_cast<DWORD>(std::size(self_buf)));
98 if (got == 0 || got >= std::size(self_buf)) die("GetModuleFileNameW failed");
99 fs::path self(std::wstring(self_buf, got));
100
101 // alias = filename stem (no .exe)
102 std::wstring alias_w = self.stem().wstring();
103 std::string alias = utf8(alias_w);
104
105 // 2. find sibling .shim-table.json
106 fs::path table_path = self.parent_path() / ".shim-table.json";
107 std::ifstream in(table_path, std::ios::binary);
108 if (!in) {
109 std::fprintf(stderr,
110 "luban-shim: could not open %s\n"
111 " hint: run `luban shim` to (re)generate the shim table.\n",
112 utf8(table_path.wstring()).c_str());
113 return 127;
114 }
115 json table;
116 try {
117 in >> table;
118 } catch (const std::exception& e) {
119 std::fprintf(stderr, "luban-shim: parse %s: %s\n",
120 utf8(table_path.wstring()).c_str(), e.what());
121 return 127;
122 }
123 if (!table.is_object() || !table.contains(alias)) {
124 std::fprintf(stderr, "luban-shim: alias '%s' not in shim table\n", alias.c_str());
125 return 127;
126 }
127 std::string target = table[alias].get<std::string>();
128 std::wstring target_w = wide(target);
129
130 // 3. preserve original commandline tail (everything after argv[0])
131 LPWSTR raw_cmdline = GetCommandLineW();
132 int n_args = 0;
133 LPWSTR* argv_w = CommandLineToArgvW(raw_cmdline, &n_args);
134 if (!argv_w) die("CommandLineToArgvW failed");
135
136 std::wstring cmdline = quote_arg(target_w);
137 for (int i = 1; i < n_args; ++i) {
138 cmdline.push_back(L' ');
139 cmdline += quote_arg(argv_w[i]);
140 }
141 LocalFree(argv_w);
142
143 // 4. spawn target, inherit handles + cwd + env
144 STARTUPINFOW si{};
145 si.cb = sizeof(si);
146 PROCESS_INFORMATION pi{};
147 std::vector<wchar_t> mut(cmdline.begin(), cmdline.end());
148 mut.push_back(L'\0');
149
150 if (!CreateProcessW(target_w.c_str(), mut.data(), nullptr, nullptr, TRUE,
151 0, nullptr, nullptr, &si, &pi)) {
152 std::fprintf(stderr, "luban-shim: CreateProcessW failed for %s (err=%lu)\n",
153 target.c_str(), GetLastError());
154 return 127;
155 }
156
157 WaitForSingleObject(pi.hProcess, INFINITE);
158 DWORD code = 0;
159 GetExitCodeProcess(pi.hProcess, &code);
160 CloseHandle(pi.hThread);
161 CloseHandle(pi.hProcess);
162 return static_cast<int>(code);
163}
