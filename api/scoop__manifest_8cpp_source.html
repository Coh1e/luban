37std::string trim(std::string s) {
38 auto not_ws = [](unsigned char c) { return !std::isspace(c); };
39 s.erase(s.begin(), std::find_if(s.begin(), s.end(), not_ws));
40 s.erase(std::find_if(s.rbegin(), s.rend(), not_ws).base(), s.end());
41 return s;
42}

50void check_no_denied(const json& obj, const std::string& name, const std::string& where) {
51 std::vector<std::string> bad;
52 for (auto& field : kDeniedFields) {
53 if (obj.contains(field)) bad.emplace_back(field);
54 }
55 if (!bad.empty()) {
56 std::sort(bad.begin(), bad.end());
57 std::string joined;
58 for (size_t i = 0; i < bad.size(); ++i) {
59 if (i) joined += ", ";
60 joined += bad[i];
61 }
62 std::string msg = std::format(
63 "manifest '{}'{} uses fields we refuse to execute: {}. "
64 "Provide an overlay manifest with a plain url+hash instead.",
65 name, where.empty() ? "" : (" (" + where + ")"), joined);
66 throw UnsafeManifest(msg);
67 }
68}

70// Scoop 的 url / hash 都允许字符串或单元素列表；多元素列表暂不支持，
71// 让用户写 overlay 改成单一 url。
72std::string normalize_str_or_singleton_list(const json& raw, const std::string& name,
73 const char* what) {
74 if (raw.is_array()) {
75 if (raw.size() != 1) {
76 throw IncompleteManifest(std::format(
77 "manifest '{}' has multi-{} list; needs overlay", name, what));
78 }
79 return raw[0].is_string() ? raw[0].get<std::string>() : "";
80 }
81 if (!raw.is_string()) {
82 throw IncompleteManifest(std::format(
83 "manifest '{}' has empty {}", name, what));
84 }
85 return raw.get<std::string>();
86}

88std::string normalize_hash(const json& raw, const std::string& name) {
89 std::string s = trim(normalize_str_or_singleton_list(raw, name, "hash"));
90 if (s.empty()) throw IncompleteManifest(std::format("manifest '{}' has empty hash", name));
91 if (s.find(':') == std::string::npos) s = "sha256:" + s;
92 return lower(std::move(s));
93}

95std::string normalize_url(const json& raw, const std::string& name) {
96 std::string s = trim(normalize_str_or_singleton_list(raw, name, "url"));
97 if (s.empty()) throw IncompleteManifest(std::format("manifest '{}' has empty url", name));
98 // Scoop 允许 "url.zip#/dest.zip" 重命名后缀——我们接受但忽略。
99 auto hash = s.find('#');
100 if (hash != std::string::npos) s = s.substr(0, hash);
101 return trim(std::move(s));
102}

110std::optional<BinEntry> coerce_bin_entry(const json& v) {
111 if (v.is_string()) {
112 std::string rel = v.get<std::string>();
113 return BinEntry{rel, fs::path(rel).stem().string(), {}};
114 }
115 if (v.is_array() && !v.empty()) {
116 if (!v[0].is_string()) return std::nullopt;
117 std::string rel = v[0].get<std::string>();
118 std::string alias = (v.size() > 1 && v[1].is_string())
119 ? v[1].get<std::string>()
120 : fs::path(rel).stem().string();
121 std::vector<std::string> prefix;
122 if (v.size() > 2 && v[2].is_string()) {
123 std::string s = trim(v[2].get<std::string>());
124 if (!s.empty()) {
125 // 按空白切分（与 Python str.split() 等价）
126 size_t start = 0;
127 while (start < s.size()) {
128 while (start < s.size() && std::isspace(static_cast<unsigned char>(s[start]))) ++start;
129 size_t end = start;
130 while (end < s.size() && !std::isspace(static_cast<unsigned char>(s[end]))) ++end;
131 if (start < end) prefix.push_back(s.substr(start, end - start));
132 start = end;
133 }
134 }
135 }
136 return BinEntry{rel, alias, prefix};
137 }
138 return std::nullopt;
139}

141std::vector<BinEntry> coerce_bins(const json& raw) {
142 std::vector<BinEntry> out;
143 if (raw.is_null()) return out;
144 if (raw.is_string()) {
145 if (auto e = coerce_bin_entry(raw)) out.push_back(*e);
146 return out;
147 }
148 if (!raw.is_array()) return out;
149 // bin 可以是 [string, ...] 或 [[rel,alias], ...]——按第一个元素类型决定。
150 bool is_list_of_lists = !raw.empty() && raw[0].is_array();
151 if (is_list_of_lists) {
152 for (auto& item : raw) {
153 if (auto e = coerce_bin_entry(item)) out.push_back(*e);
154 }
155 } else {
156 // 顶层是 string list 或单个 [rel,alias]
157 if (!raw.empty() && raw[0].is_string()) {
158 // 可能是单个 [rel, alias] (Scoop 的常见简写) 或 多个 strings
159 // Scoop 语义：list of strings 等价于多 bin。
160 for (auto& item : raw) {
161 if (auto e = coerce_bin_entry(item)) out.push_back(*e);
162 }
163 } else {
164 for (auto& item : raw) {
165 if (auto e = coerce_bin_entry(item)) out.push_back(*e);
166 }
167 }
168 }
169 return out;
170}

198ResolvedManifest parse(const json& manifest,
199 const std::string& name,
200 const std::string& arch) {
201 if (!manifest.is_object()) {
202 throw IncompleteManifest(std::format("manifest '{}' is not a JSON object", name));
203 }
204 check_no_denied(manifest, name, "");
205
206 std::string version = "unknown";
207 if (manifest.contains("version") && manifest["version"].is_string()) {
208 std::string v = trim(manifest["version"].get<std::string>());
209 if (!v.empty()) version = v;
210 }
211
212 // 选 arch 子块（如果存在）。
213 const json* arch_block = nullptr;
214 if (manifest.contains("architecture") && manifest["architecture"].is_object()) {
215 const json& arch_map = manifest["architecture"];
216 const char* scoop_key = arch_to_scoop_key(arch);
217 if (scoop_key && arch_map.contains(scoop_key) && arch_map[scoop_key].is_object()) {
218 arch_block = &arch_map[scoop_key];
219 check_no_denied(*arch_block, name, std::string("arch=") + scoop_key);
220 }
221 }
222
223 auto pick = [&](const char* field) -> json {
224 if (arch_block && arch_block->contains(field)) return (*arch_block)[field];
225 if (manifest.contains(field)) return manifest[field];
226 return json(nullptr);
227 };
228
229 std::string url = normalize_url(pick("url"), name);
230 std::string hash_spec = normalize_hash(pick("hash"), name);
231
232 auto extract_dir = as_string_or_first(pick("extract_dir"));
233 auto extract_to = as_string_or_first(pick("extract_to"));
234
235 std::vector<BinEntry> bins = coerce_bins(pick("bin"));
236 std::map<std::string, std::string> env_set = coerce_env_set(
237 manifest.contains("env_set") ? manifest["env_set"] : json::object());
238 std::vector<std::string> env_add_path = coerce_str_list(
239 manifest.contains("env_add_path") ? manifest["env_add_path"] : json(nullptr));
240 std::vector<std::string> depends = coerce_str_list(
241 manifest.contains("depends") ? manifest["depends"] : json(nullptr));
242
243 // msi/nsis 静默拒绝——需 overlay 改成 zip 等可控格式。
244 std::string lurl = lower(url);
245 if (ends_with(lurl, ".msi") || ends_with(lurl, ".nsis")) {
246 throw UnsafeManifest(std::format(
247 "manifest '{}' points to {}; msi/nsis are not supported — overlay required.",
248 name, url));
249 }
250
251 // luban-specific extension: `luban_mirrors` is an array of fallback URLs
252 // tried after `url` fails. Useful for mirror sites under restricted
253 // networks (e.g., ghproxy.com prefixing GitHub release URLs). Hash
254 // verification still applies — same bytes regardless of source.
255 std::vector<std::string> mirrors;
256 auto pick_mirrors = [&]() -> json {
257 if (arch_block && arch_block->contains("luban_mirrors")) return (*arch_block)["luban_mirrors"];
258 if (manifest.contains("luban_mirrors")) return manifest["luban_mirrors"];
259 return json(nullptr);
260 };
261 if (json m = pick_mirrors(); m.is_array()) {
262 for (auto& el : m) {
263 if (!el.is_string()) continue;
264 std::string s = trim(el.get<std::string>());
265 if (s.empty()) continue;
266 // Same '#dest.zip' rename suffix as primary URL — strip if present.
267 auto h = s.find('#');
268 if (h != std::string::npos) s = s.substr(0, h);
269 mirrors.push_back(std::move(s));
270 }
271 }
272
273 return ResolvedManifest{
274 .name = name,
275 .version = version,
276 .url = url,
277 .hash_spec = hash_spec,
278 .extract_dir = extract_dir,
279 .extract_to = extract_to,
280 .bins = std::move(bins),
281 .env_set = std::move(env_set),
282 .env_add_path = std::move(env_add_path),
283 .depends = std::move(depends),
284 .architecture = arch,
285 .mirrors = std::move(mirrors),
286 };
287}
