26int run_which(const cli::ParsedArgs& a) {
27 if (a.positional.empty()) {
28 log::err("usage: luban which <alias>");
29 return 2;
30 }
31 const std::string& alias = a.positional[0];
32
33 auto hit = registry::resolve_alias(alias);
34 if (!hit) {
35 log::errf("no alias '{}' in installed.json", alias);
36 log::infof("hint: `luban doctor` lists installed components.");
37 return 1;
38 }
39
40 // 主输出走 stdout（脚本可 pipe）：绝对路径
41 std::cout << hit->exe.string() << '\n';
42
43 // 辅助信息走 stderr
44 log::infof("from component: {}", hit->component);
45 if (hit->all_components.size() > 1) {
46 std::string joined;
47 for (size_t i = 0; i < hit->all_components.size(); ++i) {
48 if (i) joined += ", ";
49 joined += hit->all_components[i];
50 }
51 log::warnf("alias '{}' is provided by {} components: {} (using first)",
52 alias, hit->all_components.size(), joined);
53 }
54
55 std::error_code ec;
56 if (!fs::exists(hit->exe, ec)) {
57 log::warnf("path does not exist on disk! Run `luban setup --only {} --force`?",
58 hit->component);
59 return 1;
60 }
61 return 0;
62}

64int run_run(const cli::ParsedArgs& a) {
65 if (a.positional.empty()) {
66 log::err("usage: luban run <cmd> [args...]");
67 log::info("examples: luban run cmake --version");
68 log::info(" luban run clang -E -dM -x c nul");
69 return 2;
70 }
71 std::string cmd_name = a.positional[0];
72
73 // Resolve via registry first — bypasses any one-layer-of-cmd.exe overhead
74 // from .cmd shims, and prevents same-named PATH entries from winning over
75 // luban-managed binaries.
76 fs::path exe;
77 if (auto hit = registry::resolve_alias(cmd_name)) {
78 exe = hit->exe;
79 } else {
80 // Fallback: search PATH augmented with luban's toolchain dirs.
81 // path_search::on_path reads the current process PATH, so we temporarily
82 // splice luban's toolchain dirs in front, search, and restore.
83#ifdef _WIN32
84 std::string current_path;
85 if (const char* p = std::getenv("PATH"); p) current_path = p;
86 std::string augmented;
87 for (auto& d : env_snapshot::path_dirs()) {
88 if (!augmented.empty()) augmented.push_back(';');
89 augmented += d.string();
90 }
91 if (!current_path.empty()) augmented += ';' + current_path;
92 SetEnvironmentVariableA("PATH", augmented.c_str());
93
94 if (auto hit = path_search::on_path(cmd_name)) exe = *hit;
95
96 // restore PATH
97 SetEnvironmentVariableA("PATH", current_path.c_str());
98#else
99 if (auto hit = path_search::on_path(cmd_name)) exe = *hit;
100#endif
101 }
102
103 if (exe.empty() || !fs::exists(exe)) {
104 log::errf("'{}' not found in registry or PATH", cmd_name);
105 log::info("hint: `luban which <alias>` to debug, or `luban setup --only <alias>` to install");
106 return 127;
107 }
108
109 // 组合 cmd + 透传 args
110 std::vector<std::string> argv;
111 argv.push_back(exe.string());
112 for (size_t i = 1; i < a.positional.size(); ++i) argv.push_back(a.positional[i]);
113
114 // 注入 luban env：toolchain dirs 加 PATH 头部 + LUBAN_*
115 auto env_overrides = env_snapshot::apply_to({});
116 return proc::run(argv, fs::current_path().string(), env_overrides);
117}

119int run_search(const cli::ParsedArgs& a) {
120 if (a.positional.empty()) {
121 log::err("usage: luban search <pattern>");
122 return 2;
123 }
124 const std::string& pattern = a.positional[0];
125
126 auto vcpkg = registry::resolve_alias("vcpkg");
127 if (!vcpkg) {
128 log::err("vcpkg is not installed (no alias 'vcpkg' in installed.json).");
129 log::info("install it via: luban setup --only vcpkg");
130 return 1;
131 }
132
133 std::vector<std::string> cmd = {vcpkg->exe.string(), "search", pattern};
134 auto env_overrides = env_snapshot::apply_to({});
135 return proc::run(cmd, fs::current_path().string(), env_overrides);
136}

140void register_run() {
141 cli::Subcommand c;
142 c.name = "run";
143 c.help = "run a luban-managed exe with toolchain env injected (uv-style)";
144 c.group = "advanced";
145 c.long_help =
146 " `luban run <cmd> [args...]` injects luban's toolchain env (PATH +\n"
147 " VCPKG_ROOT + vcpkg cache vars) into a child process and exec's the\n"
148 " command, forwarding all args verbatim. Useful when you haven't run\n"
149 " `luban env --user` (e.g., in a fresh CI shell or container).\n"
150 "\n"
151 " `<cmd>` resolves via:\n"
152 " 1. registry alias (e.g., 'cmake' → toolchains/cmake-X/bin/cmake.exe)\n"
153 " 2. fallback PATH search using luban-augmented PATH\n"
154 "\n"
155 " All args after `<cmd>` are forwarded WITHOUT luban-side parsing\n"
156 " (so `luban run cmake --version` works, the --version goes to cmake).";
157 c.examples = {
158 "luban run cmake --version\tWithout `luban env --user` first",
159 "luban run clang -E -dM -x c nul\tDump preprocessor macros",
160 "luban run vcpkg list\tList installed ports",
161 };
162 c.forward_rest = true;
163 c.run = run_run;
164 cli::register_subcommand(std::move(c));
165}
