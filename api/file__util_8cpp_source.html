11// Tolerant slurp — never throws. Empty string covers both "missing" and
12// "empty file"; consumers fs::exists() first if they need to distinguish.
13std::string slurp(const fs::path& path) {
14 std::ifstream in(path, std::ios::binary);
15 if (!in) return {};
16 std::ostringstream ss;
17 ss << in.rdbuf();
18 return ss.str();
19}

21// EF BB BF — the UTF-8 byte-order mark. Notepad and various Scoop manifest
22// editors prepend it; tolerant parsers (nlohmann::json, toml++) reject it.
23// Strip in-place when present.
24void strip_utf8_bom(std::string& s) {
25 if (s.size() >= 3
26 && static_cast<unsigned char>(s[0]) == 0xEF
27 && static_cast<unsigned char>(s[1]) == 0xBB
28 && static_cast<unsigned char>(s[2]) == 0xBF) {
29 s.erase(0, 3);
30 }
31}

45bool write_text_atomic(const fs::path& path, std::string_view content) {
46 std::error_code ec;
47 fs::create_directories(path.parent_path(), ec);
48 // ignore create_directories ec — if parent already exists or path has no
49 // parent (relative bare name), that's fine; the open below will surface
50 // any real problem.
51
52 fs::path tmp = path; tmp += ".tmp";
53 {
54 std::ofstream out(tmp, std::ios::binary | std::ios::trunc);
55 if (!out) return false;
56 out.write(content.data(), static_cast<std::streamsize>(content.size()));
57 if (!out) return false;
58 }
59
60 fs::rename(tmp, path, ec);
61 if (ec) {
62 // Cross-volume rename returns EXDEV / ERROR_NOT_SAME_DEVICE. Fall back
63 // to copy + remove so the helper works for `<state>` on D: with
64 // `<config>` on C: and similar split-volume layouts.
65 ec.clear();
66 fs::copy_file(tmp, path, fs::copy_options::overwrite_existing, ec);

69 }
