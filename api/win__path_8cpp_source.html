18// HKCU\Environment 的子键名。
19constexpr const wchar_t* kEnvKey = L"Environment";
20constexpr const wchar_t* kPathName = L"PATH";
21
22// 读 HKCU\Environment\PATH 的当前值 + 类型（REG_SZ 或 REG_EXPAND_SZ）。
23// 不存在返回空 + REG_EXPAND_SZ（默认创建类型）。
24std::pair<std::wstring, DWORD> read_path_raw() {
25 HKEY key = nullptr;
26 LONG rc = RegOpenKeyExW(HKEY_CURRENT_USER, kEnvKey, 0, KEY_QUERY_VALUE, &key);
27 if (rc != ERROR_SUCCESS) return {L"", REG_EXPAND_SZ};
28
29 DWORD type = 0;
30 DWORD size_bytes = 0;
31 rc = RegQueryValueExW(key, kPathName, nullptr, &type, nullptr, &size_bytes);
32 if (rc != ERROR_SUCCESS || size_bytes == 0) {
33 RegCloseKey(key);
34 return {L"", REG_EXPAND_SZ};
35 }
36
37 std::wstring buf(size_bytes / sizeof(wchar_t), L'\0');
38 rc = RegQueryValueExW(key, kPathName, nullptr, &type,
39 reinterpret_cast<LPBYTE>(buf.data()), &size_bytes);
40 RegCloseKey(key);
41 if (rc != ERROR_SUCCESS) return {L"", REG_EXPAND_SZ};
42
43 while (!buf.empty() && buf.back() == L'\0') buf.pop_back();
44 return {buf, type};
45}

47bool write_path_raw(const std::wstring& value, DWORD type) {
48 HKEY key = nullptr;
49 LONG rc = RegOpenKeyExW(HKEY_CURRENT_USER, kEnvKey, 0, KEY_SET_VALUE, &key);
50 if (rc != ERROR_SUCCESS) return false;
51
52 DWORD size_bytes = static_cast<DWORD>((value.size() + 1) * sizeof(wchar_t));
53 rc = RegSetValueExW(key, kPathName, 0, type,
54 reinterpret_cast<const BYTE*>(value.data()), size_bytes);
55 RegCloseKey(key);
56 if (rc != ERROR_SUCCESS) return false;
57
58 // 广播：让 cmd / explorer / 新生 PowerShell 拿到新 PATH。
59 DWORD_PTR result = 0;
60 SendMessageTimeoutW(HWND_BROADCAST, WM_SETTINGCHANGE, 0,
61 reinterpret_cast<LPARAM>(L"Environment"),
62 SMTO_ABORTIFHUNG, 5000, &result);
63 return true;
64}

66std::vector<std::wstring> split_semi(const std::wstring& s) {
67 std::vector<std::wstring> out;
68 size_t start = 0;
69 while (start <= s.size()) {
70 size_t end = s.find(L';', start);
71 if (end == std::wstring::npos) end = s.size();
72 std::wstring piece = s.substr(start, end - start);
73 // 跳过空 entry（用户 PATH 末尾经常残留 ;）。
74 if (!piece.empty()) out.push_back(piece);
75 if (end == s.size()) break;
76 start = end + 1;
77 }
78 return out;
79}

82 std::wstring out;
83 for (size_t i = 0; i < parts.size(); ++i) {
84 if (i) out.push_back(L';');

86 }

90bool ieq(const std::wstring& a, const std::wstring& b) {
91 if (a.size() != b.size()) return false;
92 for (size_t i = 0; i < a.size(); ++i) {
93 if (std::towlower(static_cast<wint_t>(a[i])) !=
94 std::towlower(static_cast<wint_t>(b[i]))) return false;
95 }
96 return true;
97}

99// 归一化 — 路径分隔符统一成 '\\'，去尾部 '\\'，对照用。
100std::wstring normalize(std::wstring s) {
101 for (auto& c : s) if (c == L'/') c = L'\\';
102 while (s.size() > 3 /*盘符如 C:\*/ && s.back() == L'\\') s.pop_back();
103 return s;
104}

110bool add_to_user_path(const fs::path& dir) {
111#ifdef _WIN32
112 std::wstring target = normalize(win::from_utf8(dir.string()));
113 auto [current, type] = read_path_raw();
114 auto parts = split_semi(current);
115
116 for (const auto& p : parts) {
117 if (ieq(normalize(p), target)) return false;
118 }
119
120 // 头插 — toolchain 优先于既有 entries（rustup 风格）。
121 parts.insert(parts.begin(), target);
122
123 // 类型策略：原本是哪种就保哪种。新建（无 PATH）默认 REG_EXPAND_SZ
124 // 因为用户加的 entry 经常含 %USERPROFILE% 之类。
125 if (type != REG_SZ && type != REG_EXPAND_SZ) type = REG_EXPAND_SZ;
126
127 return write_path_raw(join_semi(parts), type);
128#else
129 (void)dir;
130 return false;
131#endif
132}

134bool remove_from_user_path(const fs::path& dir) {
135#ifdef _WIN32
136 std::wstring target = normalize(win::from_utf8(dir.string()));
137 auto [current, type] = read_path_raw();
138 auto parts = split_semi(current);
139
140 auto orig = parts.size();
141 parts.erase(std::remove_if(parts.begin(), parts.end(),
142 [&](const std::wstring& p) { return ieq(normalize(p), target); }),
143 parts.end());
144
145 if (parts.size() == orig) return false;
146 if (type != REG_SZ && type != REG_EXPAND_SZ) type = REG_EXPAND_SZ;
147 return write_path_raw(join_semi(parts), type);
148#else
149 (void)dir;
150 return false;
151#endif
152}
