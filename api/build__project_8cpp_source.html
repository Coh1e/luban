36// Find the real cmake.exe via the registry, bypassing .cmd shim. Falls back
37// to "cmake" on PATH.
38std::string cmake_exe() {
39 auto recs = registry::load_installed();
40 auto it = recs.find("cmake");
41 if (it != recs.end()) {
42 fs::path root = paths::toolchain_dir(it->second.toolchain_dir);
43 for (auto& [alias, rel] : it->second.bins) {
44 if (alias == "cmake") {
45 std::string norm = rel;
46 for (auto& c : norm) if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
47 return (root / norm).string();
48 }
49 }
50 }
51 fs::path shim = paths::bin_dir() / "cmake.cmd";
52 std::error_code ec;
53 if (fs::exists(shim, ec)) return shim.string();
54 return "cmake";
55}

69fs::path copy_compile_commands(const fs::path& project, const std::string& preset) {
70 fs::path src = project / "build" / preset / "compile_commands.json";
71 std::error_code ec;
72 if (!fs::exists(src, ec)) return {};
73 fs::path dst = project / "compile_commands.json";
74 fs::copy_file(src, dst, fs::copy_options::overwrite_existing, ec);
75 if (ec) return {};
76 return dst;
77}

79// 项目 CMakePresets.json 是否声明了 wasm preset。`luban new --target=wasm`
80// 生成的项目里有；普通项目没有。
81bool has_wasm_preset(const fs::path& project) {
82 fs::path p = project / "CMakePresets.json";
83 std::error_code ec;
84 if (!fs::exists(p, ec)) return false;
85 std::string text = file_util::read_text_no_bom(p);
86 nlohmann::json doc;
87 try { doc = nlohmann::json::parse(text); } catch (...) { return false; }
88 if (!doc.is_object() || !doc.contains("configurePresets")) return false;
89 for (auto& cp : doc["configurePresets"]) {
90 if (cp.is_object() && cp.contains("name") && cp["name"].is_string()
91 && cp["name"].get<std::string>() == "wasm") return true;
92 }
93 return false;
94}

96// 自动选 preset：
97// - 用户显式 --preset xxx → 直接用
98// - 项目有 wasm preset → "wasm" (emscripten 路径)
99// - 项目 vcpkg.json 有 deps → "default"（vcpkg toolchain）
100// - 否则 → "no-vcpkg"（hello-world fast path，无需 VCPKG_ROOT）
101std::string pick_preset(const fs::path& project, const std::string& explicit_preset) {
102 if (!explicit_preset.empty() && explicit_preset != "auto") return explicit_preset;
103 if (has_wasm_preset(project)) return "wasm";
104 fs::path vcpkg_json = project / "vcpkg.json";
105 std::error_code ec;
106 if (!fs::exists(vcpkg_json, ec)) return "default";
107 auto m = vcpkg_manifest::load(vcpkg_json);
108 return m.dependencies.empty() ? "no-vcpkg" : "default";
109}

111// emcmake.bat path from the installed emscripten component. Returns empty
112// if emscripten not installed (caller surfaces a friendlier error).
113std::string emcmake_bat() {
114 auto recs = registry::load_installed();
115 auto it = recs.find("emscripten");
116 if (it == recs.end()) return "";
117 fs::path root = paths::toolchain_dir(it->second.toolchain_dir);
118 for (auto& [alias, rel] : it->second.bins) {
119 if (alias == "emcmake") {
120 std::string norm = rel;
121 for (auto& c : norm) if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
122 return (root / norm).string();
123 }
124 }
125 return "";
126}

132// OQ-2: per-project toolchain version pin check. Reads `[toolchain]` from
133// the project's luban.toml and compares each entry to the installed registry.
134// Empty pins map → no-op. Mismatches log a warning and continue (we don't
135// hard-fail; the user may have intentionally upgraded). Unknown components
136// (in pins but not in installed.json) also warn — they signal either a typo
137// or a missing `luban setup`.
138//
139// This runs *before* cmake so the warnings surface above the configure noise
140// rather than getting buried under it.
141void check_toolchain_pins(const fs::path& project) {
142 fs::path toml_path = project / "luban.toml";
143 std::error_code ec;
144 if (!fs::exists(toml_path, ec)) return; // no luban.toml → no pins
145
146 auto cfg = luban::luban_toml::load(toml_path);
147 if (cfg.toolchain.empty()) return; // no [toolchain] section → no-op
148
149 auto installed = registry::load_installed();
150 for (auto const& [name, want] : cfg.toolchain) {
151 auto it = installed.find(name);
152 if (it == installed.end()) {
153 log::warnf("luban.toml [toolchain] pins {} = \"{}\", but {} is not installed. "
154 "Run: luban setup --with {}", name, want, name, name);
155 continue;
156 }
157 std::string have = it->second.version;
158 if (have != want) {
159 log::warnf("luban.toml [toolchain] pins {} = \"{}\", but installed is \"{}\". "
160 "Build proceeds; reinstall the component to match.",
161 name, want, have);
162 }
163 }

166int run_build(const cli::ParsedArgs& args) {
167 fs::path project_arg = args.opts.count("at") ? fs::path(args.opts.at("at")) : fs::current_path();
168 std::error_code ec;
169 fs::path project = fs::absolute(project_arg, ec);
170 if (!fs::exists(project / "CMakeLists.txt", ec)) {
171 log::errf("no CMakeLists.txt in {}", project.string());
172 return 2;
173 }
174
175 check_toolchain_pins(project);
176
177 std::string cmake = cmake_exe();
178 // 用户没传 --preset 时（默认值 "auto"）智能挑选；显式传了用显式值。
179 std::string preset_raw = args.opts.count("preset") ? args.opts.at("preset") : std::string("auto");
180 std::string preset = pick_preset(project, preset_raw);
181 if (preset_raw == "auto") log::infof("preset: {} (auto-picked from vcpkg.json deps)", preset);
182
183 auto env_overrides = env_snapshot::apply_to({});
184
185 // Configure. For wasm presets we wrap with emcmake.bat so it injects the
186 // emscripten cmake toolchain file before invoking cmake. After the first
187 // configure, the toolchain file is cached in CMakeCache, so subsequent
188 // builds don't need emcmake.
189 std::vector<std::string> configure_cmd;
190 if (is_wasm_preset(preset)) {
191 std::string emcmake = emcmake_bat();
192 if (emcmake.empty()) {
193 log::errf("preset '{}' needs emscripten, but it is not installed.", preset);
194 log::err("run: luban setup --with emscripten");
195 return 2;
196 }
197 configure_cmd = {emcmake, cmake, "--preset", preset};
198 } else {
199 configure_cmd = {cmake, "--preset", preset};
200 }
201 int rc = run_cmd(configure_cmd, project, env_overrides);
202 if (rc != 0) {
203 log::errf("cmake --preset {} returned {}", preset, rc);
204 return rc;
205 }
206
207 // Build
208 rc = run_cmd({cmake, "--build", "--preset", preset}, project, env_overrides);
209 if (rc != 0) {
210 log::errf("cmake --build --preset {} returned {}", preset, rc);
211 return rc;
212 }
213
214 fs::path cc = copy_compile_commands(project, preset);
215 if (!cc.empty()) log::okf("compile_commands.json synced \xe2\x86\x92 {}", cc.string());
216 else log::warn("compile_commands.json not found in build dir \xe2\x80\x94 clangd will be unhappy");
217
218 log::ok("build complete");
219 return 0;
220}

224int build_project(const fs::path& project_dir) {
225 cli::ParsedArgs a;
226 a.opts["preset"] = "auto";
227 a.opts["at"] = project_dir.string();
228 return run_build(a);
229}

231void register_build() {
232 cli::Subcommand c;
233 c.name = "build";
234 c.help = "configure + build via cmake/ninja, sync compile_commands.json";
235 c.group = "project";
236 c.long_help =
237 " Wraps `cmake --preset <P> && cmake --build --preset <P>`.\n"
238 " After build, copies build/<preset>/compile_commands.json to project root\n"
239 " so clangd / VS Code C/C++ extension auto-find it.\n"
240 "\n"
241 " Preset auto-selection (default --preset=auto):\n"
242 " - 'wasm' preset present → preset 'wasm' (wraps cmake with emcmake)\n"
243 " - vcpkg.json has deps → preset 'default' (uses VCPKG_ROOT)\n"
244 " - vcpkg.json deps empty → preset 'no-vcpkg' (no toolchain file)\n"
245 " Override with --preset=release, --preset=default, etc.";
246 c.opts = {{"preset", "auto"}, {"at", "."}};
247 c.examples = {
248 "luban build\tAuto-pick preset based on vcpkg.json",
249 "luban build --preset release\tForce release preset",
250 "luban build --at ../other\tBuild a different project dir",
251 };
252 c.run = run_build;
253 cli::register_subcommand(std::move(c));
254}
