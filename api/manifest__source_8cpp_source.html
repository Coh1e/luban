23// Locate the in-tree manifests_seed dir. Mirrors selection.cpp::seed_root()
24// — kept as a private duplicate rather than coupling the two modules,
25// because seed_root is a leaf utility and both modules independently
26// resolve it. Refactor candidate if a third caller appears.
27//
28// Candidate order (by typical install layout):
29// 1) <exe_dir>/manifests_seed (installed alongside)
30// 2) <exe_dir>/../manifests_seed (build/ next to repo root)
31// 3) <exe_dir>/../../manifests_seed (build/release/luban.exe)
32fs::path seed_root() {
33#ifdef _WIN32
34 wchar_t buf[MAX_PATH * 4];
35 DWORD got = GetModuleFileNameW(nullptr, buf, static_cast<DWORD>(std::size(buf)));
36 fs::path exe = (got == 0) ? fs::current_path() / "luban.exe"
37 : fs::path(std::wstring(buf, got));
38#else
39 fs::path exe = fs::current_path() / "luban";
40#endif
41 fs::path d = exe.parent_path();
42 std::vector<fs::path> candidates = {
43 d / "manifests_seed",
44 d.parent_path() / "manifests_seed",
45 d.parent_path().parent_path() / "manifests_seed",
46 };
47 std::error_code ec;
48 for (auto& c : candidates) {
49 if (fs::is_directory(c, ec)) return c;
50 }
51 return {};
52}

54// Try to parse a single candidate file. Returns nullopt if the file doesn't
55// exist or the JSON parse fails. Parse failures are logged at warn level —
56// the caller may still succeed via another source. BOM-stripping handles
57// Scoop / Notepad-saved manifests that include a leading UTF-8 BOM that
58// nlohmann::json would otherwise reject.
59std::optional<json> try_parse(const fs::path& path) {
60 std::error_code ec;
61 if (!fs::exists(path, ec)) return std::nullopt;
62 std::string text = file_util::read_text_no_bom(path);
63 try {
64 return json::parse(text);
65 } catch (const std::exception& e) {
66 log::warnf("could not parse {}: {}", path.string(), e.what());
67 return std::nullopt;
68 }
69}

73std::optional<LoadResult> load(const std::string& name) {
74 // 1. overlay (user-overridable, populated on first setup from seed)
75 fs::path overlay = paths::overlay_dir() / (name + ".json");
76 if (auto j = try_parse(overlay)) {
77 return LoadResult{std::move(*j), overlay, "overlay"};
78 }
79
80 // 2. seed (in-tree manifests_seed/ next to luban.exe). Lets a fresh
81 // LUBAN_PREFIX work without first running selection::deploy_overlays.
82 fs::path seed = seed_root();
83 if (!seed.empty()) {
84 fs::path direct = seed / (name + ".json");
85 if (auto j = try_parse(direct)) {
86 return LoadResult{std::move(*j), direct, "seed"};
87 }
88 }
89
90 return std::nullopt;
91}
