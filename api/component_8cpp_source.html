41// 把 bin rel 路径里的分隔符归一化（manifest 里大多是 '/'）
42fs::path resolve_bin(const fs::path& root, const std::string& rel) {
43 std::string norm = rel;
44 for (auto& c : norm) {
45 if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
46 }

50// staging 内按 extract_dir 选出 toolchain 真根（manifest 指明 wrapper 名）
51fs::path resolve_toolchain_root(const fs::path& staging,
52 const scoop_manifest::ResolvedManifest& m) {
53 if (!m.extract_dir) return staging;
54 std::error_code ec;
55 fs::path candidate = staging / *m.extract_dir;
56 if (fs::is_directory(candidate, ec)) return candidate;
57 // 尝试斜杠归一化
58 std::string norm = *m.extract_dir;
59 for (auto& c : norm) {
60 if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
61 }
62 candidate = staging / norm;
63 if (fs::is_directory(candidate, ec)) return candidate;
64 return staging;
65}

76std::string archive_filename_from_url(const std::string& url, const std::string& fallback) {
77 auto pos = url.rfind('/');
78 if (pos == std::string::npos || pos + 1 >= url.size()) return fallback;
79 std::string name = url.substr(pos + 1);
80 auto q = name.find('?');
81 if (q != std::string::npos) name = name.substr(0, q);
82 return name.empty() ? fallback : name;
83}

109std::expected<InstallReport, Error> install(const std::string& name, bool force,
110 const std::string& arch) {
111 paths::ensure_dirs();
112
113 // 1. Locate manifest. Only overlay + in-tree seed are searched (no
114 // network); see manifest_source.hpp for the reasoning.
115 auto fetched = manifest_source::load(name);
116 if (!fetched) {
117 return std::unexpected(Error{ErrorKind::ManifestNotFound,
118 "no manifest for '" + name + "' in overlay or manifests_seed/. "
119 "Add a manifests_seed/" + name + ".json (commit it) to install."});
120 }
121
122 // 2. Parse + safety whitelist (rejects installer.script / pre_install /
123 // .msi / etc. — see scoop_manifest.cpp).
124 scoop_manifest::ResolvedManifest parsed;
125 try {
126 parsed = scoop_manifest::parse(fetched->manifest, name, arch);
127 } catch (scoop_manifest::UnsafeManifest& e) {
128 return std::unexpected(Error{ErrorKind::UnsafeManifest, e.what()});
129 } catch (scoop_manifest::IncompleteManifest& e) {
130 return std::unexpected(Error{ErrorKind::IncompleteManifest, e.what()});
131 }
132
133 // 3. plan 路径
134 std::string tc_name = toolchain_dir_name(parsed);
135 fs::path final = final_path(tc_name);
136
137 // 4. 幂等检查
138 auto installed = registry::load_installed();
139 auto it = installed.find(name);
140 std::error_code ec;
141 if (!force && it != installed.end()
142 && it->second.version == parsed.version
143 && fs::exists(final, ec)) {
144 log::okf("{} {} already installed at {}", name, parsed.version, final.string());
145 InstallReport r;
146 r.name = name;
147 r.version = parsed.version;
148 r.toolchain_dir = final;
149 for (auto& [alias, rel] : it->second.bins) {
150 r.bins.emplace_back(alias, resolve_bin(final, rel));
151 }
152 r.was_already_installed = true;
153 return r;
154 }
155
156 // 5. download archive（带 SHA 校验）
157 std::string archive_name = archive_filename_from_url(
158 parsed.url, parsed.name + "-" + parsed.version + ".bin");
159 fs::path archive_path = paths::downloads_dir() / archive_name;
160
161 auto expected_hash = hash::parse(parsed.hash_spec);
162 if (!expected_hash) {
163 return std::unexpected(Error{ErrorKind::IncompleteManifest,
164 "bad hash spec: " + parsed.hash_spec});
165 }
166
167 // 缓存命中也再校验一次
168 bool cached_ok = false;
169 if (fs::exists(archive_path, ec)) {
170 if (hash::verify_file(archive_path, *expected_hash)) {
171 cached_ok = true;
172 } else {
173 log::warnf("cached {} hash mismatch; re-downloading", archive_name);
174 fs::remove(archive_path, ec);
175 }
176 }
177 if (!cached_ok) {
178 fs::create_directories(archive_path.parent_path(), ec);
179 download::DownloadOptions opts;
180 opts.expected_hash = *expected_hash;
181 opts.label = archive_name;
182 opts.retries = 3;
183 opts.timeout_seconds = 60;
184 // Toolchain archives are typically 10-280 MB; the chunked path
185 // shaves real time off LLVM-MinGW (the 280 MB outlier) and is
186 // negligible overhead on smaller files (HEAD probe + skip).
187 // Falls back to single-stream on any failure (HEAD, range, etc.).
188 opts.parallel_chunks = 4;
189
190 // Try the canonical URL, then each mirror in order. A hash mismatch
191 // is FATAL (don't iterate to mirrors — that means the published hash
192 // is wrong, and trying mirrors won't help). Network errors fall
193 // through to the next URL. Same hash applies to whichever URL serves.
194 std::vector<std::string> candidates = {parsed.url};
195 for (auto& m : parsed.mirrors) candidates.push_back(m);
196
197 std::optional<download::Error> last_error;
198 bool ok = false;
199 for (size_t i = 0; i < candidates.size(); ++i) {
200 log::stepf("download {} ({} of {}): {}",
201 archive_name, i + 1, candidates.size(), candidates[i]);
202 auto dl = download::download(candidates[i], archive_path, opts);
203 if (dl) { ok = true; break; }
204 last_error = dl.error();
205 if (dl.error().kind == download::ErrorKind::HashMismatch) {
206 // Bytes mismatched the expected hash — don't try mirrors,
207 // that's a manifest bug. Return immediately.
208 return std::unexpected(Error{ErrorKind::HashMismatch, dl.error().message});
209 }
210 if (i + 1 < candidates.size()) {
211 log::warnf(" failed: {}; trying next mirror", dl.error().message);
212 }
213 }
214 if (!ok) {
215 return std::unexpected(Error{ErrorKind::DownloadFailed,
216 last_error ? last_error->message
217 : std::string("all mirrors failed")});
218 }
219 }
220
221 // 6. extract 到 staging
222 fs::path staging = staging_path(tc_name);
223 if (fs::exists(staging, ec)) wipe(staging);
224 fs::create_directories(staging, ec);
225 log::stepf("extract {}", archive_name);
226 if (auto rc = archive::extract(archive_path, staging); !rc) {
227 wipe(staging);
228 return std::unexpected(Error{ErrorKind::ExtractFailed, rc.error().message});
229 }
230
231 // 7. extract_dir：选 toolchain 真根
232 fs::path tc_root = resolve_toolchain_root(staging, parsed);
233
234 // 8. promote staging → final
235 if (fs::exists(final, ec)) wipe(final);
236 fs::create_directories(final.parent_path(), ec);
237 fs::rename(tc_root, final, ec);
238 if (ec) {
239 // 跨卷 fallback：递归 copy
240 ec.clear();
241 fs::copy(tc_root, final, fs::copy_options::recursive, ec);
242 if (ec) {
243 wipe(staging);
244 return std::unexpected(Error{ErrorKind::Filesystem,
245 "rename/copy failed: " + ec.message()});
246 }
247 }
248 if (tc_root != staging) wipe(staging); // 清理空 wrapper
249
250 // 8b. vcpkg 特殊路径：extract 出来的源树没有 vcpkg.exe；跑一次 bootstrap-vcpkg.bat
251 // 就地下载匹配的 vcpkg.exe 到 final 根目录。这是 vcpkg 上游 blessed 的安装步骤，
252 // 不是 manifest 注入的脚本——所以不违反"不执行 manifest 脚本"的安全策略。
253 // bootstrap-vcpkg.bat 内部读 scripts/vcpkg-tool-metadata.txt 选 vcpkg-tool 版本，
254 // 从 microsoft/vcpkg-tool/releases 下载。需要 PATH 上有 mingit（cert + curl）。
255 if (name == "vcpkg") {
256 fs::path bootstrap = final / "bootstrap-vcpkg.bat";
257 fs::path vcpkg_exe = final / "vcpkg.exe";
258 if (!fs::exists(vcpkg_exe, ec) && fs::exists(bootstrap, ec)) {
259 log::stepf("bootstrapping vcpkg.exe (microsoft/vcpkg-tool)");
260 std::map<std::string, std::string> env_overrides;
261 // 让 bootstrap 自己用系统 cmd.exe，不需特别注入。
262 int rc = proc::run({"cmd.exe", "/c", bootstrap.string(), "-disableMetrics"},
263 final.string(), env_overrides);
264 if (rc != 0) {
265 log::warnf("bootstrap-vcpkg.bat exited {}; vcpkg may be incomplete", rc);
266 } else {
267 log::ok("vcpkg.exe ready");
268 }
269 }
270 }

272 // 8c. emscripten special-case: write the python-style config so emcc /
273 // emcmake can locate LLVM_ROOT / BINARYEN_ROOT / EMSCRIPTEN_ROOT /
274 // NODE_JS without an emsdk activation step.
275 //
276 // Path: <config>/emscripten/config (XDG-respecting). Older luban
277 // (pre-v0.2) wrote it next to the install at <install>/emscripten/
278 // .emscripten — but that location has two problems:
279 // (1) it's data-dir-side state under config-dir's job, breaking
280 // our 4-home contract;
281 // (2) `emcc` looks at $EM_CONFIG when set, otherwise probes a hard-
282 // coded list (~/.emscripten on Linux, %USERPROFILE%\.emscripten
283 // on Windows). We expose EM_CONFIG via env_snapshot + HKCU so
284 // the file's actual location is decoupled from where luban
285 // drops the toolchain.
286 //
287 // Looks up node from registry; if missing, falls back to PATH-resolved
288 // node so emcc can at least try.
289 if (name == "emscripten") {
290 fs::path config = paths::config_dir() / "emscripten" / "config";
291 std::string node_path;
292 {
293 auto recs = registry::load_installed();
294 auto it = recs.find("node");
295 if (it != recs.end()) {
296 fs::path node_root = paths::toolchain_dir(it->second.toolchain_dir);
297 fs::path node_exe = node_root / "node.exe";
298 if (fs::exists(node_exe, ec)) {
299 node_path = node_exe.string();
300 std::replace(node_path.begin(), node_path.end(), '\\', '/');
301 }
302 }
303 }
304 std::string llvm_root = (final / "bin").string();
305 std::string binaryen_root = final.string();
306 std::string emscripten_root = (final / "emscripten").string();
307 for (auto* p : {&llvm_root, &binaryen_root, &emscripten_root}) {
308 std::replace(p->begin(), p->end(), '\\', '/');
309 }
310 std::ostringstream cfg;
311 cfg << "# generated by luban — DO NOT EDIT BY HAND\n";
312 cfg << "# luban regenerates this every `luban setup --with emscripten`.\n";
313 cfg << "# emcc reads via $EM_CONFIG (set by `luban env --user` or env_snapshot).\n";
314 cfg << "LLVM_ROOT = '" << llvm_root << "'\n";
315 cfg << "BINARYEN_ROOT = '" << binaryen_root << "'\n";
316 cfg << "EMSCRIPTEN_ROOT = '" << emscripten_root << "'\n";
317 if (!node_path.empty()) {
318 cfg << "NODE_JS = '" << node_path << "'\n";
319 } else {
320 log::warn("node not in registry; emscripten will fall back to PATH-resolved node");
321 cfg << "NODE_JS = 'node'\n";
322 }
323 cfg << "COMPILER_ENGINE = NODE_JS\n";
324 cfg << "JS_ENGINES = [NODE_JS]\n";
325 std::error_code ec2;
326 fs::create_directories(config.parent_path(), ec2);
327 std::ofstream out(config, std::ios::binary | std::ios::trunc);
328 std::string text = cfg.str();
329 out.write(text.data(), static_cast<std::streamsize>(text.size()));
330 log::okf("emscripten config written \xe2\x86\x92 {}", config.string());
331 log::info("set EM_CONFIG=<that path> in your shell, or run `luban env --user` to register it on HKCU");
332 }

334 // 9. 写 shim + 收集 bin 路径
335 //
336 // bin_dir() is <data>/bin/ (cargo's ~/.cargo/bin/ pattern — luban-owned,
337 // not shared with other XDG bin tenants). Collisions there should be
338 // rare since users don't manually drop files into luban's data dir, but
339 // shim::write_shim still checks the .shim-table.json before clobbering.
340 std::vector<std::pair<std::string, fs::path>> bins_out;
341 std::vector<std::string> aliases_taken;
342 for (auto& entry : parsed.bins) {
343 fs::path exe = resolve_bin(final, entry.relative_path);
344 if (!fs::exists(exe, ec)) {
345 log::warnf(" bin '{}' points at missing file {}", entry.alias, exe.string());
346 continue;
347 }
348 auto wr = shim::write_shim(entry.alias, exe, entry.prefix_args);
349 if (wr == shim::WriteResult::Skipped) {
350 log::warnf(" shim skipped: {} already exists in {} (not luban-managed)",
351 entry.alias, paths::bin_dir().string());
352 log::info(" → run `luban shim --force` to overwrite");
353 }
354 bins_out.emplace_back(entry.alias, exe);
355 aliases_taken.push_back(entry.alias);
356 }
357 // env_add_path 暴露的目录里的 *.exe / .cmd / .bat 也自动 shim（git 风格 manifest 用这个）
358 for (auto& rel_dir : parsed.env_add_path) {
359 std::string norm = rel_dir;
360 for (auto& c : norm) {
361 if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
362 }
363 fs::path dir_path = final / norm;
364 if (!fs::is_directory(dir_path, ec)) continue;
365 std::vector<fs::directory_entry> entries;
366 for (auto& e : fs::directory_iterator(dir_path, ec)) entries.push_back(e);
367 std::sort(entries.begin(), entries.end(),
368 [](auto& a, auto& b) { return a.path() < b.path(); });
369 for (auto& e : entries) {
370 if (!e.is_regular_file()) continue;
371 std::string ext = e.path().extension().string();
372 std::transform(ext.begin(), ext.end(), ext.begin(),
373 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
374 if (ext != ".exe" && ext != ".cmd" && ext != ".bat") continue;
375 std::string alias = e.path().stem().string();
376 if (std::find(aliases_taken.begin(), aliases_taken.end(), alias) != aliases_taken.end()) continue;
377 auto wr = shim::write_shim(alias, e.path());
378 if (wr == shim::WriteResult::Skipped) {
379 log::warnf(" shim skipped: {} already exists in {} (not luban-managed)",
380 alias, paths::bin_dir().string());
381 continue;
382 }
383 bins_out.emplace_back(alias, e.path());
384 aliases_taken.push_back(alias);
385 }
386 }
387
388 // 10. 写 installed.json
389 registry::ComponentRecord rec;
390 rec.name = name;
391 rec.version = parsed.version;
392 rec.source = fetched->source_label;
393 rec.url = parsed.url;
394 rec.hash_spec = parsed.hash_spec;
395 rec.toolchain_dir = tc_name;
396 rec.architecture = parsed.architecture;
397 rec.installed_at = now_iso();
398 for (auto& [alias, exe] : bins_out) {
399 std::string rel = fs::relative(exe, final, ec).string();
400 std::replace(rel.begin(), rel.end(), '\\', '/');
401 rec.bins.emplace_back(alias, rel);
402 }
403
404 auto recs = registry::load_installed();
405 recs[name] = rec;
406 registry::save_installed(recs);
407
408 log::okf("installed {} {} → {}", name, parsed.version, final.string());
409
410 InstallReport r;
411 r.name = name;
412 r.version = parsed.version;
413 r.toolchain_dir = final;
414 r.bins = std::move(bins_out);
415 return r;
416}

418bool uninstall(const std::string& name) {
419 auto recs = registry::load_installed();
420 auto it = recs.find(name);
421 if (it == recs.end()) return false;
422 registry::ComponentRecord rec = it->second;
423 recs.erase(it);
424 registry::save_installed(recs);
425
426 if (!rec.toolchain_dir.empty()) {
427 fs::path target = paths::toolchains_dir() / rec.toolchain_dir;
428 std::error_code ec;
429 if (fs::exists(target, ec)) wipe(target);
430 }
431 for (auto& [alias, _] : rec.bins) {
432 shim::remove_shim(alias);
433 }
434 return true;
435}
