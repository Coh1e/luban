130ParsedArgs parse_subargs(const Subcommand& c, std::span<char*> args, bool& help_wanted) {
131 ParsedArgs out;
132 for (const auto& f : c.flags) out.flags[f] = false;
133 for (const auto& [k, v] : c.opts) out.opts[k] = v;
134
135 // forward_rest: subcommand wants all argv after itself as positional, verbatim.
136 // Used by `luban run <cmd> [args...]` so flags meant for the child cmd
137 // (e.g. `--version`) don't get parsed by luban's CLI.
138 // Only `luban run --help` (single arg, no other words) prints luban's help.
139 if (c.forward_rest) {
140 if (args.size() == 1 && (args[0] == std::string_view("--help") ||
141 args[0] == std::string_view("-h"))) {
142 help_wanted = true;
143 return out;
144 }
145 for (auto* a : args) out.positional.emplace_back(a);
146 return out;
147 }
148
149 for (size_t i = 0; i < args.size(); ++i) {
150 std::string_view a = args[i];
151 if (a == "--help" || a == "-h") { help_wanted = true; continue; }
152 if (a.starts_with("--")) {
153 std::string_view name = a.substr(2);
154 auto eq = name.find('=');
155 std::string_view val;
156 if (eq != std::string_view::npos) {
157 val = name.substr(eq + 1);
158 name = name.substr(0, eq);
159 }
160 // Match against this subcommand's options/flags.
161 auto opt_it = std::find_if(c.opts.begin(), c.opts.end(),
162 [&](auto& p) { return p.first == name; });
163 if (opt_it != c.opts.end()) {
164 if (eq == std::string_view::npos) {
165 if (i + 1 >= args.size()) {
166 log::errf("missing value for --{}", name);
167 out.opts["__error"] = "1";
168 return out;
169 }
170 val = args[++i];
171 }
172 out.opts[std::string(name)] = std::string(val);
173 continue;
174 }
175 if (std::find(c.flags.begin(), c.flags.end(), std::string(name)) != c.flags.end()) {
176 out.flags[std::string(name)] = true;
177 continue;
178 }
179 log::errf("unknown option: --{}", name);
180 out.opts["__error"] = "1";
181 return out;
182 }
183 if (is_flag_or_opt(a)) {
184 log::errf("unknown short option: {}", a);
185 out.opts["__error"] = "1";
186 return out;
187 }
188 out.positional.emplace_back(a);
189 }
190 return out;
191}

199int dispatch(int argc, char** argv) {
200 if (argc < 2) {
201 print_top_help();
202 return 0;
203 }
204 std::span<char*> rest(argv + 1, static_cast<size_t>(argc - 1));
205
206 // Pre-pass: -v/--verbose can sit before the command.
207 std::vector<char*> filtered;
208 for (auto* a : rest) {
209 std::string_view s = a;
210 if (s == "-v" || s == "--verbose") log::set_verbose(true);
211 else filtered.push_back(a);
212 }
213 if (filtered.empty()) { print_top_help(); return 0; }
214
215 std::string_view first = filtered[0];
216 if (first == "-V" || first == "--version") {
217 std::cout << "luban " << kLubanVersion << '\n';
218 return 0;
219 }
220 if (first == "-h" || first == "--help") {
221 print_top_help();
222 return 0;
223 }
224
225 const Subcommand* c = find_cmd(first);
226 if (!c) {
227 log::errf("unknown command: {}", first);
228 print_top_help();
229 return 2;
230 }
231
232 std::span<char*> sub(filtered.data() + 1, filtered.size() - 1);
233 bool help_wanted = false;
234 ParsedArgs parsed = parse_subargs(*c, sub, help_wanted);
235 if (help_wanted) { print_cmd_help(*c); return 0; }
236 if (parsed.opts.contains("__error")) return 2;
237 if (static_cast<int>(parsed.positional.size()) < c->n_positional) {
238 log::errf("expected {} positional argument(s) for `{}`, got {}",
239 c->n_positional, c->name, parsed.positional.size());
240 print_cmd_help(*c);
241 return 2;
242 }
243 return c->run(parsed);
244}
