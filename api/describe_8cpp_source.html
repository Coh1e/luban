30// 找项目根：cwd 向上一直到根，第一个含 luban.cmake 或 vcpkg.json 的目录。
31fs::path find_project_root() {
32 std::error_code ec;
33 fs::path d = fs::current_path(ec);
34 while (!d.empty()) {
35 if (fs::exists(d / "luban.cmake", ec) || fs::exists(d / "vcpkg.json", ec)) return d;
36 fs::path parent = d.parent_path();
37 if (parent == d) break;
38 d = parent;
39 }
40 return {};
41}

43json build_state() {
44 json out;
45
46 // ---- system tier ----
47 out["luban_version"] = "0.1.0";
48
49 json paths_obj;
50 for (auto& [name, p] : paths::all_dirs()) paths_obj[name] = p.string();
51 out["paths"] = paths_obj;
52
53 json comps = json::array();
54 for (auto& [name, rec] : registry::load_installed()) {
55 json bins_arr = json::array();
56 for (auto& [alias, rel] : rec.bins) {
57 json b;
58 b["alias"] = alias;
59 b["relative_path"] = rel;
60 b["absolute_path"] = (paths::toolchain_dir(rec.toolchain_dir) / rel).string();
61 bins_arr.push_back(b);
62 }
63 json c;
64 c["name"] = name;
65 c["version"] = rec.version;
66 c["source"] = rec.source;
67 c["url"] = rec.url;
68 c["hash"] = rec.hash_spec;
69 c["toolchain_dir"] = (paths::toolchains_dir() / rec.toolchain_dir).string();
70 c["architecture"] = rec.architecture;
71 c["installed_at"] = rec.installed_at;
72 c["bins"] = bins_arr;
73 comps.push_back(c);
74 }
75 out["installed_components"] = comps;
76
77 // ---- project tier (if cwd is in a luban project) ----
78 fs::path proj = find_project_root();
79 if (!proj.empty()) {
80 json p;
81 p["root"] = proj.string();
82
83 // vcpkg.json deps
84 std::error_code ec;
85 if (fs::exists(proj / "vcpkg.json", ec)) {
86 auto m = vcpkg_manifest::load(proj / "vcpkg.json");
87 json deps = json::array();
88 for (auto& d : m.dependencies) {
89 json dj;
90 dj["name"] = d.name;
91 if (d.version_ge) dj["version_ge"] = *d.version_ge;
92 deps.push_back(dj);
93 }
94 p["name"] = m.name;
95 p["version"] = m.version;
96 p["dependencies"] = deps;
97 }
98
99 // luban.toml prefs
100 if (fs::exists(proj / "luban.toml", ec)) {
101 auto t = luban_toml::load(proj / "luban.toml");
102 json toml_obj;
103 toml_obj["cpp"] = t.project.cpp;
104 toml_obj["triplet"] = t.project.triplet;
105 toml_obj["default_preset"] = t.project.default_preset;
106 const char* warn_str = "normal";
107 switch (t.scaffold.warnings) {
108 case luban_toml::WarningLevel::Off: warn_str = "off"; break;
109 case luban_toml::WarningLevel::Strict: warn_str = "strict"; break;
110 default: break;
111 }
112 toml_obj["warnings"] = warn_str;
113 toml_obj["sanitizers"] = t.scaffold.sanitizers;
114 p["luban_toml"] = toml_obj;
115 }
116
117 // build state
118 json builds = json::array();
119 if (fs::exists(proj / "build", ec)) {
120 for (auto& entry : fs::directory_iterator(proj / "build", ec)) {
121 if (!entry.is_directory()) continue;
122 json b;
123 b["preset"] = entry.path().filename().string();
124 b["dir"] = entry.path().string();
125 b["compile_commands"] = fs::exists(entry.path() / "compile_commands.json", ec);
126 builds.push_back(b);
127 }
128 }
129 p["builds"] = builds;
130 p["compile_commands_root"] = (fs::exists(proj / "compile_commands.json", ec))
131 ? (proj / "compile_commands.json").string()
132 : "";
133 out["project"] = p;
134 }
135
136 return out;
137}

139void print_text(const json& s) {
140 std::cout << "Luban " << s.value("luban_version", "?") << "\n\n";
141
142 std::cout << "── Paths ──\n";
143 if (s.contains("paths")) {
144 for (auto& [k, v] : s["paths"].items()) {
145 std::cout << " " << k;
146 for (size_t i = k.size(); i < 18; ++i) std::cout << ' ';
147 std::cout << " " << v.get<std::string>() << '\n';
148 }
149 }
150 std::cout << '\n';
151
152 std::cout << "── Installed components (" << s["installed_components"].size() << ") ──\n";
153 for (auto& c : s["installed_components"]) {
154 std::cout << " " << c["name"].get<std::string>()
155 << " " << c["version"].get<std::string>()
156 << " → " << c["toolchain_dir"].get<std::string>() << '\n';
157 std::cout << " "
158 << c["bins"].size() << " alias(es), source: "
159 << c["source"].get<std::string>() << '\n';
160 }
161 std::cout << '\n';
162
163 if (s.contains("project")) {
164 auto& p = s["project"];
165 std::cout << "── Project: " << p.value("name", "(unnamed)")
166 << " " << p.value("version", "") << " ──\n";
167 std::cout << " root: " << p["root"].get<std::string>() << '\n';
168 if (p.contains("dependencies")) {
169 std::cout << " deps (" << p["dependencies"].size() << "):";
170 for (auto& d : p["dependencies"]) {
171 std::cout << " " << d["name"].get<std::string>();
172 if (d.contains("version_ge")) std::cout << "@>=" << d["version_ge"].get<std::string>();
173 }
174 std::cout << '\n';
175 }
176 if (p.contains("luban_toml")) {
177 auto& t = p["luban_toml"];
178 std::cout << " luban.toml: cpp=" << t.value("cpp", 23)
179 << " triplet=" << t.value("triplet", "")
180 << " warnings=" << t.value("warnings", "")
181 << " sanitizers=" << t.value("sanitizers", json::array()).dump() << '\n';
182 }
183 if (p.contains("builds") && !p["builds"].empty()) {
184 std::cout << " builds:";
185 for (auto& b : p["builds"]) {
186 std::cout << " " << b["preset"].get<std::string>()
187 << "(" << (b.value("compile_commands", false) ? "✓" : "—") << ")";
188 }
189 std::cout << '\n';
190 }
191 } else {
192 std::cout << "(not in a luban project — `cd` into one for project info)\n";
193 }
194}

196int run_describe(const cli::ParsedArgs& a) {
197 auto get_flag = [&](const char* name) {
198 auto it = a.flags.find(name);
199 return it != a.flags.end() && it->second;
200 };
201 bool as_json = get_flag("json");
202 bool want_host = get_flag("host");
203
204 // --host is a focused mode: just the perception snapshot, JSON or
205 // pretty-printed. Bypasses the project-state discovery so it's fast
206 // (sub-millisecond) and useful in any directory.
207 if (want_host) {
208 auto h = perception::snapshot();
209 json doc = perception::to_json(h);
210 if (as_json) {
211 std::cout << doc.dump(2) << '\n';
212 } else {
213 std::cout << "host: " << h.os_name << " " << h.os_version
214 << " (" << h.arch << ")\n";
215 std::cout << "cpu : " << h.cpu_brand
216 << " — " << h.cpu_cores << " cores / "
217 << h.cpu_threads << " threads\n";
218 if (!h.cpu_features.empty()) {
219 std::cout << "simd: ";
220 for (size_t i = 0; i < h.cpu_features.size(); ++i) {
221 if (i) std::cout << ", ";
222 std::cout << h.cpu_features[i];
223 }
224 std::cout << "\n";
225 }
226 if (h.ram_total > 0) {
227 double gb = static_cast<double>(h.ram_total) / (1024.0 * 1024.0 * 1024.0);
228 std::cout << "ram : " << gb << " GB total\n";
229 }
230 if (!h.tools_on_path.empty()) {
231 std::cout << "tools on PATH: ";
232 for (size_t i = 0; i < h.tools_on_path.size(); ++i) {
233 if (i) std::cout << ", ";
234 std::cout << h.tools_on_path[i];
235 }
236 std::cout << "\n";
237 }
238 }
239 return 0;
240 }
241
242 json s = build_state();
243 if (as_json) std::cout << s.dump(2) << '\n';
244 else print_text(s);
245 return 0;
246}

250void register_describe() {
251 cli::Subcommand c;
252 c.name = "describe";
253 c.help = "print system + project state (--json for machines)";
254 c.group = "advanced";
255 c.long_help =
256 " Dump everything luban knows: paths, installed toolchains, bin\n"
257 " aliases, and (when run inside a luban project) vcpkg deps +\n"
258 " luban.toml prefs + build state.\n"
259 "\n"
260 " --host Skip project state; emit only a host capability snapshot\n"
261 " (OS, CPU, RAM, SIMD features, dev tools on PATH, XDG env).\n"
262 " Cheap (~1 ms); usable from any directory.\n"
263 " --json Machine-readable JSON output. Combines with --host.\n"
264 "\n"
265 " Default output is human-readable text. The JSON form has schema=1\n"
266 " and is intended for IDE plugins / AI agents / pipeline scripts.";
267 c.flags = {"json", "host"};
268 c.examples = {
269 "luban describe\tHuman-readable summary",
270 "luban describe --json\tMachine-readable JSON dump",
271 "luban describe --host\tHost capability snapshot only",
272 "luban describe --host --json | jq .cpu_features\tJq filter on host",
273 };
274 c.run = run_describe;
275 cli::register_subcommand(std::move(c));
276}
