24// Quote a single argv element per CommandLineToArgvW rules. Sufficient for
25// our usage (cmake/clang invocations).
26std::wstring quote_arg(const std::wstring& a) {
27 if (!a.empty() && a.find_first_of(L" \t\"") == std::wstring::npos) return a;
28 std::wstring out = L"\"";
29 int backslashes = 0;
30 for (wchar_t c : a) {
31 if (c == L'\\') { ++backslashes; continue; }
32 if (c == L'"') {
33 out.append(static_cast<size_t>(backslashes) * 2 + 1, L'\\');
34 backslashes = 0;
35 out.push_back(L'"');
36 continue;
37 }
38 out.append(static_cast<size_t>(backslashes), L'\\');
39 backslashes = 0;
40 out.push_back(c);
41 }
42 out.append(static_cast<size_t>(backslashes) * 2, L'\\');
43 out.push_back(L'"');
44 return out;
45}

56// Build a WCHAR env block: KEY=VAL\0KEY=VAL\0\0.
57// Starts from current process env, then overlays overrides (case-insensitive
58// on Windows — we lowercase the lookup key).
59std::wstring build_env_block(const std::map<std::string, std::string>& overrides) {
60 std::map<std::wstring, std::wstring> merged;
61
62 LPWCH parent = GetEnvironmentStringsW();
63 if (parent) {
64 for (LPWCH p = parent; *p; ) {
65 std::wstring entry(p);
66 p += entry.size() + 1;
67 auto eq = entry.find(L'=');
68 if (eq == std::wstring::npos || eq == 0) continue;
69 std::wstring key = entry.substr(0, eq);
70 std::wstring val = entry.substr(eq + 1);
71 // Case-insensitive merge: lowercase key for the map.
72 std::wstring k = key;
73 for (auto& ch : k) ch = static_cast<wchar_t>(towlower(ch));
74 merged[k] = key + L'=' + val;
75 }
76 FreeEnvironmentStringsW(parent);
77 }
78 for (auto& [k, v] : overrides) {
79 std::wstring wk = win::from_utf8(k);
80 std::wstring wv = win::from_utf8(v);
81 std::wstring lk = wk;
82 for (auto& ch : lk) ch = static_cast<wchar_t>(towlower(ch));
83 merged[lk] = wk + L'=' + wv;
84 }
85
86 std::wstring block;
87 for (auto& [_, entry] : merged) {
88 block += entry;
89 block.push_back(L'\0');
90 }
91 block.push_back(L'\0');
92 return block;
93}

99int run(const std::vector<std::string>& cmd,
100 const std::string& cwd,
101 const std::map<std::string, std::string>& env_overrides) {
102 if (cmd.empty()) return -1;
103
104#ifdef _WIN32
105 std::wstring cmdline = build_cmdline(cmd);
106 std::wstring env_block = build_env_block(env_overrides);
107 std::wstring wcwd = win::from_utf8(cwd);
108
109 STARTUPINFOW si{};
110 si.cb = sizeof(si);
111 PROCESS_INFORMATION pi{};
112
113 // CreateProcessW mutates the cmdline buffer; give it a writable copy.
114 std::vector<wchar_t> mutable_cmdline(cmdline.begin(), cmdline.end());
115 mutable_cmdline.push_back(L'\0');
116
117 BOOL ok = CreateProcessW(
118 /*lpApplicationName=*/nullptr,
119 /*lpCommandLine=*/mutable_cmdline.data(),
120 /*lpProcessAttributes=*/nullptr,
121 /*lpThreadAttributes=*/nullptr,
122 /*bInheritHandles=*/TRUE,
123 /*dwCreationFlags=*/CREATE_UNICODE_ENVIRONMENT,
124 /*lpEnvironment=*/env_block.data(),
125 /*lpCurrentDirectory=*/wcwd.empty() ? nullptr : wcwd.c_str(),
126 &si, &pi);
127
128 if (!ok) return -1;
129
130 WaitForSingleObject(pi.hProcess, INFINITE);
131 DWORD code = 0;
132 GetExitCodeProcess(pi.hProcess, &code);
133 CloseHandle(pi.hThread);
134 CloseHandle(pi.hProcess);
135 return static_cast<int>(code);
136#else
137 // POSIX path: fork + chdir + apply env overrides + execvp + wait.
138 // Uses posix_spawn-style direct fork/exec — single-threaded, simpler than spawn.
139 pid_t pid = fork();
140 if (pid < 0) return -1;
141
142 if (pid == 0) {
143 // child
144 if (!cwd.empty()) {
145 if (chdir(cwd.c_str()) != 0) {
146 std::fprintf(stderr, "luban: chdir(%s) failed: %s\n",
147 cwd.c_str(), std::strerror(errno));
148 _exit(127);
149 }
150 }
151 // Apply env overrides on top of inherited env. Simple model: setenv
152 // each key. Doesn't unset keys not in overrides — caller can pass
153 // empty value to clear if needed (we don't use that semantic today).
154 for (auto& [k, v] : env_overrides) {
155 setenv(k.c_str(), v.c_str(), /*overwrite=*/1);
156 }
157 // build argv (NULL-terminated)
158 std::vector<char*> argv;
159 argv.reserve(cmd.size() + 1);
160 for (auto& s : cmd) argv.push_back(const_cast<char*>(s.c_str()));
161 argv.push_back(nullptr);
162 execvp(argv[0], argv.data());
163 // execvp returned → failed
164 std::fprintf(stderr, "luban: execvp(%s) failed: %s\n",
165 argv[0], std::strerror(errno));
166 _exit(127);
167 }
168
169 // parent: wait
170 int status = 0;
171 while (waitpid(pid, &status, 0) < 0) {
172 if (errno != EINTR) return -1;
173 }
174 if (WIFEXITED(status)) return WEXITSTATUS(status);
175 if (WIFSIGNALED(status)) return 128 + WTERMSIG(status);
176 return -1;
177#endif
178}
