24// Quoting helpers for the .cmd shim. We don't generate .ps1 / .sh anymore.
25bool needs_cmd_quoting(const std::string& arg) {
26 // Conservative whitelist: any of these triggers double-quote wrapping.
27 static constexpr std::string_view kSpecial = " \t\"&|<>";
28 return arg.find_first_of(kSpecial) != std::string::npos;
29}

31std::string quote_cmd(const std::string& arg) {
32 if (!needs_cmd_quoting(arg)) return arg;
33 // CMD has no real escape mechanism; backslash-escaping inner quotes is the
34 // best we can do (matches most user expectations).
35 std::string out = "\"";
36 for (char c : arg) {
37 if (c == '"') out += "\\\""; else out.push_back(c);
38 }
39 out.push_back('"');
40 return out;
41}

43std::string join_prefix_cmd(const std::vector<std::string>& args) {
44 if (args.empty()) return "";
45 std::string out;
46 for (auto& a : args) { out.push_back(' '); out += quote_cmd(a); }
47 return out;
48}

50void write_text_crlf(const fs::path& path, const std::string& content) {
51 // CRLF — .cmd parsers tolerate LF but produce subtle hangs on some
52 // versions of cmd.exe with embedded LF. CRLF is the safe choice.
53 std::ofstream out(path, std::ios::binary | std::ios::trunc);
54 out.write(content.data(), static_cast<std::streamsize>(content.size()));
55}

57// Read the shim table; returns empty map if file is missing or unparseable.
58// Must NOT throw — called from collision-detection paths during install.
59std::map<std::string, std::string> load_table() {
60 std::map<std::string, std::string> out;
61 std::error_code ec;
62 fs::path p = table_path();
63 if (!fs::exists(p, ec)) return out;
64 std::string text = file_util::read_text_no_bom(p);
65 if (text.empty()) return out;
66 json doc;
67 try {
68 doc = json::parse(text);
69 } catch (...) {
70 // Malformed table — treat as empty rather than blocking installs.
71 return out;
72 }
73 if (!doc.is_object()) return out;
74 for (auto& [k, v] : doc.items()) {
75 if (v.is_string()) out[k] = v.get<std::string>();
76 }
77 return out;
78}

87WriteResult write_shim(const std::string& alias,
88 const fs::path& exe,
89 const std::vector<std::string>& prefix_args,
90 bool force) {
91 std::error_code ec;
92 fs::create_directories(paths::bin_dir(), ec);
93
94 fs::path cmd_path = paths::bin_dir() / (alias + ".cmd");
95
96 // Collision check: bin_dir is shared (~/.local/bin alongside uv/pipx/etc.).
97 // If <alias>.cmd already exists and we're not its owner, refuse to clobber
98 // unless caller passed force=true. The shim table is our source of truth
99 // for "files we own".
100 if (!force && fs::exists(cmd_path, ec) && !is_managed(alias)) {
101 return WriteResult::Skipped;
102 }
103
104 std::ostringstream cmd;
105 cmd << "@echo off\r\n\""
106 << exe.string() << "\""
107 << join_prefix_cmd(prefix_args)
108 << " %*\r\n";
109 write_text_crlf(cmd_path, cmd.str());
110 return WriteResult::Wrote;
111}

113int remove_shim(const std::string& alias) {
114 std::error_code ec;
115 int n = 0;
116 // .cmd is the current format. .ps1 / extensionless / .exe are legacy or
117 // sibling artifacts we still want to clean up on removal.
118 for (auto suffix : {".cmd", ".ps1", "", ".exe"}) {
119 fs::path p = paths::bin_dir() / (alias + suffix);
120 if (fs::exists(p, ec) && fs::remove(p, ec)) ++n;
121 }
122 return n;
123}
