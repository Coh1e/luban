18std::map<std::string, std::string> extract_template_blocks(const std::string& rendered) {
19 std::map<std::string, std::string> out;
20 // Section names are slug-style: lowercase letters, digits, hyphens. The
21 // body is greedy-stopped at the first END marker (.*? non-greedy); we
22 // rely on the [\s\S] class because std::regex's `.` doesn't cross
23 // newlines by default.
24 std::regex re(R"(<!-- BEGIN luban-managed: ([a-z0-9-]+) -->\s*\n([\s\S]*?)<!-- END luban-managed -->)",
25 std::regex::ECMAScript);
26 auto begin = std::sregex_iterator(rendered.begin(), rendered.end(), re);
27 auto end = std::sregex_iterator();
28 for (auto it = begin; it != end; ++it) {
29 std::string section = (*it)[1].str();
30 std::string body = (*it)[2].str();
31 out[section] = std::move(body);
32 }
33 return out;
34}

36std::string sync_managed_blocks(const std::string& existing,
37 const std::map<std::string, std::string>& tpl_blocks) {
38 std::string out = existing;
39
40 for (auto& section : managed_section_order()) {
41 std::string begin_marker = "<!-- BEGIN luban-managed: " + section + " -->";
42 std::string end_marker = "<!-- END luban-managed -->";
43
44 // Find the next BEGIN for this specific section. We don't iterate
45 // multiple instances — duplicates would be a user authoring bug
46 // and silently rewriting all of them is more confusing than the
47 // single-instance contract.
48 size_t b = out.find(begin_marker);
49 if (b == std::string::npos) continue; // user removed → skip
50
51 // END marker must come AFTER the BEGIN. We pick the nearest END
52 // (so nested marker blocks don't break the algorithm).
53 size_t e = out.find(end_marker, b + begin_marker.size());
54 if (e == std::string::npos) {
55 log::warnf("AGENTS.md: '{}' has no matching END marker; skipping sync", section);
56 continue;
57 }
58
59 // Replace from end of BEGIN line through end of END marker exclusive
60 // — i.e., the body, preserving the markers themselves.
61 size_t body_start = b + begin_marker.size();
62 // Skip immediate trailing newline after BEGIN if present (so we keep
63 // one-line separation between marker and content). Handles both \n
64 // and \r\n (CRLF written by Windows editors).
65 if (body_start < out.size() && out[body_start] == '\n') {
66 body_start += 1;
67 } else if (body_start + 1 < out.size()
68 && out[body_start] == '\r' && out[body_start + 1] == '\n') {
69 body_start += 2;
70 }
71
72 auto it = tpl_blocks.find(section);
73 if (it == tpl_blocks.end()) continue; // template didn't have this section
74
75 // Replace [body_start, e) with the new content + a trailing \n
76 // before the END marker (so the END marker stays on its own line).
77 std::string new_body = it->second;
78 if (new_body.empty() || new_body.back() != '\n') new_body.push_back('\n');
79 out.replace(body_start, e - body_start, new_body);
80 }
81
82 return out;
83}
