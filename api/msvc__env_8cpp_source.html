26// Variables vcvarsall sets that are noisy / transient / not needed to
27// reproduce the build environment in a child process. We strip these from
28// the diff to keep the persisted file readable and to avoid confusing the
29// child shell with vcvars's internal bookkeeping.
30const std::set<std::string, std::less<>>& transient_vars() {
31 static const std::set<std::string, std::less<>> s = {
32 "PROMPT", // vcvarsall changes the prompt
33 "__VSCMD_PREINIT_PATH", // bookkeeping for the script itself
34 "__VSCMD_script_err_count",
35 "VSCMD_DEBUG",
36 "VS170COMNTOOLS", // version-specific aliases — not portable
37 };
38 return s;
39}

41// Run cmd.exe /c "<command>" and capture stdout. Caller is responsible for
42// quoting `command` (which is appended after `/c ` literally, so embedded
43// quotes need to be balanced). On failure returns empty string.
44std::string run_capture_cmd(const std::string& command) {
45#ifdef _WIN32
46 SECURITY_ATTRIBUTES sa{};
47 sa.nLength = sizeof(sa);
48 sa.bInheritHandle = TRUE;
49
50 HANDLE rd = nullptr, wr = nullptr;
51 if (!CreatePipe(&rd, &wr, &sa, 0)) return {};
52 if (!SetHandleInformation(rd, HANDLE_FLAG_INHERIT, 0)) {
53 CloseHandle(rd); CloseHandle(wr);
54 return {};
55 }
56
57 STARTUPINFOW si{};
58 si.cb = sizeof(si);
59 si.dwFlags = STARTF_USESTDHANDLES | STARTF_USESHOWWINDOW;
60 si.wShowWindow = SW_HIDE;
61 si.hStdOutput = wr;
62 si.hStdError = wr;
63 si.hStdInput = GetStdHandle(STD_INPUT_HANDLE);
64
65 PROCESS_INFORMATION pi{};
66 std::wstring full = L"cmd.exe /d /c " + win::from_utf8(command);
67 std::vector<wchar_t> mut(full.begin(), full.end());
68 mut.push_back(L'\0');
69
70 if (!CreateProcessW(nullptr, mut.data(), nullptr, nullptr, TRUE,
71 CREATE_NO_WINDOW, nullptr, nullptr, &si, &pi)) {
72 CloseHandle(rd); CloseHandle(wr);
73 return {};
74 }
75 CloseHandle(wr); // close write end so ReadFile sees EOF when child exits
76
77 std::string out;
78 char buf[4096];
79 DWORD n = 0;
80 while (ReadFile(rd, buf, sizeof(buf), &n, nullptr) && n > 0) {
81 out.append(buf, buf + n);
82 }
83 CloseHandle(rd);
84 WaitForSingleObject(pi.hProcess, INFINITE);
85 CloseHandle(pi.hProcess);
86 CloseHandle(pi.hThread);
87 return out;
88#else
89 (void)command;
90 return {}; // POSIX has no MSVC; caller already guards via Win32 ifdef
91#endif
92}

94// Parse the output of `set` (one VAR=VALUE per line) into a map. Skips
95// blank lines and lines without `=`. Trailing CR/LF stripped per-line.
96std::map<std::string, std::string> parse_set_output(const std::string& text) {
97 std::map<std::string, std::string> out;
98 std::istringstream iss(text);
99 std::string line;
100 while (std::getline(iss, line)) {
101 // Strip trailing \r (Windows cmd uses CRLF).
102 while (!line.empty() && (line.back() == '\r' || line.back() == '\n')) {
103 line.pop_back();
104 }
105 if (line.empty()) continue;
106 auto eq = line.find('=');
107 if (eq == std::string::npos) continue;
108 std::string key = line.substr(0, eq);
109 std::string val = line.substr(eq + 1);
110 if (key.empty()) continue;
111 out[std::move(key)] = std::move(val);
112 }
113 return out;
114}

116// Diff two PATH strings (semicolon-separated). Returns the leading
117// segment of `after` that isn't present anywhere in `before` — this is
118// what vcvarsall prepended. We assume vcvarsall ALWAYS prepends (it does
119// in practice; never appends or rewrites in-place).
120std::string path_prefix_diff(const std::string& before, const std::string& after) {
121 if (after.size() <= before.size()) return {};
122 // vcvarsall's pattern: "<msvc dirs>;<original PATH>". So if `after`
123 // ends with `;<before>`, the prefix is `after.substr(0, after.size() - before.size() - 1)`.
124 if (after.size() >= before.size() + 1
125 && after.compare(after.size() - before.size(), before.size(), before) == 0
126 && after[after.size() - before.size() - 1] == ';') {
127 return after.substr(0, after.size() - before.size() - 1);
128 }
129 // Fallback: no clean suffix match — return the full `after`. Caller will
130 // prepend the lot, accepting some PATH duplication; safer than dropping
131 // the MSVC dirs entirely.
132 return after;
133}

135// ISO-8601 timestamp (UTC, second precision). Used for the captured_at
136// metadata field — surfaces in `describe --host` so users can tell when
137// the capture happened (and re-capture if VS got upgraded since).
138std::string iso_now() {
139 using namespace std::chrono;
140 auto t = system_clock::to_time_t(system_clock::now());
141 std::tm gm{};
142#ifdef _WIN32
143 gmtime_s(&gm, &t);
144#else
145 gmtime_r(&t, &gm);
146#endif
147 std::ostringstream ss;
148 ss << std::put_time(&gm, "%Y-%m-%dT%H:%M:%SZ");
149 return ss.str();
150}

156fs::path find_vswhere() {
157#ifdef _WIN32
158 // Microsoft-documented well-known path. vswhere ships with every VS
159 // install (>=2017) at this exact location.
160 auto pf86 = paths::from_env("ProgramFiles(x86)");
161 if (!pf86) return {};

165#else

170fs::path find_install() {
171 fs::path vswhere = find_vswhere();
172 if (vswhere.empty()) return {};
173
174 // -latest pick newest install if multiple
175 // -property installationPath
176 // -products * accept Build Tools too (not just full VS)
177 std::string cmd = "\"" + vswhere.string() + "\" -latest -products * "
178 "-requires Microsoft.VisualStudio.Component.VC.Tools.x86.x64 "
179 "-property installationPath";
180 std::string out = run_capture_cmd(cmd);
181 while (!out.empty() && (out.back() == '\r' || out.back() == '\n' || out.back() == ' ')) {
182 out.pop_back();
183 }
184 if (out.empty()) return {};
185 return fs::path(out);
186}

195std::optional<Captured> capture(const fs::path& install_path, const std::string& arch) {
196 fs::path vcvars = find_vcvarsall(install_path);
197 if (vcvars.empty()) {
198 log::errf("vcvarsall.bat not found under {}", install_path.string());
199 return std::nullopt;
200 }
201
202 // Pass 1: baseline env. We use cmd's `set` builtin so we get exactly
203 // what cmd thinks is in the env (case-insensitive var names, etc.).
204 auto before = parse_set_output(run_capture_cmd("set"));
205
206 // Pass 2: env after vcvarsall. Quoting: the whole command has to be
207 // wrapped in double quotes for cmd /c, and inner quotes around paths
208 // with spaces. cmd's parser handles two leading quotes as "the outer
209 // pair" and lets the rest pass through.
210 std::string p2 = "\"\"" + vcvars.string() + "\" " + arch + " >nul && set\"";
211 auto after = parse_set_output(run_capture_cmd(p2));
212
213 if (after.empty()) {
214 log::err("vcvarsall produced no output — capture failed");
215 return std::nullopt;
216 }
217
218 Captured c;
219 c.vs_install_path = install_path.string();
220 c.arch = arch;
221 c.captured_at = iso_now();
222
223 // Diff: keys in `after` that are new or have different values than
224 // `before`, minus the transient bookkeeping vars.
225 auto& skip = transient_vars();
226 for (auto& [k, v] : after) {
227 if (skip.count(k)) continue;
228 if (k == "PATH" || k == "Path") continue; // handled separately
229 auto it = before.find(k);
230 if (it == before.end() || it->second != v) {
231 c.vars[k] = v;
232 }
233 }
234
235 // PATH delta: extract just the leading segment vcvarsall prepended.
236 std::string before_path, after_path;
237 auto find_path_value = [](const std::map<std::string, std::string>& m) -> std::string {
238 for (auto& [k, v] : m) {
239 // cmd's `set` uses the case-preserving form (usually "Path");
240 // accept both for portability.
241 if (k == "PATH" || k == "Path") return v;
242 }
243 return {};
244 };
245 before_path = find_path_value(before);
246 after_path = find_path_value(after);
247 c.path_addition = path_prefix_diff(before_path, after_path);
248
249 return c;
250}

252bool save(const Captured& c) {
253 json doc = json::object();
254 doc["schema"] = 1;
255 doc["vs_install_path"] = c.vs_install_path;
256 doc["arch"] = c.arch;
257 doc["captured_at"] = c.captured_at;
258 doc["path_addition"] = c.path_addition;
259 json kv = json::object();
260 for (auto& [k, v] : c.vars) kv[k] = v;
261 doc["vars"] = kv;
262
263 fs::path target = file_path();
264 std::string text = doc.dump(2);
265 text.push_back('\n'); // trailing newline matches prior behaviour
266 return file_util::write_text_atomic(target, text);
267}

269std::optional<Captured> load() {
270 fs::path p = file_path();
271 std::error_code ec;
272 if (!fs::exists(p, ec)) return std::nullopt;
273 std::string text = file_util::read_text_no_bom(p);
274 if (text.empty()) return std::nullopt;
275 json doc;
276 try { doc = json::parse(text); } catch (...) { return std::nullopt; }
277 if (!doc.is_object()) return std::nullopt;
278
279 Captured c;
280 c.vs_install_path = doc.value("vs_install_path", "");
281 c.arch = doc.value("arch", "");
282 c.captured_at = doc.value("captured_at", "");
283 c.path_addition = doc.value("path_addition", "");
284 if (doc.contains("vars") && doc["vars"].is_object()) {
285 for (auto& [k, v] : doc["vars"].items()) {
286 if (v.is_string()) c.vars[k] = v.get<std::string>();
287 }
288 }
289 return c;
290}
