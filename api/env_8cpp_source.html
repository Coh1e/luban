1// `luban env` — display environment state, register on HKCU (--user), or
2// emit eval-able env exports for the requested shell (--print).
3//
4// The `--apply` activate-scripts mode was removed in v0.2; --print covers
5// the same use case (CI / containers / one-off shells) without leaving
6// stale .cmd/.ps1/.sh files on disk.
7//
8// LUBAN_DATA/CACHE/STATE/CONFIG HKCU writes were also removed: they had
9// zero on-disk consumers and only created visual clutter in
10// HKCU\Environment. Today HKCU writes are limited to PATH + VCPKG_ROOT
11// (+ EM_CONFIG when emscripten is installed).

39// Quote helpers per shell. None of these are perfect (they assume the
40// values don't contain the delimiter being used), but luban-managed paths
41// don't have single/double quotes, so it works in practice.
42std::string quote_bash(const std::string& s) {
43 // Bash single-quoted: only ' itself needs special handling.
44 std::string out = "'";
45 for (char c : s) {
46 if (c == '\'') out += "'\"'\"'"; else out.push_back(c);
47 }
48 out.push_back('\'');
49 return out;
50}

52std::string quote_cmd(const std::string& s) {
53 // CMD: wrap whole "set" in double quotes so spaces / & / ; don't break.
54 // Per `set "NAME=value"` syntax, the value is everything between = and ".
55 std::string out;
56 for (char c : s) {
57 if (c == '%') out += "%%"; else out.push_back(c);
58 }
59 return out;
60}

62std::string quote_pwsh(const std::string& s) {
63 // PowerShell single-quoted: '' escapes a literal '. No backtick parsing.
64 std::string out = "'";
65 for (char c : s) {
66 if (c == '\'') out += "''"; else out.push_back(c);
67 }
68 out.push_back('\'');
69 return out;
70}

72// Emit eval-able env exports + PATH prepend for the requested shell. Reads
73// env_snapshot::path_dirs() (toolchain bin dirs + bin_dir) and env_dict()
74// (VCPKG_ROOT / vcpkg caches / EM_CONFIG) so the output mirrors exactly
75// what luban-spawned children see.
76//
77// Use case: CI scripts, one-off containers, fresh shells where running
78// `luban env --user` (which writes HKCU) is undesirable. Like
79// `eval "$(luban env --print --shell bash)"`.
80void print_shell_exports(std::string_view shell) {
81 auto dirs = env_snapshot::path_dirs();
82 auto extras = env_snapshot::env_dict();
83
84 if (shell == "bash") {
85 // POSIX-style. PATH separator is ':'; we convert backslashes to forward
86 // slashes since this output is meant for git-bash / WSL / msys2 / Linux.
87 std::string joined;
88 for (size_t i = 0; i < dirs.size(); ++i) {
89 if (i) joined.push_back(':');
90 std::string s = dirs[i].string();
91 for (auto& c : s) if (c == '\\') c = '/';
92 joined += s;
93 }
94 std::cout << "export PATH=" << quote_bash(joined + ":${PATH:-}") << "\n";
95 for (auto& [k, v] : extras) {
96 std::cout << "export " << k << "=" << quote_bash(v) << "\n";
97 }
98 } else if (shell == "cmd") {
99 // CMD batch. PATH separator is ';'. `set "name=value"` quotes the whole
100 // assignment so embedded spaces / & don't break.
101 std::string joined;
102 for (size_t i = 0; i < dirs.size(); ++i) {
103 if (i) joined.push_back(';');
104 joined += dirs[i].string();
105 }
106 std::cout << "set \"PATH=" << quote_cmd(joined) << ";%PATH%\"\r\n";
107 for (auto& [k, v] : extras) {
108 std::cout << "set \"" << k << "=" << quote_cmd(v) << "\"\r\n";
109 }
110 } else if (shell == "powershell" || shell == "pwsh") {
111 // PowerShell. Path separator follows OS; on Windows that's ';'.
112 std::string joined;
113 for (size_t i = 0; i < dirs.size(); ++i) {
114 if (i) joined.push_back(';');
115 joined += dirs[i].string();
116 }
117 std::cout << "$env:PATH = " << quote_pwsh(joined) << " + ';' + $env:PATH\r\n";
118 for (auto& [k, v] : extras) {
119 std::cout << "$env:" << k << " = " << quote_pwsh(v) << "\r\n";
120 }
121 } else {
122 log::errf("--shell {} unrecognized; supported: bash, cmd, powershell", shell);
123 }
124}

126int run_env(const cli::ParsedArgs& a) {
127 fs::path bin = paths::bin_dir();
128
129 auto get_flag = [&](const char* name) {
130 auto it = a.flags.find(name);
131 return it != a.flags.end() && it->second;
132 };
133 auto get_opt = [&](const char* name) -> std::string {
134 auto it = a.opts.find(name);
135 return (it != a.opts.end()) ? it->second : std::string();
136 };
137
138 // --print suppresses status output and emits eval-able exports only.
139 if (get_flag("print")) {
140 std::string shell = get_opt("shell");
141 if (shell.empty()) shell = "bash"; // default: most CI runs bash
142 print_shell_exports(shell);
143 return 0;
144 }
145
146 // --msvc-init: detect existing MSVC install via vswhere + capture
147 // vcvarsall env, persist to <state>/msvc-env.json. From then on,
148 // env_snapshot injects those vars into luban-spawned children, so
149 // `luban build` etc. can drive cl.exe without manual vcvarsall.
150 // We don't install MSVC ourselves — user has it via Microsoft's installer.
151 if (get_flag("msvc-init")) {
152 std::string arch = get_opt("arch");
153 if (arch.empty()) arch = "x64";
154
155 log::step("locating MSVC install via vswhere.exe");
156 fs::path vsw = msvc_env::find_vswhere();
157 if (vsw.empty()) {
158 log::err("vswhere.exe not found at the standard path");
159 log::info("install Visual Studio or VS Build Tools first; vswhere ships with both.");
160 return 1;
161 }
162 log::okf("found vswhere: {}", vsw.string());
163
164 fs::path install = msvc_env::find_install();
165 if (install.empty()) {
166 log::err("vswhere reported no VS install with VC.Tools.x86.x64");
167 log::info("install Visual Studio or VS Build Tools with the C++ workload first.");
168 return 1;
169 }
170 log::okf("found VS install: {}", install.string());
171
172 log::stepf("running vcvarsall {} and capturing env diff", arch);
173 auto cap = msvc_env::capture(install, arch);
174 if (!cap) {
175 log::err("vcvarsall capture failed");
176 return 1;
177 }
178 if (!msvc_env::save(*cap)) {
179 log::err("failed to write msvc-env.json");
180 return 1;
181 }
182 log::okf("captured {} env vars + path addition; saved to {}",
183 cap->vars.size(), msvc_env::file_path().string());
184 log::info("from now on, `luban build` / `luban run` see MSVC env automatically.");
185 return 0;
186 }
187
188 // --msvc-clear: forget the captured MSVC env. Use after uninstalling
189 // VS or to force a fresh capture next --msvc-init.
190 if (get_flag("msvc-clear")) {
191 msvc_env::clear();
192 log::okf("removed {}", msvc_env::file_path().string());
193 return 0;
194 }
195
196 log::infof("luban data dir : {}", paths::data_dir().string());
197 log::infof("PATH directory : {}", bin.string());
198
199 // --user: rustup-style. Add bin_dir() to HKCU PATH and set the env vars
200 // tools outside luban need to see (VCPKG_ROOT for cmake presets, EM_CONFIG
201 // for emcc). After this, any new shell sees luban's toolchain through PATH
202 // and these tools resolve their config without further activation.
203 if (get_flag("user")) {
204 if (win_path::add_to_user_path(bin)) {
205 log::okf("added {} to HKCU PATH", bin.string());
206 } else {
207 log::infof("{} already on HKCU PATH (no change)", bin.string());
208 }
209
210 auto recs = registry::load_installed();
211
212 // VCPKG_ROOT: CMakePresets.json uses
213 // $env{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake
214 // so the var must be in HKCU for users running cmake outside luban.
215 auto vcpkg_it = recs.find("vcpkg");
216 if (vcpkg_it != recs.end() && !vcpkg_it->second.toolchain_dir.empty()) {
217 fs::path vcpkg_root = paths::toolchain_dir(vcpkg_it->second.toolchain_dir);
218 if (win_path::set_user_env("VCPKG_ROOT", vcpkg_root.string())) {
219 log::okf("set HKCU VCPKG_ROOT = {}", vcpkg_root.string());
220 }
221 } else {
222 log::info("VCPKG_ROOT not set (no vcpkg in registry; run `luban setup --with vcpkg` first)");
223 }
224
225 // EM_CONFIG: emscripten's config file lives at <config>/emscripten/config
226 // (XDG-respecting, written by component.cpp). Setting EM_CONFIG lets
227 // users run emcc directly in any shell — no activation, no PATH dance.
228 if (recs.find("emscripten") != recs.end()) {
229 fs::path em_config = paths::config_dir() / "emscripten" / "config";
230 if (win_path::set_user_env("EM_CONFIG", em_config.string())) {
231 log::okf("set HKCU EM_CONFIG = {}", em_config.string());
232 }
233 }
234
235 // MSVC Phase 2: if the user previously ran `luban env --msvc-init`,
236 // write the captured INCLUDE / LIB / LIBPATH / WindowsSdk* / etc.
237 // (~35 vars) to HKCU. After this, fresh shells see cl.exe via the
238 // single PATH entry below — without going through `luban run`.
239 //
240 // PATH addition strategy: vcvarsall spits out ~14 dirs into PATH
241 // (MSVC compiler bin + VS IDE + Win SDK + MSBuild + perf tools).
242 // Adding all of them to HKCU PATH is unacceptable pollution. We
243 // add only the FIRST entry — by vcvarsall convention that's the
244 // MSVC HostX64\x64 dir which contains cl/link/lib/ml64/dumpbin/
245 // editbin/cvtres. Covers the standard "compile + link" workflow.
246 //
247 // For msbuild / rc / mt: user runs `luban run msbuild ...` (uses
248 // env_snapshot's full path injection); or hand-adds the SDK dir
249 // to HKCU PATH if they need fresh-shell access.
250 if (auto cap = msvc_env::load()) {
251 int count = 0;
252 for (auto& [k, v] : cap->vars) {
253 if (win_path::set_user_env(k, v)) ++count;
254 }
255 // First entry of path_addition only — see comment above.
256 std::string first_msvc_dir;
257 if (!cap->path_addition.empty()) {
258 size_t semi = cap->path_addition.find(';');
259 first_msvc_dir = (semi == std::string::npos)
260 ? cap->path_addition
261 : cap->path_addition.substr(0, semi);
262 }
263 if (!first_msvc_dir.empty()) {
264 win_path::add_to_user_path(first_msvc_dir);
265 log::okf("set {} MSVC env vars + {} on HKCU PATH",
266 count, first_msvc_dir);
267 } else {
268 log::okf("set {} MSVC env vars (no PATH addition)", count);
269 }
270 }
271
272 log::info("open a new shell for the changes to take effect.");
273 }
274
275 // --unset-user: reverse --user. bin_dir is luban-owned (<data>/bin),
276 // so safe to remove from HKCU PATH; users who want luban-managed
277 // toolchains in their shells should run --user again.
278 if (get_flag("unset-user")) {
279 if (win_path::remove_from_user_path(bin)) {
280 log::okf("removed {} from HKCU PATH", bin.string());
281 } else {
282 log::infof("{} not on HKCU PATH (nothing to do)", bin.string());
283 }
284 for (const char* name : kLegacyHKCUVars) {
285 win_path::unset_user_env(name);
286 }
287 win_path::unset_user_env("VCPKG_ROOT");
288 win_path::unset_user_env("EM_CONFIG");
289
290 // MSVC Phase 2: also unset the captured MSVC vars + remove the
291 // PATH entries luban prepended on `--user`. Reads msvc-env.json
292 // for the var names and PATH entry list rather than guessing —
293 // matches what --user wrote symmetrically.
294 if (auto cap = msvc_env::load()) {
295 for (auto& [k, _] : cap->vars) win_path::unset_user_env(k);
296 std::string s = cap->path_addition;
297 size_t start = 0;
298 int removed = 0;
299 while (start <= s.size()) {
300 size_t end = s.find(';', start);
301 if (end == std::string::npos) end = s.size();
302 std::string dir = s.substr(start, end - start);
303 if (!dir.empty() && win_path::remove_from_user_path(dir)) ++removed;
304 if (end == s.size()) break;
305 start = end + 1;
306 }
307 log::okf("removed {} MSVC env vars + {} PATH entries from HKCU",
308 cap->vars.size(), removed);
309 }
310
311 log::ok("removed legacy LUBAN_* / VCPKG_ROOT / EM_CONFIG / MSVC vars from HKCU env");
312 }
313
314 return 0;
315}

319void register_env() {
320 cli::Subcommand c;
321 c.name = "env";
322 c.help = "show env state; register bin_dir on HKCU PATH (rustup-style)";
323 c.group = "setup";
324 c.long_help =
325 " Manage environment integration. With no flags, prints the bin dir\n"
326 " luban writes toolchain shims into (<data>/bin, luban-owned).\n"
327 "\n"
328 " --user Add bin_dir() to HKCU PATH + set HKCU\n"
329 " VCPKG_ROOT (and EM_CONFIG if emscripten).\n"
330 " After this, any new shell can run cmake /\n"
331 " clang / ninja / etc. through luban-managed\n"
332 " shims. The dir is luban-only (cargo's\n"
333 " ~/.cargo/bin/ pattern); luban.exe itself\n"
334 " lives under ~/.local/bin/ (XDG).\n"
335 " --unset-user Reverse --user. Also cleans up legacy\n"
336 " LUBAN_* HKCU env vars from pre-v0.2 installs.\n"
337 " --print [--shell X] Emit eval-able env exports for shell X\n"
338 " (bash | cmd | powershell). Default: bash.\n"
339 " For CI / containers / fresh shells where\n"
340 " writing HKCU is undesirable. Example:\n"
341 " eval \"$(luban env --print --shell bash)\"\n"
342 " --msvc-init [--arch X] Detect existing MSVC install (via vswhere)\n"
343 " and capture vcvarsall env into\n"
344 " <state>/msvc-env.json. After this:\n"
345 " * luban-spawned cmake/cl.exe just works\n"
346 " (in-process injection)\n"
347 " * the next `luban env --user` writes the\n"
348 " ~30 captured vars + MSVC PATH dirs to\n"
349 " HKCU so fresh shells see cl.exe too\n"
350 " Default --arch is x64.\n"
351 " --msvc-clear Forget the captured MSVC env (also makes\n"
352 " the next --user / --unset-user skip MSVC\n"
353 " vars).";
354 c.flags = {"user", "unset-user", "print", "msvc-init", "msvc-clear"};
355 c.opts = {{"shell", ""}, {"arch", ""}};
356 c.examples = {
357 "luban env\tShow current env state",
358 "luban env --user\tOne-time HKCU PATH registration (rustup-style)",
359 "luban env --unset-user\tUndo the above + clean legacy vars",
360 "luban env --print --shell bash\tEmit `export ...` lines for `eval`",
361 "luban env --msvc-init\tCapture vcvarsall env (requires VS installed)",
362 };
363 c.run = run_env;
364 cli::register_subcommand(std::move(c));
365}
