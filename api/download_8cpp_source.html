35// ---- Progress bar (stderr, TTY only) ----
36// Cross-platform: uses cstdio + chrono only. Both Win32 (do_request) and
37// POSIX (libcurl-backed download) populate it via the sink callback.
38struct Progress {
39 std::string label;
40 int64_t total = -1; // -1 = 未知（Content-Length 缺失）
41 int64_t done = 0;
42 std::chrono::steady_clock::time_point t0;
43 std::chrono::steady_clock::time_point last;
44 bool enabled = false;
45
46 Progress(std::string lbl, int64_t total_bytes) : label(std::move(lbl)), total(total_bytes) {
47 t0 = std::chrono::steady_clock::now();
48 last = t0;
49 enabled = _isatty(_fileno(stderr)) && std::getenv("LUBAN_NO_PROGRESS") == nullptr;
50 }
51
52 static std::string format_bytes(int64_t n) {
53 const char* units[] = {"B", "KiB", "MiB", "GiB"};
54 double f = static_cast<double>(n);
55 int u = 0;
56 while (f >= 1024.0 && u < 3) { f /= 1024.0; ++u; }
57 if (u == 0) return std::format("{} B", n);
58 return std::format("{:.1f} {}", f, units[u]);
59 }
60
61 void update(size_t n) {
62 done += static_cast<int64_t>(n);
63 if (!enabled) return;
64 auto now = std::chrono::steady_clock::now();
65 double dt = std::chrono::duration<double>(now - last).count();
66 bool finished = (total > 0 && done >= total);
67 if (dt < 0.1 && !finished) return;
68 last = now;
69 render(now);
70 }
71
72 void render(std::chrono::steady_clock::time_point now) {
73 double elapsed = std::chrono::duration<double>(now - t0).count();
74 if (elapsed < 1e-3) elapsed = 1e-3;
75 double rate = static_cast<double>(done) / elapsed;
76 std::string rate_str = format_bytes(static_cast<int64_t>(rate)) + "/s";
77 std::string line;
78 if (total > 0) {
79 double pct = 100.0 * static_cast<double>(done) / static_cast<double>(total);
80 constexpr int W = 24;
81 int filled = static_cast<int>(W * static_cast<double>(done) / total);
82 std::string bar;
83 for (int i = 0; i < W; ++i) bar += (i < filled ? "#" : "\xc2\xb7");
84 line = std::format(" [{}] {:5.1f}% {}/{} {} {}",
85 bar, pct, format_bytes(done), format_bytes(total),
86 rate_str, label);
87 } else {
88 line = std::format(" {} {} {}", format_bytes(done), rate_str, label);
89 }
90 std::fprintf(stderr, "\r%s ", line.c_str());
91 std::fflush(stderr);
92 }
93
94 void finish() {
95 if (!enabled) return;
96 std::fprintf(stderr, "\r%-100s\r", "");
97 std::fflush(stderr);
98 }
99};

117// ---- BCrypt 流式 SHA256 ----
118struct StreamingSha256 {
119 BCRYPT_ALG_HANDLE alg = nullptr;
120 BCRYPT_HASH_HANDLE h = nullptr;
121 bool ok = false;
122
123 StreamingSha256() {
124 if (BCryptOpenAlgorithmProvider(&alg, BCRYPT_SHA256_ALGORITHM, nullptr, 0) != 0) return;
125 if (BCryptCreateHash(alg, &h, nullptr, 0, nullptr, 0, 0) != 0) {
126 BCryptCloseAlgorithmProvider(alg, 0);
127 alg = nullptr;
128 return;
129 }
130 ok = true;
131 }
132 ~StreamingSha256() {
133 if (h) BCryptDestroyHash(h);
134 if (alg) BCryptCloseAlgorithmProvider(alg, 0);
135 }
136 void update(const unsigned char* data, ULONG n) {
137 if (ok) BCryptHashData(h, const_cast<PUCHAR>(data), n, 0);
138 }
139 std::string finish() {
140 if (!ok) return {};
141 unsigned char digest[32];
142 if (BCryptFinishHash(h, digest, sizeof(digest), 0) != 0) return {};
143 static constexpr char hexc[] = "0123456789abcdef";
144 std::string out(64, '\0');
145 for (int i = 0; i < 32; ++i) {
146 out[2*i] = hexc[(digest[i] >> 4) & 0xF];
147 out[2*i + 1] = hexc[digest[i] & 0xF];
148 }
149 return out;
150 }

159// 解析 url（只支持 https://host[:port]/path）—— manifest URL 都满足这个。
160struct ParsedUrl {
161 std::wstring host;
162 INTERNET_PORT port = INTERNET_DEFAULT_HTTPS_PORT;
163 std::wstring path;
164 bool secure = true;
165};
166
167std::optional<ParsedUrl> parse_url(const std::string& url) {
168 URL_COMPONENTS uc{};
169 uc.dwStructSize = sizeof(uc);
170 wchar_t host[256] = {0};
171 wchar_t path[2048] = {0};
172 uc.lpszHostName = host;
173 uc.dwHostNameLength = static_cast<DWORD>(std::size(host));
174 uc.lpszUrlPath = path;
175 uc.dwUrlPathLength = static_cast<DWORD>(std::size(path));
176
177 std::wstring wurl = win::from_utf8(url);
178 if (!WinHttpCrackUrl(wurl.c_str(), 0, 0, &uc)) return std::nullopt;
179 if (uc.nScheme != INTERNET_SCHEME_HTTPS && uc.nScheme != INTERNET_SCHEME_HTTP)
180 return std::nullopt;
181 ParsedUrl out;
182 out.host = std::wstring(host, uc.dwHostNameLength);
183 out.path = std::wstring(path, uc.dwUrlPathLength);
184 out.port = uc.nPort;
185 out.secure = (uc.nScheme == INTERNET_SCHEME_HTTPS);
186 if (out.path.empty()) out.path = L"/";
187 return out;
188}

190// 一次完整 HTTP 请求；按需 dump 到 sink（写文件 + 哈希 + 进度）或者 buffer（fetch_text）。
191template <class Sink>
192std::expected<int64_t, Error> do_request(
193 const std::string& url, int timeout_seconds, Sink&& sink) {
194 auto p = parse_url(url);
195 if (!p) return std::unexpected(Error{ErrorKind::Network, "invalid URL: " + url});
196
197 HttpHandle session;
198 session.h = WinHttpOpen(user_agent().c_str(),
199 WINHTTP_ACCESS_TYPE_AUTOMATIC_PROXY,
200 WINHTTP_NO_PROXY_NAME, WINHTTP_NO_PROXY_BYPASS, 0);
201 if (!session.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpOpen failed"});
202
203 int ms = timeout_seconds * 1000;
204 WinHttpSetTimeouts(session.h, ms, ms, ms, ms);
205
206 HttpHandle conn;
207 conn.h = WinHttpConnect(session.h, p->host.c_str(), p->port, 0);
208 if (!conn.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpConnect failed"});
209
210 HttpHandle req;
211 DWORD flags = p->secure ? WINHTTP_FLAG_SECURE : 0;
212 req.h = WinHttpOpenRequest(conn.h, L"GET", p->path.c_str(), nullptr,
213 WINHTTP_NO_REFERER, WINHTTP_DEFAULT_ACCEPT_TYPES, flags);
214 if (!req.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpOpenRequest failed"});
215
216 // 关闭证书检查的"放任"开关——默认 winhttp 会验证 cert chain（系统 CA store），
217 // 这就是我们想要的安全默认。
218
219 // 跟随重定向（默认行为，但显式声明）。
220 DWORD redir = WINHTTP_OPTION_REDIRECT_POLICY_ALWAYS;
221 WinHttpSetOption(req.h, WINHTTP_OPTION_REDIRECT_POLICY, &redir, sizeof(redir));
222
223 if (!WinHttpSendRequest(req.h, WINHTTP_NO_ADDITIONAL_HEADERS, 0,
224 WINHTTP_NO_REQUEST_DATA, 0, 0, 0)) {
225 return std::unexpected(Error{ErrorKind::Network, "WinHttpSendRequest failed"});
226 }
227 if (!WinHttpReceiveResponse(req.h, nullptr)) {
228 return std::unexpected(Error{ErrorKind::Network, "WinHttpReceiveResponse failed"});
229 }
230
231 DWORD status = 0;
232 DWORD slen = sizeof(status);
233 WinHttpQueryHeaders(req.h, WINHTTP_QUERY_STATUS_CODE | WINHTTP_QUERY_FLAG_NUMBER,
234 WINHTTP_HEADER_NAME_BY_INDEX, &status, &slen,
235 WINHTTP_NO_HEADER_INDEX);
236 if (status >= 400) {
237 ErrorKind k = (status < 500) ? ErrorKind::HttpClient : ErrorKind::HttpServer;
238 return std::unexpected(Error{k, std::format("HTTP {} for {}", status, url)});
239 }
240
241 int64_t total = -1;
242 {
243 wchar_t cl[64] = {0};
244 DWORD cllen = sizeof(cl);
245 if (WinHttpQueryHeaders(req.h, WINHTTP_QUERY_CONTENT_LENGTH,
246 WINHTTP_HEADER_NAME_BY_INDEX, cl, &cllen,
247 WINHTTP_NO_HEADER_INDEX)) {
248 try { total = std::stoll(win::to_utf8(cl)); } catch (...) { total = -1; }
249 }
250 }
251
252 int64_t done = 0;
253 std::vector<unsigned char> buf(kChunk);
254 while (true) {
255 DWORD avail = 0;
256 if (!WinHttpQueryDataAvailable(req.h, &avail))
257 return std::unexpected(Error{ErrorKind::Network, "WinHttpQueryDataAvailable failed"});
258 if (avail == 0) break;
259 if (avail > buf.size()) buf.resize(avail);
260 DWORD got = 0;
261 if (!WinHttpReadData(req.h, buf.data(), avail, &got))
262 return std::unexpected(Error{ErrorKind::Network, "WinHttpReadData failed"});
263 if (got == 0) break;
264 if (auto e = sink(buf.data(), got, total); !e.has_value()) return std::unexpected(e.error());
265 done += got;
266 }
267 return done;
268}

270// ---- HEAD probe + chunked Range download (Phase 14b) -----------------------
271//
272// Decision tree (in download() below):
273// parallel_chunks <= 1 → single-stream (do_request)
274// HEAD fails OR no Content-Length → single-stream (caller doesn't know size)
275// no Accept-Ranges: bytes header → single-stream (server can't slice)
276// Content-Length < threshold → single-stream (overhead > benefit)
277// else → download_chunked
278//
279// Chunked path opens one WinHttp session per worker thread (handles aren't
280// thread-safe), each fetches `Range: bytes=<lo>-<hi>` and writes to its slice
281// of a pre-allocated output file. Hash verification runs once at the end via
282// hash::verify_file (re-reads the file) since concurrent writers can't share
283// a streaming-hash state.
284
285struct HeadInfo {
286 int64_t content_length = -1; // -1 if header missing / unparseable
287 bool accepts_ranges = false; // true iff `Accept-Ranges: bytes`
288};
289
290// HEAD probe. Same WinHttp shape as do_request but with method=HEAD and we
291// only inspect headers (no body to read). Failures yield `nullopt` so the
292// caller can gracefully fall back to single-stream.
293std::optional<HeadInfo> head_info(const std::string& url, int timeout_seconds) {
294 auto p = parse_url(url);
295 if (!p) return std::nullopt;
296
297 HttpHandle session;
298 session.h = WinHttpOpen(user_agent().c_str(), WINHTTP_ACCESS_TYPE_AUTOMATIC_PROXY,
299 WINHTTP_NO_PROXY_NAME, WINHTTP_NO_PROXY_BYPASS, 0);
300 if (!session.h) return std::nullopt;
301 int ms = timeout_seconds * 1000;
302 WinHttpSetTimeouts(session.h, ms, ms, ms, ms);
303
304 HttpHandle conn;
305 conn.h = WinHttpConnect(session.h, p->host.c_str(), p->port, 0);
306 if (!conn.h) return std::nullopt;
307
308 HttpHandle req;
309 DWORD flags = p->secure ? WINHTTP_FLAG_SECURE : 0;
310 req.h = WinHttpOpenRequest(conn.h, L"HEAD", p->path.c_str(), nullptr,
311 WINHTTP_NO_REFERER, WINHTTP_DEFAULT_ACCEPT_TYPES, flags);
312 if (!req.h) return std::nullopt;
313
314 DWORD redir = WINHTTP_OPTION_REDIRECT_POLICY_ALWAYS;
315 WinHttpSetOption(req.h, WINHTTP_OPTION_REDIRECT_POLICY, &redir, sizeof(redir));
316
317 if (!WinHttpSendRequest(req.h, WINHTTP_NO_ADDITIONAL_HEADERS, 0,
318 WINHTTP_NO_REQUEST_DATA, 0, 0, 0))
319 return std::nullopt;
320 if (!WinHttpReceiveResponse(req.h, nullptr)) return std::nullopt;
321
322 DWORD status = 0;
323 DWORD slen = sizeof(status);
324 WinHttpQueryHeaders(req.h, WINHTTP_QUERY_STATUS_CODE | WINHTTP_QUERY_FLAG_NUMBER,
325 WINHTTP_HEADER_NAME_BY_INDEX, &status, &slen,
326 WINHTTP_NO_HEADER_INDEX);
327 if (status >= 400) return std::nullopt;
328
329 HeadInfo h;
330 {
331 wchar_t cl[64] = {0};
332 DWORD cllen = sizeof(cl);
333 if (WinHttpQueryHeaders(req.h, WINHTTP_QUERY_CONTENT_LENGTH,
334 WINHTTP_HEADER_NAME_BY_INDEX, cl, &cllen,
335 WINHTTP_NO_HEADER_INDEX)) {
336 try { h.content_length = std::stoll(win::to_utf8(cl)); }
337 catch (...) { h.content_length = -1; }
338 }
339 }
340 {
341 wchar_t ar[64] = {0};
342 DWORD arlen = sizeof(ar);
343 // WINHTTP_QUERY_ACCEPT_RANGES = 0x0027 — well-known header index.
344 if (WinHttpQueryHeaders(req.h, WINHTTP_QUERY_ACCEPT_RANGES,
345 WINHTTP_HEADER_NAME_BY_INDEX, ar, &arlen,
346 WINHTTP_NO_HEADER_INDEX)) {
347 std::string v = win::to_utf8(std::wstring(ar, arlen / sizeof(wchar_t)));
348 // "bytes" or "bytes,foo" → supports byte ranges. "none" → no.
349 for (auto& c : v) c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
350 if (v.find("bytes") != std::string::npos) h.accepts_ranges = true;
351 }
352 }
353 return h;
354}

356// Fetch one byte range [lo, hi] (inclusive on both ends per RFC 7233) and
357// write it directly into `dest` at offset `lo`. Each call creates its own
358// WinHttp + file handles so it's safe to run from multiple threads.
359//
360// Returns the number of bytes actually written, or an error.
361std::expected<int64_t, Error> download_range_to_file(
362 const std::string& url, const fs::path& dest,
363 int64_t lo, int64_t hi, int timeout_seconds)
364{
365 auto p = parse_url(url);
366 if (!p) return std::unexpected(Error{ErrorKind::Network, "invalid URL: " + url});
367
368 HttpHandle session;
369 session.h = WinHttpOpen(user_agent().c_str(), WINHTTP_ACCESS_TYPE_AUTOMATIC_PROXY,
370 WINHTTP_NO_PROXY_NAME, WINHTTP_NO_PROXY_BYPASS, 0);
371 if (!session.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpOpen failed"});
372 int ms = timeout_seconds * 1000;
373 WinHttpSetTimeouts(session.h, ms, ms, ms, ms);
374
375 HttpHandle conn;
376 conn.h = WinHttpConnect(session.h, p->host.c_str(), p->port, 0);
377 if (!conn.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpConnect failed"});
378
379 HttpHandle req;
380 DWORD flags = p->secure ? WINHTTP_FLAG_SECURE : 0;
381 req.h = WinHttpOpenRequest(conn.h, L"GET", p->path.c_str(), nullptr,
382 WINHTTP_NO_REFERER, WINHTTP_DEFAULT_ACCEPT_TYPES, flags);
383 if (!req.h) return std::unexpected(Error{ErrorKind::Network, "WinHttpOpenRequest failed"});
384
385 DWORD redir = WINHTTP_OPTION_REDIRECT_POLICY_ALWAYS;
386 WinHttpSetOption(req.h, WINHTTP_OPTION_REDIRECT_POLICY, &redir, sizeof(redir));
387
388 // Range header. RFC 7233 inclusive on both sides.
389 std::wstring range_hdr = L"Range: bytes=" +
390 std::to_wstring(lo) + L"-" + std::to_wstring(hi);
391 if (!WinHttpAddRequestHeaders(req.h, range_hdr.c_str(),
392 static_cast<DWORD>(range_hdr.size()),
393 WINHTTP_ADDREQ_FLAG_ADD)) {
394 return std::unexpected(Error{ErrorKind::Network, "WinHttpAddRequestHeaders Range failed"});
395 }
396
397 if (!WinHttpSendRequest(req.h, WINHTTP_NO_ADDITIONAL_HEADERS, 0,
398 WINHTTP_NO_REQUEST_DATA, 0, 0, 0))
399 return std::unexpected(Error{ErrorKind::Network, "WinHttpSendRequest failed"});
400 if (!WinHttpReceiveResponse(req.h, nullptr))
401 return std::unexpected(Error{ErrorKind::Network, "WinHttpReceiveResponse failed"});
402
403 DWORD status = 0;
404 DWORD slen = sizeof(status);
405 WinHttpQueryHeaders(req.h, WINHTTP_QUERY_STATUS_CODE | WINHTTP_QUERY_FLAG_NUMBER,
406 WINHTTP_HEADER_NAME_BY_INDEX, &status, &slen,
407 WINHTTP_NO_HEADER_INDEX);
408 // Expected: 206 Partial Content. Some servers return 200 if the range
409 // covers the whole file — accept that too.
410 if (status != 206 && status != 200) {
411 ErrorKind k = (status < 500) ? ErrorKind::HttpClient : ErrorKind::HttpServer;
412 return std::unexpected(Error{k,
413 std::format("range request returned HTTP {} (expected 206)", status)});
414 }
415
416 // Open per-thread file handle. FILE_SHARE_WRITE lets sibling threads
417 // also have their own handle open at disjoint offsets; NTFS handles the
418 // disjoint write parallelism correctly without further synchronization.
419 HANDLE fh = CreateFileW(dest.wstring().c_str(),
420 GENERIC_WRITE,
421 FILE_SHARE_READ | FILE_SHARE_WRITE,
422 nullptr,
423 OPEN_EXISTING,
424 FILE_ATTRIBUTE_NORMAL,
425 nullptr);
426 if (fh == INVALID_HANDLE_VALUE) {
427 return std::unexpected(Error{ErrorKind::Io, "CreateFile (chunk dest) failed"});
428 }
429
430 LARGE_INTEGER off; off.QuadPart = lo;
431 if (!SetFilePointerEx(fh, off, nullptr, FILE_BEGIN)) {
432 CloseHandle(fh);
433 return std::unexpected(Error{ErrorKind::Io, "SetFilePointerEx failed"});
434 }
435
436 int64_t written = 0;
437 std::vector<unsigned char> buf(kChunk);
438 while (true) {
439 DWORD avail = 0;
440 if (!WinHttpQueryDataAvailable(req.h, &avail)) {
441 CloseHandle(fh);
442 return std::unexpected(Error{ErrorKind::Network, "WinHttpQueryDataAvailable failed"});
443 }
444 if (avail == 0) break;
445 if (avail > buf.size()) buf.resize(avail);
446 DWORD got = 0;
447 if (!WinHttpReadData(req.h, buf.data(), avail, &got)) {
448 CloseHandle(fh);
449 return std::unexpected(Error{ErrorKind::Network, "WinHttpReadData failed"});
450 }
451 if (got == 0) break;
452 DWORD wrote = 0;
453 if (!WriteFile(fh, buf.data(), got, &wrote, nullptr) || wrote != got) {
454 CloseHandle(fh);
455 return std::unexpected(Error{ErrorKind::Io, "WriteFile (chunk) failed"});
456 }
457 written += wrote;
458 }
459 CloseHandle(fh);
460 return written;
461}

463// Pre-allocate `dest` to `total` bytes so each thread can SetFilePointerEx +
464// WriteFile into its slice without extending the file. SetEndOfFile sets
465// physical size; allocation is sparse on NTFS (no zero-fill cost).
466bool preallocate_file(const fs::path& dest, int64_t total) {
467 HANDLE fh = CreateFileW(dest.wstring().c_str(),
468 GENERIC_WRITE,
469 FILE_SHARE_READ | FILE_SHARE_WRITE,
470 nullptr,
471 CREATE_ALWAYS,
472 FILE_ATTRIBUTE_NORMAL,
473 nullptr);
474 if (fh == INVALID_HANDLE_VALUE) return false;
475 LARGE_INTEGER eof; eof.QuadPart = total;
476 bool ok = SetFilePointerEx(fh, eof, nullptr, FILE_BEGIN) && SetEndOfFile(fh);
477 CloseHandle(fh);
478 return ok;
479}

481// Orchestrate N parallel range fetches into `dest`. Caller has already
482// confirmed the server supports byte ranges and `total` is the
483// Content-Length. Returns total bytes written or first error.
484std::expected<int64_t, Error> download_chunked(
485 const std::string& url, const fs::path& dest,
486 int64_t total, int n, int timeout_seconds, const std::string& label)
487{
488 if (n < 2) n = 2;
489 if (n > 16) n = 16; // sanity cap; CDNs throttle anyway after 4-8 conns
490
491 if (!preallocate_file(dest, total)) {
492 return std::unexpected(Error{ErrorKind::Io, "could not preallocate output file"});
493 }
494
495 // Compute disjoint inclusive ranges. Last chunk takes any remainder.
496 struct Range { int64_t lo, hi; };
497 std::vector<Range> ranges;
498 int64_t per = total / n;
499 for (int i = 0; i < n; ++i) {
500 int64_t lo = i * per;
501 int64_t hi = (i == n - 1) ? (total - 1) : (lo + per - 1);
502 ranges.push_back({lo, hi});
503 }
504
505 std::vector<std::expected<int64_t, Error>> results(n);
506 std::vector<std::thread> threads;
507 threads.reserve(n);
508 std::atomic<int64_t> bytes_done{0};
509
510 log::stepf("chunked download: {} ({}, {} threads × ~{} MiB)",
511 label, url, n, (per + 1024 * 1024 - 1) / (1024 * 1024));
512
513 for (int i = 0; i < n; ++i) {
514 threads.emplace_back([&, i] {
515 results[i] = download_range_to_file(url, dest, ranges[i].lo, ranges[i].hi,
516 timeout_seconds);
517 if (results[i].has_value()) bytes_done.fetch_add(*results[i]);
518 });
519 }
520 for (auto& t : threads) t.join();
521
522 int64_t total_written = 0;
523 for (int i = 0; i < n; ++i) {
524 if (!results[i].has_value()) return std::unexpected(results[i].error());
525 total_written += *results[i];
526 }
527 if (total_written != total) {
528 return std::unexpected(Error{ErrorKind::Network,
529 std::format("chunked total mismatch: wrote {} bytes, expected {}",
530 total_written, total)});
531 }
532 return total_written;
533}

538// Common post-download finalization: verify hash + atomic rename to dest.
539// Both the chunked and single-stream branches need exactly this sequence
540// after writing tmp; extracting it into a helper removes ~30 lines of
541// near-duplicate code from each branch (F6).
542//
543// `precomputed_sha256_hex` lets the single-stream path pass its
544// streaming-computed sha256 to skip a re-read; the chunked path leaves it
545// nullopt because concurrent writers can't share a streaming hash state,
546// so it always recomputes via hash::hash_file. If `expected_hash` is set
547// to a non-sha256 algo, we re-read regardless.
548//
549// On any failure, tmp is removed and the original Error is returned.
550// On success, tmp has been moved to dest and its sha256 HashSpec is returned.
551//
552// Cross-platform: uses hash::hash_file (POSIX impl via OpenSSL EVP) +
553// std::filesystem only.
554std::expected<hash::HashSpec, Error> verify_and_promote(
555 const fs::path& tmp,
556 const fs::path& dest,
557 const std::optional<hash::HashSpec>& expected,
558 const std::string& url,
559 const std::optional<std::string>& precomputed_sha256_hex = std::nullopt) {
560 std::error_code ec;
561
562 // Compute (or accept) the sha256 of the tmp file.
563 std::string sha256_hex;
564 if (precomputed_sha256_hex) {
565 sha256_hex = *precomputed_sha256_hex;
566 } else {
567 auto sha = hash::hash_file(tmp, hash::Algorithm::Sha256);
568 if (!sha) {
569 fs::remove(tmp, ec);
570 return std::unexpected(Error{ErrorKind::Io, "post-download hash_file failed"});
571 }
572 sha256_hex = sha->hex;
573 }
574
575 // Hash check against expected (if caller specified one).
576 if (expected) {
577 const auto& exp = *expected;
578 std::string actual_hex;
579 if (exp.algo == hash::Algorithm::Sha256) {
580 actual_hex = sha256_hex;
581 } else {
582 // Non-sha256 expected algo — re-read the file with that algo.
583 auto h = hash::hash_file(tmp, exp.algo);
584 actual_hex = h ? h->hex : "";
585 }
586 if (actual_hex != exp.hex) {
587 fs::remove(tmp, ec);
588 return std::unexpected(Error{
589 ErrorKind::HashMismatch,
590 std::format("hash mismatch for {}\n expected {}:{}\n actual {}:{}",
591 url, hash::algo_name(exp.algo), exp.hex,
592 hash::algo_name(exp.algo), actual_hex)});
593 }
594 }
595
596 // Atomic rename tmp → dest with cross-volume copy fallback.
597 fs::remove(dest, ec);
598 fs::rename(tmp, dest, ec);
599 if (ec) {
600 ec.clear();
601 fs::copy_file(tmp, dest, fs::copy_options::overwrite_existing, ec);
602 std::error_code rm_ec;
603 fs::remove(tmp, rm_ec);
604 if (ec) return std::unexpected(Error{ErrorKind::Io, "rename/copy failed"});
605 }
606 return hash::HashSpec{hash::Algorithm::Sha256, sha256_hex};
607}

621 // UTF-8 since libcurl's CURLOPT_USERAGENT wants `const char*`.
622 static const std::string ua =

627// libcurl write callback. userdata points to a Sink&-typed lambda
628// stored in CurlSinkCtx. Returning fewer bytes than supplied aborts
629// the transfer (libcurl maps that to CURLE_WRITE_ERROR).
630template <class Sink>
631struct CurlSinkCtx {
632 Sink* sink;
633 int64_t total = -1;
634 std::optional<Error> err;
635};
636
637template <class Sink>
638size_t curl_write_cb(char* ptr, size_t size, size_t nmemb, void* userdata) {
639 auto* c = static_cast<CurlSinkCtx<Sink>*>(userdata);
640 size_t n = size * nmemb;
641 auto rc = (*c->sink)(reinterpret_cast<const unsigned char*>(ptr), n, c->total);
642 if (!rc) { c->err = rc.error(); return 0; }
643 return n;
644}
645
646template <class Sink>
647std::expected<int64_t, Error> do_request(
648 const std::string& url, int timeout_seconds, Sink&& sink) {
649 CURL* curl = curl_easy_init();
650 if (!curl) return std::unexpected(Error{ErrorKind::Network, "curl_easy_init failed"});
651
652 CurlSinkCtx<Sink> ctx{&sink, -1, std::nullopt};
653
654 curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
655 curl_easy_setopt(curl, CURLOPT_USERAGENT, user_agent_utf8().c_str());
656 curl_easy_setopt(curl, CURLOPT_TIMEOUT, static_cast<long>(timeout_seconds));
657 curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
658 curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 1L);
659 curl_easy_setopt(curl, CURLOPT_FAILONERROR, 0L); // we inspect status ourselves
660 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, curl_write_cb<Sink>);
661 curl_easy_setopt(curl, CURLOPT_WRITEDATA, &ctx);
662
663 // Probe Content-Length up front via HEAD-equivalent isn't done here —
664 // libcurl gives us CURLINFO_CONTENT_LENGTH_DOWNLOAD_T after perform.
665 // For Progress display, the sink can read total from the ctx if we
666 // populate it in a header-callback; keeping minimal for now.
667
668 CURLcode rc = curl_easy_perform(curl);
669
670 long http_code = 0;
671 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &http_code);
672 curl_off_t bytes_dl = 0;
673 curl_easy_getinfo(curl, CURLINFO_SIZE_DOWNLOAD_T, &bytes_dl);
674
675 curl_easy_cleanup(curl);
676
677 if (rc != CURLE_OK) {
678 // Sink-injected error wins (more specific than libcurl's generic
679 // CURLE_WRITE_ERROR when the sink itself rejected a chunk).
680 if (ctx.err) return std::unexpected(*ctx.err);
681 return std::unexpected(Error{
682 ErrorKind::Network,
683 std::format("libcurl: {} (curl error {})",
684 curl_easy_strerror(rc), static_cast<int>(rc))});
685 }
686 if (http_code >= 400 && http_code < 500) {
687 return std::unexpected(Error{
688 ErrorKind::HttpClient,
689 std::format("HTTP {} for {}", http_code, url)});
690 }
691 if (http_code >= 500) {
692 return std::unexpected(Error{
693 ErrorKind::Network,
694 std::format("HTTP {} for {}", http_code, url)});
695 }
696 return static_cast<int64_t>(bytes_dl);
697}

702std::expected<DownloadResult, Error> download(
703 const std::string& url, const fs::path& dest, const DownloadOptions& opts) {
704#ifdef _WIN32
705 std::error_code ec;
706 fs::create_directories(dest.parent_path(), ec);
707
708 std::string label = opts.label.empty()
709 ? (url.rfind('/') != std::string::npos ? url.substr(url.rfind('/') + 1) : url)
710 : opts.label;
711
712 // Chunked-download decision. We do this BEFORE the retry loop because a
713 // failed HEAD or unsupported Range never recovers on retry — we just
714 // fall back to single-stream and start the loop with that strategy.
715 bool use_chunked = false;
716 int64_t known_size = -1;
717 if (opts.parallel_chunks > 1) {
718 if (auto h = head_info(url, opts.timeout_seconds)) {
719 if (h->accepts_ranges
720 && h->content_length >= opts.chunk_threshold) {
721 use_chunked = true;
722 known_size = h->content_length;
723 }
724 }
725 // Fall-through: single-stream if HEAD failed or server doesn't
726 // chunk. We don't log a warning here because for many small files
727 // the HEAD probe is the expected outcome of "not chunked".
728 }
729
730 if (use_chunked) {
731 // Chunked path: pre-allocate dest, N threads do Range GETs, single
732 // verify_file at end (concurrent writers can't share a streaming
733 // hash state). Retries: chunked failure falls back to single-stream
734 // (we don't retry chunked itself — a partial output file from a
735 // half-completed chunked download is hard to resume cleanly).
736 wchar_t tmp_buf[MAX_PATH * 2];
737 std::wstring parent = dest.parent_path().wstring();
738 std::wstring prefix = L".dl-";
739 if (GetTempFileNameW(parent.c_str(), prefix.c_str(), 0, tmp_buf) == 0) {
740 return std::unexpected(Error{ErrorKind::Io, "GetTempFileName failed"});
741 }
742 fs::path tmp = fs::path(tmp_buf);
743 // GetTempFileName creates a 0-byte file; preallocate_file then
744 // CREATE_ALWAYS-overwrites it to the right size.
745
746 auto rc = download_chunked(url, tmp, known_size,
747 opts.parallel_chunks, opts.timeout_seconds, label);
748 if (!rc.has_value()) {
749 log::warnf("chunked download failed ({}); falling back to single-stream",
750 rc.error().message);
751 fs::remove(tmp, ec);
752 // Fall through to single-stream below.
753 } else {
754 // Chunked path can't share a streaming hash state across N
755 // threads — verify_and_promote re-reads the assembled file.
756 // Slight cost (~0.5s per 100 MB on SSD), only correct option.
757 auto promoted = verify_and_promote(tmp, dest, opts.expected_hash, url);
758 if (!promoted) return std::unexpected(promoted.error());
759 return DownloadResult{*promoted, *rc};
760 }
761 }
762
763 Error last_err{ErrorKind::Network, "no attempts"};
764 int retries = std::max(1, opts.retries);
765 for (int attempt = 1; attempt <= retries; ++attempt) {
766 // tmp 文件：dest 同目录下 .dl-<rand>，避免跨卷 rename。
767 wchar_t tmp_buf[MAX_PATH * 2];
768 std::wstring parent = dest.parent_path().wstring();
769 std::wstring prefix = L".dl-";
770 if (GetTempFileNameW(parent.c_str(), prefix.c_str(), 0, tmp_buf) == 0) {
771 return std::unexpected(Error{ErrorKind::Io, "GetTempFileName failed"});
772 }
773 fs::path tmp = fs::path(tmp_buf);
774
775 FILE* fp = nullptr;
776#ifdef _MSC_VER
777 if (_wfopen_s(&fp, tmp.wstring().c_str(), L"wb") != 0) fp = nullptr;
778#else
779 fp = _wfopen(tmp.wstring().c_str(), L"wb");
780#endif
781 if (!fp) {
782 fs::remove(tmp, ec);
783 return std::unexpected(Error{ErrorKind::Io, "open tmp failed"});
784 }
785
786 StreamingSha256 sha;
787 std::optional<Progress> prog;
788
789 auto sink = [&](const unsigned char* data, size_t n, int64_t total)
790 -> std::expected<void, Error> {
791 if (!prog) prog.emplace(label, total);
792 if (std::fwrite(data, 1, n, fp) != n) {
793 return std::unexpected(Error{ErrorKind::Io, "fwrite failed"});
794 }
795 sha.update(data, static_cast<ULONG>(n));
796 prog->update(n);
797 return {};
798 };
799
800 auto rc = do_request(url, opts.timeout_seconds, sink);
801 if (prog) prog->finish();
802 std::fclose(fp);
803
804 if (!rc.has_value()) {
805 last_err = rc.error();
806 fs::remove(tmp, ec);
807 // 4xx 不重试
808 if (last_err.kind == ErrorKind::HttpClient) return std::unexpected(last_err);
809 // 退避后重试
810 if (attempt < retries) {
811 int wait = 1 << (attempt - 1);
812 log::warnf("download attempt {} failed ({}); retry in {}s", attempt, last_err.message, wait);
813 std::this_thread::sleep_for(std::chrono::seconds(wait));
814 }
815 continue;
816 }
817
818 // Single-stream path computed sha256 incrementally — pass to
819 // verify_and_promote so it can skip the re-read for the sha256
820 // check (the non-sha256 case still re-reads, by design).
821 std::string actual_sha = sha.finish();
822 auto promoted = verify_and_promote(
823 tmp, dest, opts.expected_hash, url, actual_sha);
824 if (!promoted) return std::unexpected(promoted.error());
825 return DownloadResult{*promoted, *rc};
826 }
827 return std::unexpected(last_err);
828#else
829 // POSIX download path (libcurl). Minimum-viable Phase B half 2:
830 // single-stream only (no chunked Range yet). Hash is recomputed on
831 // disk via verify_and_promote — slightly slower than Win32's
832 // streaming sha but correct and unblocks `luban setup` end-to-end.
833 std::error_code ec;
834 fs::create_directories(dest.parent_path(), ec);
835
836 std::string label = opts.label.empty()
837 ? (url.rfind('/') != std::string::npos ? url.substr(url.rfind('/') + 1) : url)
838 : opts.label;
839
840 Error last_err{ErrorKind::Network, "no attempts"};
841 int retries = std::max(1, opts.retries);
842 for (int attempt = 1; attempt <= retries; ++attempt) {
843 // mkstemp creates an o600 file in dest's parent dir, returning fd.
844 // The XXXXXX template gets replaced in-place; we pull the resulting
845 // path out via the buffer.
846 std::string tmpl = (dest.parent_path() / ".dl-XXXXXX").string();
847 std::vector<char> buf(tmpl.begin(), tmpl.end());
848 buf.push_back('\0');
849 int fd = mkstemp(buf.data());
850 if (fd < 0) return std::unexpected(Error{ErrorKind::Io, "mkstemp failed"});
851 fs::path tmp = fs::path(buf.data());
852
853 FILE* fp = fdopen(fd, "wb");
854 if (!fp) {
855 ::close(fd);
856 fs::remove(tmp, ec);
857 return std::unexpected(Error{ErrorKind::Io, "fdopen failed"});
858 }
859
860 std::optional<Progress> prog;
861 auto sink = [&](const unsigned char* data, size_t n, int64_t total)
862 -> std::expected<void, Error> {
863 if (!prog) prog.emplace(label, total);
864 if (std::fwrite(data, 1, n, fp) != n) {
865 return std::unexpected(Error{ErrorKind::Io, "fwrite failed"});
866 }
867 prog->update(n);
868 return {};
869 };
870
871 auto rc = do_request(url, opts.timeout_seconds, sink);
872 if (prog) prog->finish();
873 std::fclose(fp);
874
875 if (!rc.has_value()) {
876 last_err = rc.error();
877 fs::remove(tmp, ec);
878 // 4xx not retried — server says it'll never succeed.
879 if (last_err.kind == ErrorKind::HttpClient) return std::unexpected(last_err);
880 if (attempt < retries) {
881 int wait = 1 << (attempt - 1);
882 log::warnf("download attempt {} failed ({}); retry in {}s",
883 attempt, last_err.message, wait);
884 std::this_thread::sleep_for(std::chrono::seconds(wait));
885 }
886 continue;
887 }
888
889 // No streaming sha on POSIX yet — verify_and_promote recomputes
890 // by re-reading. Future: parallel POSIX StreamingSha256 via EVP.
891 auto promoted = verify_and_promote(tmp, dest, opts.expected_hash, url);
892 if (!promoted) return std::unexpected(promoted.error());
893 return DownloadResult{*promoted, *rc};
894 }
895 return std::unexpected(last_err);
896#endif
897}
