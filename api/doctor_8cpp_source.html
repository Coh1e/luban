1// `luban doctor` — health report.
2//
3// Modes:
4// - default : pretty text output, always exit 0 (for human + bug-report use)
5// - --strict : same text, but exit 1 if any check failed (CI gate)
6// - --json : machine-readable JSON, schema=1; for IDE plugins / agents
7//
8// Checks (each yields a pass/fail):
9// 1. 4 canonical homes (data / cache / state / config) — exist on disk?
10// 2. Subdirectories (toolchains, bin, registry/overlay, downloads, ...) — exist?
11// 3. installed.json present and non-empty?
12// 4. Each expected tool (cmake / ninja / clang / git / vcpkg / clangd / ...) on PATH?
13//
14// Failure semantics: a missing dir auto-created by ensure_dirs() is still a fail
15// from doctor's POV, since the user just observed the world without auto-fix.
16// The path layer in paths::ensure_dirs() is what creates them; doctor doesn't
17// silently fix.

45// Single check outcome — used by both renderers (text + JSON) so they
46// stay in sync without re-running the work.
47struct CheckRow {
48 std::string label; // human label (printed) + key (in JSON)
49 std::string detail; // path / version / found-at — empty allowed
50 bool ok = false;
51};
52
53struct Report {
54 std::vector<CheckRow> homes;
55 std::vector<CheckRow> subdirs;
56 std::vector<CheckRow> components; // detail = "<version>"; ok=false if empty
57 std::vector<CheckRow> tools;
58 bool volume_warning = false; // data/cache on different volumes
59
60 // True iff every recorded check passed. Drives --strict exit code.
61 bool all_ok() const {
62 for (auto* list : {&homes, &subdirs, &components, &tools}) {
63 for (auto& r : *list) if (!r.ok) return false;
64 }
65 return true;
66 }
67};

69Report build_report() {
70 Report r;
71 std::error_code ec;
72
73 // 1. Homes (4 canonical dirs).
74 {
75 struct Pair { std::string role; fs::path p; };
76 std::array<Pair, 4> rows = {{
77 {"data", paths::data_dir()},
78 {"cache", paths::cache_dir()},
79 {"state", paths::state_dir()},
80 {"config", paths::config_dir()},
81 }};
82 for (auto& row : rows) {
83 r.homes.push_back({row.role, row.p.string(), fs::exists(row.p, ec)});
84 }
85 r.volume_warning = !paths::same_volume(paths::data_dir(), paths::cache_dir());
86 }
87
88 // 2. Sub-dirs (everything in all_dirs minus the 4 homes).
89 {
90 for (auto& [label, p] : paths::all_dirs()) {
91 if (label == "data" || label == "cache" || label == "state" || label == "config") continue;
92 r.subdirs.push_back({label, p.string(), fs::exists(p, ec)});
93 }
94 }
95
96 // 3. Installed components — pulled from installed.json.
97 {
98 if (!fs::exists(paths::installed_json_path(), ec)) {
99 r.components.push_back({"<registry>", "missing — run `luban setup`", false});
100 } else {
101 auto recs = registry::load_installed();
102 if (recs.empty()) {
103 r.components.push_back({"<registry>", "empty — run `luban setup`", false});
104 } else {
105 for (auto& [name, rec] : recs) {
106 std::string ver = rec.version.empty() ? std::string("?") : rec.version;
107 r.components.push_back({name, ver, true});
108 }
109 }
110 }
111 }
112
113 // 4. Tools on PATH.
114 {
115 for (auto tool : tool_list::kCoreTools) {
116 std::string found = path_search::on_path_str(tool);
117 r.tools.push_back({std::string(tool), found, !found.empty()});
118 }
119 }
120
121 return r;
122}

131std::string pad(std::string_view s, size_t width) {
132 if (s.size() >= width) return std::string(s);
133 std::string out(s);
134 out.append(width - s.size(), ' ');
135 return out;
136}

138std::string marker(bool ok) {
139 return ok ? log::green("\xe2\x9c\x93") : log::dim("\xc2\xb7");

142void render_text(const Report& r) {
143 log::step("Canonical homes");
144 size_t home_w = 0;
145 for (auto& row : r.homes) home_w = std::max(home_w, row.label.size());
146 for (auto& row : r.homes) {
147 println(" " + marker(row.ok) + " " + pad(row.label, home_w) + " " + row.detail);
148 }
149 if (r.volume_warning) {
150 log::warn("data and cache are on different volumes \xe2\x80\x94 hardlink "
151 "deduplication will fall back to copy.");
152 }
153
154 std::cout << '\n';
155 log::step("Sub-directories");
156 for (auto& row : r.subdirs) {
157 println(" " + marker(row.ok) + " " + pad(row.label, 18) + " " + row.detail);
158 }
159
160 std::cout << '\n';
161 log::step("Installed components");
162 if (r.components.size() == 1 && r.components[0].label == "<registry>") {
163 println(" (" + r.components[0].detail + ")");
164 } else {
165 for (auto& row : r.components) {
166 std::string m = row.ok ? log::green("\xe2\x9c\x93") : log::red("\xe2\x9c\x97");
167 println(" " + m + " " + pad(row.label, 28) + " " + row.detail);
168 }
169 }
170
171 std::cout << '\n';
172 log::step("Tools on PATH");
173 for (auto& row : r.tools) {
174 std::string loc = row.ok ? row.detail : log::dim("(not found)");
175 println(" " + marker(row.ok) + " " + pad(row.label, 14) + " " + loc);
176 }
177}

181json row_to_json(const CheckRow& r) {
182 return json::object({{"label", r.label}, {"detail", r.detail}, {"ok", r.ok}});
183}
184
185json rows_to_json(const std::vector<CheckRow>& rows) {
186 json arr = json::array();
187 for (auto& r : rows) arr.push_back(row_to_json(r));
188 return arr;
189}
190
191void render_json(const Report& r) {
192 json doc = json::object({
193 {"schema", 1},
194 {"all_ok", r.all_ok()},
195 {"volume_warning", r.volume_warning},
196 {"homes", rows_to_json(r.homes)},
197 {"subdirs", rows_to_json(r.subdirs)},
198 {"components", rows_to_json(r.components)},
199 {"tools", rows_to_json(r.tools)},
200 });
201 std::cout << doc.dump(2) << '\n';
202}

204int run_doctor(const cli::ParsedArgs& args) {
205 bool want_json = args.flags.count("json") && args.flags.at("json");
206 bool want_strict = args.flags.count("strict") && args.flags.at("strict");
207
208 Report r = build_report();
209
210 if (want_json) {
211 render_json(r);
212 } else {
213 render_text(r);
214 }
215
216 return (want_strict && !r.all_ok()) ? 1 : 0;
217}
