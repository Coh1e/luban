36// Locate the templates/ directory relative to the running luban.exe.
37// Search order:
38// <exe_dir>/templates/<kind> (installed layout — future)
39// <exe_dir>/../templates/<kind> (uncommon)
40// <repo_root>/templates/<kind> (build/default/luban.exe → repo)
41fs::path find_template_root(std::string_view kind) {
42#ifdef _WIN32
43 wchar_t buf[MAX_PATH * 4];
44 DWORD got = GetModuleFileNameW(nullptr, buf, static_cast<DWORD>(std::size(buf)));
45 fs::path exe = (got == 0) ? fs::current_path() / "luban.exe" : fs::path(std::wstring(buf, got));
46#else
47 fs::path exe = fs::current_path() / "luban";
48#endif
49 fs::path exe_dir = exe.parent_path();
50 std::vector<fs::path> candidates = {
51 exe_dir / "templates" / std::string(kind),
52 exe_dir.parent_path() / "templates" / std::string(kind),
53 exe_dir.parent_path().parent_path() / "templates" / std::string(kind),
54 };
55 std::error_code ec;
56 for (auto& c : candidates) {
57 if (fs::is_directory(c, ec)) return c;
58 }
59 return {};
60}

62std::string expand(std::string text, const std::map<std::string, std::string>& ctx) {
63 for (auto& [k, v] : ctx) {
64 std::string needle = "{{" + k + "}}";
65 size_t pos = 0;
66 while ((pos = text.find(needle, pos)) != std::string::npos) {
67 text.replace(pos, needle.size(), v);
68 pos += v.size();
69 }
70 }
71 return text;
72}

78void write_file(const fs::path& p, const std::string& content) {
79 std::error_code ec;
80 fs::create_directories(p.parent_path(), ec);
81 std::ofstream out(p, std::ios::binary | std::ios::trunc);
82 out.write(content.data(), static_cast<std::streamsize>(content.size()));
83}

85// 把 `{{name}}` 之类的占位符也在路径组件里展开。
86// 模板里 `templates/app/src/{{name}}/main.cpp.tmpl` → 渲染时变成 `dst/src/<name>/main.cpp`。
87fs::path expand_path(const fs::path& p, const std::map<std::string, std::string>& ctx) {
88 std::string s = p.generic_string();
89 s = expand(std::move(s), ctx);
90 return fs::path(s);
91}

93int materialize(const fs::path& src_dir, const fs::path& dst_dir,
94 const std::map<std::string, std::string>& ctx) {
95 int n = 0;
96 for (auto& entry : fs::recursive_directory_iterator(src_dir)) {
97 if (!entry.is_regular_file()) continue;
98 fs::path rel = fs::relative(entry.path(), src_dir);
99 bool is_template = (rel.extension() == ".tmpl");
100 if (is_template) rel.replace_extension(); // strip ".tmpl"
101
102 // 路径里也展开 {{name}} → name
103 fs::path expanded_rel = expand_path(rel, ctx);
104 fs::path dst = dst_dir / expanded_rel;
105 if (is_template) {
106 std::string text = read_text_no_bom(entry.path());
107 write_file(dst, expand(std::move(text), ctx));
108 } else {
109 std::error_code ec;
110 fs::create_directories(dst.parent_path(), ec);
111 fs::copy_file(entry.path(), dst, fs::copy_options::overwrite_existing, ec);
112 }
113 ++n;
114 }
115 return n;
116}

118int run_new(const cli::ParsedArgs& args) {
119 if (args.positional.size() < 2) {
120 log::err("usage: luban new {app|lib} <name>");
121 return 2;
122 }
123 const std::string& kind = args.positional[0];
124 const std::string& name = args.positional[1];
125
126 if (kind != "app" && kind != "lib") {
127 log::errf("unsupported kind: {}", kind);
128 return 2;
129 }
130 if (kind == "lib") log::warn("lib templates are not differentiated yet; using app template.");
131 if (!valid_name(name)) {
132 log::errf("invalid project name '{}'. Use lowercase letters, digits, '-' and '_'; must start with a letter.", name);
133 return 2;
134 }
135
136 // --target=wasm picks templates/wasm-app/ (sets emscripten-friendly luban.cmake +
137 // CMakePresets). Default = native (templates/app/).
138 std::string target = "native";
139 {
140 auto it = args.opts.find("target");
141 if (it != args.opts.end() && !it->second.empty()) target = it->second;
142 }
143 std::string template_kind;
144 if (target == "native") {
145 template_kind = "app";
146 } else if (target == "wasm") {
147 template_kind = "wasm-app";
148 } else {
149 log::errf("unknown --target '{}'. Supported: native, wasm.", target);
150 return 2;
151 }
152
153 fs::path src = find_template_root(template_kind);
154 if (src.empty()) {
155 log::errf("template '{}' not found (looked relative to luban.exe and repo root)", template_kind);
156 return 1;
157 }
158
159 fs::path parent_arg = args.opts.count("at") ? fs::path(args.opts.at("at")) : fs::current_path();
160 std::error_code ec;
161 fs::path parent = fs::absolute(parent_arg, ec);
162 fs::path dst = parent / name;
163
164 if (fs::exists(dst, ec)) {
165 bool empty = true;
166 for (auto& _ : fs::directory_iterator(dst, ec)) { (void)_; empty = false; break; }
167 if (!empty) { log::errf("target {} already exists and is not empty", dst.string()); return 1; }
168 }
169
170 log::stepf("Scaffolding {} '{}' at {}", kind, name, dst.string());
171 fs::create_directories(dst, ec);
172 int n = materialize(src, dst, {{"name", name}});
173 log::okf("wrote {} files", n);
174
175 // Scenario 2 硬验收线（plan §0）：末尾自动跑一次 luban build，
176 // 这样 nvim / vscode 第一次打开就有 compile_commands.json 可吃。
177 // hello-world 没 vcpkg deps，build_project 自动选 no-vcpkg preset，~3-5s。
178 bool no_build = false;
179 auto it_nb = args.flags.find("no-build");
180 if (it_nb != args.flags.end() && it_nb->second) no_build = true;
181
182 // wasm target's auto-build needs `emcmake cmake --preset wasm` plumbing
183 // which build_project handles when it sees the wasm preset shape. Done.
184
185 if (no_build) {
186 log::infof("next: cd {} && luban build", name);
187 } else {
188 log::stepf("running initial `luban build` to produce compile_commands.json");
189 int rc = build_project(dst);
190 if (rc != 0) {
191 log::warnf("initial build failed (rc={}); cd {} && luban build to retry", rc, name);
192 } else {
193 log::infof("next: cd {} && nvim src/{}/main.cpp (clangd ready out of the box)", name, name);
194 }
195 }
196 return 0;
197}

212 c.n_positional = 2;

214 c.examples = {
215 "luban new app hello\tScaffold + auto-build (~3s, no deps)",
216 "luban new lib mylib\tSame, but lib instead of executable",
217 "luban new app foo --no-build\tSkip the initial build",
218 "luban new app foo --at C:\\projects\tParent directory override",

222 cli::register_subcommand(std::move(c));
223}
