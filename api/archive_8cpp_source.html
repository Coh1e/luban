19bool ends_with(const std::string& s, std::string_view suffix) {

24bool ends_with_ci(const std::string& s, std::string_view suffix) {
25 if (s.size() < suffix.size()) return false;
26 for (size_t i = 0; i < suffix.size(); ++i) {
27 char a = s[s.size() - suffix.size() + i];
28 char b = suffix[i];
29 if (std::tolower(static_cast<unsigned char>(a)) !=
30 std::tolower(static_cast<unsigned char>(b))) return false;
31 }
32 return true;
33}

35// 路径穿越检查：拒绝 ".." 段、绝对路径、根目录指代。
36// Python 用 .resolve() 比较；这里直接看路径段。
37bool is_unsafe_entry(const std::string& name) {
38 if (name.empty()) return true;
39 if (name.front() == '/' || name.front() == '\\') return true;
40 if (name.size() >= 2 && name[1] == ':') return true; // C:/ 之类
41 // 切段，每段不能是 ".."
42 size_t start = 0;
43 while (start < name.size()) {
44 size_t end = start;
45 while (end < name.size() && name[end] != '/' && name[end] != '\\') ++end;
46 std::string seg = name.substr(start, end - start);
47 if (seg == "..") return true;
48 if (end == name.size()) break;
49 start = end + 1;
50 }
51 return false;
52}

55std::expected<void, Error> extract_to(const fs::path& archive_path, const fs::path& dest_root) {
56 mz_zip_archive zip{};
57 std::string apath = archive_path.string();
58
59 if (!mz_zip_reader_init_file(&zip, apath.c_str(), 0)) {
60 return std::unexpected(Error{
61 ErrorKind::Corrupt,
62 "miniz: not a valid zip: " + apath,
63 });
64 }
65
66 mz_uint nfiles = mz_zip_reader_get_num_files(&zip);
67 std::error_code ec;
68 fs::create_directories(dest_root, ec);
69
70 for (mz_uint i = 0; i < nfiles; ++i) {
71 char name_buf[1024];
72 mz_zip_reader_get_filename(&zip, i, name_buf, sizeof(name_buf));
73 std::string name = name_buf;
74
75 if (is_unsafe_entry(name)) {
76 mz_zip_reader_end(&zip);
77 return std::unexpected(Error{
78 ErrorKind::UnsafeEntry,
79 "unsafe zip entry escapes dest: " + name,
80 });
81 }
82
83 bool is_dir = mz_zip_reader_is_file_a_directory(&zip, i);
84 fs::path target = dest_root / fs::path(name).make_preferred();
85
86 if (is_dir) {
87 fs::create_directories(target, ec);
88 continue;
89 }
90 fs::create_directories(target.parent_path(), ec);
91
92 // miniz 写文件到磁盘（zip64 大文件也走这条）
93 std::string tpath = target.string();
94 if (!mz_zip_reader_extract_to_file(&zip, i, tpath.c_str(), 0)) {
95 mz_zip_reader_end(&zip);
96 return std::unexpected(Error{
97 ErrorKind::Io,
98 "miniz: extract_to_file failed for " + name + " → " + tpath,
99 });
100 }
101
102 // 注意：miniz 不带文件 mode；POSIX 下可执行位丢失（M3 跨平台时再补）
103 }
104
105 mz_zip_reader_end(&zip);
106 return {};
107}

109// 扁平化：如果 staging 下只有一个目录（不算隐藏文件），把其内容上提一层。
110void flatten_single_wrapper(const fs::path& dir) {
111 std::error_code ec;
112 std::vector<fs::path> entries;
113 for (auto& e : fs::directory_iterator(dir, ec)) {
114 std::string n = e.path().filename().string();
115 if (!n.empty() && n.front() == '.') continue; // 跳过 .
116 entries.push_back(e.path());
117 }
118 if (entries.size() != 1) return;
119 if (!fs::is_directory(entries[0], ec)) return;
120
121 fs::path wrapper = entries[0];
122 // 把 wrapper 内每个孩子搬到 dir，避免覆盖冲突时清理已存在
123 for (auto& child : fs::directory_iterator(wrapper, ec)) {
124 fs::path target = dir / child.path().filename();
125 if (fs::exists(target, ec)) {
126 if (fs::is_directory(target, ec) && !fs::is_symlink(target, ec)) {
127 fs::remove_all(target, ec);
128 } else {
129 fs::remove(target, ec);
130 }
131 }
132 fs::rename(child.path(), target, ec);
133 if (ec) {
134 // 跨卷 fallback
135 std::error_code ec2;
136 fs::copy(child.path(), target, fs::copy_options::recursive, ec2);
137 fs::remove_all(child.path(), ec2);
138 ec.clear();
139 }
140 }
141 fs::remove_all(wrapper, ec);
142}

146std::expected<void, Error> extract(const fs::path& archive_path, const fs::path& dest_dir) {
147 std::error_code ec;
148 if (!fs::exists(archive_path, ec)) {
149 return std::unexpected(Error{ErrorKind::Io, "archive not found: " + archive_path.string()});
150 }
151
152 // 拒绝 .7z / .exe 自解压（Python 端同行为）
153 std::string lname = archive_path.filename().string();
154 std::transform(lname.begin(), lname.end(), lname.begin(),
155 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
156 if (ends_with(lname, ".7z") || ends_with(lname, ".exe") || ends_with(lname, ".msi")) {
157 return std::unexpected(Error{
158 ErrorKind::Unsupported,
159 archive_path.filename().string() + ": 7z/msi/self-extracting EXE not supported. "
160 "Provide an overlay manifest pointing to .zip or .tar.gz."
161 });
162 }
163
164 // M2 仅支持 ZIP（与 Python MVP 一致；几乎所有 toolchain 都是 zip）
165 bool is_zip_ext = ends_with_ci(lname, ".zip");
166 // 也兼容 zip 内容但扩展名不是 .zip 的情况——交给 miniz 头探测决定
167 if (!is_zip_ext) {
168 // 试探：mz_zip_reader_init_file 失败就报 Unsupported
169 mz_zip_archive probe{};
170 if (!mz_zip_reader_init_file(&probe, archive_path.string().c_str(), 0)) {
171 return std::unexpected(Error{
172 ErrorKind::Unsupported,
173 archive_path.filename().string() + ": only .zip supported in M2 (tar.gz/xz on the roadmap)",
174 });
175 }
176 mz_zip_reader_end(&probe);
177 }
178
179 fs::create_directories(dest_dir.parent_path(), ec);
180
181 // staging：dest_dir 同级临时目录，miniz 抽到这；成功后再合并
182 fs::path staging = dest_dir.parent_path() /
183 (dest_dir.filename().string() + ".luban-staging");
184 if (fs::exists(staging, ec)) fs::remove_all(staging, ec);
185 fs::create_directories(staging, ec);
186
187 if (auto rc = extract_to(archive_path, staging); !rc) {
188 fs::remove_all(staging, ec);
189 return std::unexpected(rc.error());
190 }
191
192 // 扁平化单顶层 wrapper
193 flatten_single_wrapper(staging);
194
195 // 把 staging/* 移到 dest_dir/*
196 fs::create_directories(dest_dir, ec);
197 for (auto& child : fs::directory_iterator(staging, ec)) {
198 fs::path target = dest_dir / child.path().filename();
199 if (fs::exists(target, ec)) {
200 if (fs::is_directory(target, ec) && !fs::is_symlink(target, ec)) {
201 fs::remove_all(target, ec);
202 } else {
203 fs::remove(target, ec);
204 }
205 }
206 fs::rename(child.path(), target, ec);
207 if (ec) {
208 std::error_code ec2;
209 fs::copy(child.path(), target, fs::copy_options::recursive, ec2);
210 fs::remove_all(child.path(), ec2);
211 ec.clear();
212 }
213 }
214 fs::remove_all(staging, ec);
215 return {};
216}
