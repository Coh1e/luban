40std::string render(const GenInputs& in) {
41 std::ostringstream o;
42
43 o << "# generated by luban — DO NOT EDIT BY HAND\n";
44 o << "# regenerate via: luban add / remove / sync / target add / target remove\n";
45 o << "# project deps come from vcpkg.json; flags come from luban.toml ([scaffold] section).\n";
46 o << "\n";
47
48 // ---- targets list ----
49 o << "# ---- targets (LUBAN_TARGETS) ----\n";
50 o << "set(LUBAN_TARGETS";
51 for (auto& t : in.targets) o << " " << t;
52 o << ")\n\n";
53
54 // ---- find_package 块 ----
55 // 收集 + 去重的 find_package 名（多个 boost-子库都映射到 Boost）。
56 std::vector<std::string> find_pkgs;
57 std::set<std::string> seen_find;
58 std::vector<std::string> link_targets;
59
60 for (const auto& dep : in.manifest.dependencies) {
61 auto m = lib_targets::lookup(dep.name);
62 if (m) {
63 if (seen_find.insert(m->find_package_name).second) {
64 find_pkgs.push_back(m->find_package_name);
65 }
66 for (auto& t : m->link_targets) link_targets.push_back(t);
67 } else {
68 // miss：用 vcpkg port 名直接试 find_package
69 std::string fp = dep.name;
70 // Replace '-' with '_' for cmake-friendly fallback
71 if (seen_find.insert(fp).second) find_pkgs.push_back(fp);
72 // link target 用户需要在 luban.toml [targets] 段或 src/<x>/CMakeLists.txt 自填
73 // 这里写注释提示，不阻塞构建
74 }
75 }
76
77 o << "# ---- find_package (auto from vcpkg.json dependencies) ----\n";
78 if (find_pkgs.empty()) {
79 o << "# (no dependencies declared)\n";
80 } else {
81 for (auto& p : find_pkgs) {
82 o << "find_package(" << p << " CONFIG REQUIRED)\n";
83 }
84 }
85 o << "\n";
86
87 // ---- luban_apply ----
88 o << "# ---- luban_apply: call once per target after add_executable / add_library ----\n";
89 o << "function(luban_apply target)\n";
90 o << " target_compile_features(${target} PRIVATE cxx_std_" << in.toml.project.cpp << ")\n";
91
92 std::string warns = warning_flags(in.toml.scaffold.warnings);
93 std::string sans = sanitizer_flags(in.toml.scaffold.sanitizers);
94 o << " if(NOT MSVC)\n";
95 if (!warns.empty()) {
96 o << " target_compile_options(${target} PRIVATE";
97 std::istringstream iss(warns);
98 std::string flag;
99 while (iss >> flag) o << " " << flag;
100 o << ")\n";
101 }
102 if (!sans.empty()) {
103 o << " target_compile_options(${target} PRIVATE " << sans << ")\n";
104 o << " target_link_options(${target} PRIVATE " << sans << ")\n";
105 }
106 // 静态链接 toolchain 运行时（默认）—— 让产物在裸 cmd 直接跑，
107 // 不依赖 toolchain bin/ 在 PATH 上的 libc++.dll / libwinpthread-1.dll 等。
108 // 用户想动态链接可在 src/<target>/CMakeLists.txt 里 target_link_options(... -shared-libgcc) 覆盖。
109 o << " # static-link toolchain runtime so the exe runs without toolchain DLLs on PATH.\n";
110 o << " target_link_options(${target} PRIVATE -static -static-libgcc -static-libstdc++)\n";
111 o << " endif()\n";
112
113 if (!link_targets.empty()) {
114 o << " target_link_libraries(${target} PRIVATE\n";
115 for (auto& t : link_targets) o << " " << t << "\n";
116 o << " )\n";
117 }
118
119 o << "endfunction()\n\n";
120
121 // ---- luban_register_targets ----
122 o << "# ---- luban_register_targets: call once from root CMakeLists.txt ----\n";
123 o << "function(luban_register_targets)\n";
124 o << " foreach(t IN LISTS LUBAN_TARGETS)\n";
125 o << " if(EXISTS ${CMAKE_SOURCE_DIR}/src/${t}/CMakeLists.txt)\n";
126 o << " add_subdirectory(src/${t})\n";
127 o << " else()\n";
128 o << " message(WARNING \"luban: target '${t}' has no src/${t}/CMakeLists.txt; skipping\")\n";
129 o << " endif()\n";
130 o << " endforeach()\n";
131 o << "endfunction()\n";
132
133 return o.str();
134}

136void regenerate_in_project(const fs::path& project_dir,
137 const std::vector<std::string>& targets) {
138 GenInputs in;
139 in.toml = luban_toml::load(project_dir / "luban.toml");
140 in.manifest = vcpkg_manifest::load(project_dir / "vcpkg.json");
141 in.targets = targets;
142
143 std::string content = render(in);
144
145 fs::path out = project_dir / "luban.cmake";
146 std::error_code ec;
147 fs::path tmp = out; tmp += ".tmp";
148 {
149 std::ofstream of(tmp, std::ios::binary | std::ios::trunc);
150 of.write(content.data(), static_cast<std::streamsize>(content.size()));
151 }
152 fs::rename(tmp, out, ec);
153 if (ec) {
154 fs::copy_file(tmp, out, fs::copy_options::overwrite_existing, ec);
155 fs::remove(tmp, ec);
156 }
157}

159std::vector<std::string> read_targets_from_cmake(const fs::path& project_dir) {
160 std::vector<std::string> out;
161 std::ifstream in(project_dir / "luban.cmake", std::ios::binary);
162 if (!in) return out;
163
164 std::string line;
165 while (std::getline(in, line)) {
166 // 找 set(LUBAN_TARGETS ...) 行（忽略前导空白）
167 size_t p = line.find_first_not_of(" \t");
168 if (p == std::string::npos) continue;
169 if (line.compare(p, 4, "set(") != 0) continue;
170 // 解析 set(LUBAN_TARGETS foo bar baz)
171 size_t open = line.find('(', p);
172 size_t close = line.find(')', open == std::string::npos ? p : open);
173 if (open == std::string::npos || close == std::string::npos) continue;
174 std::string body = line.substr(open + 1, close - open - 1);
175 std::istringstream iss(body);
176 std::string head;
177 if (!(iss >> head) || head != "LUBAN_TARGETS") continue;
178 std::string tok;
179 while (iss >> tok) out.push_back(tok);
180 return out; // 只取第一个 set(LUBAN_TARGETS ...)
181 }
182 return out;
183}
