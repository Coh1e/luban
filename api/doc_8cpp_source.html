1// `luban doc` — generate API docs for a luban-managed project (cargo doc-style).
2//
3// Wraps doxygen with sensible defaults so users don't author a Doxyfile by
4// hand. Two passes:
5//
6// 1. If no Doxyfile exists at the project root, materialize one from
7// `templates/Doxyfile.tmpl`, filling project_name / version / brief
8// from luban.toml + vcpkg.json (best-effort).
9//
10// 2. Spawn doxygen via env_snapshot::apply_to so it picks up any
11// luban-installed doxygen on PATH (`luban setup --with doxygen` puts
12// one there). Output lands at `<project>/build/doc/html/index.html`.
13//
14// `--open` opens the generated index in the user's default browser.
15// `--clean` deletes `<project>/build/doc/` and exits.
16//
17// Doxygen itself is NOT bundled with luban — the user runs `luban setup
18// --with doxygen` first (or installs it manually). We surface a clear
19// error pointing there if doxygen isn't on PATH.

51// Walk up from cwd to find a directory containing vcpkg.json or
52// CMakeLists.txt — that's "the project". Mirrors what luban add / build do.
53fs::path find_project_root() {
54 fs::path cur = fs::current_path();
55 std::error_code ec;
56 while (!cur.empty()) {
57 if (fs::exists(cur / "vcpkg.json", ec) || fs::exists(cur / "CMakeLists.txt", ec)) {
58 return cur;
59 }
60 fs::path parent = cur.parent_path();
61 if (parent == cur) break;
62 cur = parent;
63 }
64 return fs::current_path();
65}

67// Locate templates/Doxyfile.tmpl. Mirrors selection.cpp::seed_root /
68// manifest_source.cpp::seed_root layout — luban.exe ships with a sibling
69// templates/ directory.
70fs::path find_doxyfile_template() {
71#ifdef _WIN32
72 wchar_t buf[MAX_PATH * 4];
73 DWORD got = GetModuleFileNameW(nullptr, buf, static_cast<DWORD>(std::size(buf)));
74 fs::path exe = (got == 0) ? fs::current_path() / "luban.exe"
75 : fs::path(std::wstring(buf, got));
76#else
77 fs::path exe = fs::current_path() / "luban";
78#endif
79 fs::path d = exe.parent_path();
80 std::vector<fs::path> candidates = {
81 d / "templates" / "Doxyfile.tmpl",
82 d.parent_path() / "templates" / "Doxyfile.tmpl",
83 d.parent_path().parent_path() / "templates" / "Doxyfile.tmpl",
84 };
85 std::error_code ec;
86 for (auto& c : candidates) {
87 if (fs::exists(c, ec)) return c;
88 }
89 return {};
90}

92// Best-effort project metadata sniffer. Looks at luban.toml first
93// (canonical for luban-scaffolded projects), falls back to vcpkg.json's
94// name/version, then to the directory name.
95struct ProjectMeta {
96 std::string name;
97 std::string version;
98 std::string brief;
99};
100
101ProjectMeta sniff_project_meta(const fs::path& root) {
102 ProjectMeta m;
103 m.name = root.filename().string();
104 m.version = "0.1.0";
105 m.brief = "C++ project";
106
107 // vcpkg.json gives us name + version-string for free.
108 std::error_code ec;
109 fs::path vcpkg_json = root / "vcpkg.json";
110 if (fs::exists(vcpkg_json, ec)) {
111 try {
112 std::string text = file_util::read_text_no_bom(vcpkg_json);
113 json doc = json::parse(text);
114 if (doc.contains("name") && doc["name"].is_string()) {
115 m.name = doc["name"].get<std::string>();
116 }
117 // Both `version` and `version-string` show up in the wild.
118 for (auto* k : {"version-string", "version", "version-semver"}) {
119 if (doc.contains(k) && doc[k].is_string()) {
120 m.version = doc[k].get<std::string>();
121 break;
122 }
123 }
124 if (doc.contains("description") && doc["description"].is_string()) {
125 m.brief = doc["description"].get<std::string>();
126 }
127 } catch (...) {
128 // Malformed vcpkg.json — silently fall back to defaults.
129 }
130 }
131 return m;
132}

134// Naive {{key}} substitution. Same machinery used by new_project's template
135// engine; intentionally not regex / mustache to keep dependencies tight.
136std::string render_template(std::string text,
137 const std::map<std::string, std::string>& ctx) {
138 for (auto& [k, v] : ctx) {
139 std::string needle = "{{" + k + "}}";
140 size_t pos = 0;
141 while ((pos = text.find(needle, pos)) != std::string::npos) {
142 text.replace(pos, needle.size(), v);
143 pos += v.size();
144 }
145 }
146 return text;
147}

149// Materialize Doxyfile from template + project metadata. Idempotent — never
150// overwrites an existing Doxyfile (so user customizations survive `luban doc`
151// re-runs); returns true if it newly created the file.
152bool ensure_doxyfile(const fs::path& project_root) {
153 fs::path target = project_root / "Doxyfile";
154 std::error_code ec;
155 if (fs::exists(target, ec)) return false;
156
157 fs::path tpl = find_doxyfile_template();
158 if (tpl.empty()) {
159 log::err("could not locate Doxyfile.tmpl alongside luban.exe (templates/ dir missing?)");
160 return false;
161 }
162
163 auto meta = sniff_project_meta(project_root);
164 std::map<std::string, std::string> ctx = {
165 {"project_name", meta.name},
166 {"project_version", meta.version},
167 {"project_brief", meta.brief},
168 };
169
170 std::string tpl_text = file_util::read_text(tpl);
171 std::string out = render_template(tpl_text, ctx);
172
173 file_util::write_text_atomic(target, out);
174 log::okf("wrote {} (edit freely; `luban doc` won't overwrite it)", target.string());
175 return true;
176}

178// Open a path in the user's default browser. Best-effort; any failure is
179// non-fatal (the docs were still generated and the path was printed).
180void open_in_browser(const fs::path& p) {
181#ifdef _WIN32
182 ShellExecuteW(nullptr, L"open", p.wstring().c_str(), nullptr, nullptr, SW_SHOWNORMAL);
183#else
184 std::string cmd = "xdg-open " + p.string() + " 2>/dev/null || open " + p.string() + " 2>/dev/null";
185 std::system(cmd.c_str());
186#endif
187}

189int run_doc(const cli::ParsedArgs& a) {
190 bool open = a.flags.count("open") && a.flags.at("open");
191 bool clean = a.flags.count("clean") && a.flags.at("clean");
192
193 fs::path root = find_project_root();
194 fs::path build_doc = root / "build" / "doc";
195 std::error_code ec;
196
197 if (clean) {
198 if (fs::exists(build_doc, ec)) {
199 fs::remove_all(build_doc, ec);
200 log::okf("removed {}", build_doc.string());
201 } else {
202 log::infof("nothing to clean ({} doesn't exist)", build_doc.string());
203 }
204 return 0;
205 }
206
207 ensure_doxyfile(root);
208
209 // Spawn doxygen. env_snapshot::apply_to merges luban's PATH overlay so
210 // a `luban setup --with doxygen`-installed binary takes precedence.
211 auto env_overrides = env_snapshot::apply_to({});
212 std::vector<std::string> cmd = {"doxygen", "Doxyfile"};
213 log::stepf("running doxygen in {}", root.string());
214 int rc = proc::run(cmd, root.string(), env_overrides);
215 if (rc != 0) {
216 log::errf("doxygen exited with code {}", rc);
217 log::info("not on PATH? run `luban setup --with doxygen` and retry.");
218 return rc;
219 }
220
221 fs::path index = build_doc / "html" / "index.html";
222 if (fs::exists(index, ec)) {
223 log::okf("docs generated at {}", index.string());
224 if (open) {
225 open_in_browser(index);
226 } else {
227 log::info("re-run with --open to view in browser.");
228 }
229 } else {
230 log::warnf("doxygen finished but {} not found — check Doxyfile", index.string());
231 }
232 return 0;
233}

237void register_doc() {
238 cli::Subcommand c;
239 c.name = "doc";
240 c.help = "generate project API docs via doxygen (cargo doc-style)";
241 c.group = "project";
242 c.long_help =
243 " `luban doc` runs doxygen on the current project and writes HTML\n"
244 " output to build/doc/html/. On first run, materializes a default\n"
245 " Doxyfile from luban's template (sniffs project name/version from\n"
246 " vcpkg.json); subsequent runs reuse the file — so user edits stick.\n"
247 "\n"
248 " Doxygen is not bundled with luban. Install once via:\n"
249 " luban setup --with doxygen\n"
250 "\n"
251 " --open Open build/doc/html/index.html in the default browser.\n"
252 " --clean Delete build/doc/ and exit (no doxygen run).";
253 c.flags = {"open", "clean"};
254 c.examples = {
255 "luban doc\tGenerate (or regenerate) HTML docs",
256 "luban doc --open\tGenerate and open index.html",
257 "luban doc --clean\tWipe build/doc/",
258 };
259 c.run = run_doc;
260 cli::register_subcommand(std::move(c));
261}
