25 for (auto ext : exts) {

27 if (ext[0] != 0) trial += ext;
28 DWORD got = SearchPathW(nullptr, trial.c_str(), nullptr,
29 static_cast<DWORD>(std::size(buf)), buf, &fp);
30 if (got > 0 && got < std::size(buf)) {
31 return fs::path(win::to_utf8(std::wstring(buf, got)));

33 }
34 return std::nullopt;
35#else

40 while (start <= p.size()) {
