31std::vector<fs::path> path_dirs() {
32 std::set<std::string> seen;
33 std::vector<fs::path> ordered;
34 auto add = [&](const fs::path& p) {
35 std::error_code ec;
36 if (!fs::exists(p, ec)) return;
37#ifdef _WIN32
38 std::string key = lower(p.string());
39#else
40 std::string key = p.string();
41#endif
42 if (!seen.insert(key).second) return;
43 ordered.push_back(p);
44 };
45
46 auto recs = registry::load_installed();
47 for (auto& [_, rec] : recs) {
48 fs::path root = paths::toolchain_dir(rec.toolchain_dir);
49 for (auto& [alias, rel] : rec.bins) {
50 // Normalize separators — bin entries are stored with '/' on disk.
51 std::string norm = rel;
52 for (auto& c : norm) if (c == '/' || c == '\\') c = static_cast<char>(fs::path::preferred_separator);
53 fs::path exe = root / norm;
54 add(exe.parent_path());
55 }
56 }
57 add(paths::bin_dir());
58
59 // MSVC tool dirs (when --msvc-init has been run). Without these,
60 // which_search wouldn't be able to resolve `luban run cl ...` even
61 // though apply_to puts them on PATH for the spawned cmake. Splitting
62 // path_addition by ';' (Windows separator) into individual entries.
63 if (auto cap = msvc_env::load(); cap && !cap->path_addition.empty()) {
64 std::string s = cap->path_addition;
65 size_t start = 0;
66 while (start <= s.size()) {
67 size_t end = s.find(';', start);
68 if (end == std::string::npos) end = s.size();
69 std::string dir = s.substr(start, end - start);
70 if (!dir.empty()) add(fs::path(dir));
71 if (end == s.size()) break;
72 start = end + 1;
73 }
74 }
75
76 return ordered;
77}

79std::vector<std::pair<std::string, std::string>> env_dict() {
80 // Vars luban spawn-children should see beyond plain PATH:
81 //
82 // VCPKG_ROOT — CMakePresets.json generated by `luban new` references
83 // $env{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake.
84 // Was missing here in v0.1.x (only set in HKCU by env
85 // --user); luban-spawned cmake without env --user
86 // couldn't find vcpkg. This fixes that.
87 //
88 // VCPKG_DOWNLOADS / VCPKG_DEFAULT_BINARY_CACHE / X_VCPKG_REGISTRIES_CACHE
89 // Route vcpkg's own caches into <cache>/vcpkg/* instead
90 // of vcpkg's scattered defaults. One place to clear
91 // everything.
92 //
93 // EM_CONFIG — emscripten reads its python-style config from this
94 // path. We write the file at install time (component.cpp
95 // emscripten special case) and tell every spawned shell
96 // here so `emcc` works without a per-shell setup step.
97 std::vector<std::pair<std::string, std::string>> out;
98
99 auto recs = registry::load_installed();
100 auto vcpkg_it = recs.find("vcpkg");
101 if (vcpkg_it != recs.end() && !vcpkg_it->second.toolchain_dir.empty()) {
102 out.emplace_back("VCPKG_ROOT",
103 paths::toolchain_dir(vcpkg_it->second.toolchain_dir).string());
104 }
105
106 // vcpkg insists the cache dirs exist before it touches them. We
107 // create-on-emit here (rather than only via setup's ensure_dirs) so
108 // users who upgraded from v0.1.x without re-running `luban setup` still
109 // get a working `luban build` — the first call to env_dict materializes
110 // the dirs.
111 {
112 std::error_code ec;
113 fs::create_directories(paths::vcpkg_downloads_dir(), ec);
114 fs::create_directories(paths::vcpkg_archives_dir(), ec);
115 fs::create_directories(paths::vcpkg_registries_dir(), ec);
116 }
117 out.emplace_back("VCPKG_DOWNLOADS", paths::vcpkg_downloads_dir().string());
118 out.emplace_back("VCPKG_DEFAULT_BINARY_CACHE", paths::vcpkg_archives_dir().string());
119 out.emplace_back("X_VCPKG_REGISTRIES_CACHE", paths::vcpkg_registries_dir().string());
120
121 if (recs.find("emscripten") != recs.end()) {
122 out.emplace_back("EM_CONFIG",
123 (paths::config_dir() / "emscripten" / "config").string());
124 }
125
126 // MSVC: if the user ran `luban env --msvc-init`, the captured INCLUDE /
127 // LIB / LIBPATH / WindowsSdk* / VCToolsInstallDir / etc. live in
128 // <state>/msvc-env.json. Merge them so luban-spawned cmake can drive
129 // cl.exe without the user sourcing vcvarsall first. PATH addition is
130 // handled separately in apply_to() so it merges cleanly with luban's
131 // own PATH overlay.
132 if (auto cap = msvc_env::load()) {
133 for (auto& [k, v] : cap->vars) out.emplace_back(k, v);
134 }
135
136 return out;
137}

139std::map<std::string, std::string> apply_to(const std::map<std::string, std::string>& env) {
140 std::map<std::string, std::string> out;
141
142 // path_dirs() already includes MSVC tool dirs when --msvc-init has been
143 // run, so we just join + prepend onto whatever PATH the caller had.
144 auto extras = path_dirs();
145 if (!extras.empty()) {
146 std::string joined;
147 for (size_t i = 0; i < extras.size(); ++i) {
148 if (i) joined.push_back(kPathSep);
149 joined += extras[i].string();
150 }
151 // existing PATH: env arg first, fall back to current process env.
152 // Critical — child cmake / vcpkg need C:\Windows\System32 etc. to
153 // find ninja / system DLLs.
154 std::string existing;
155 auto it = env.find("PATH");
156 if (it != env.end()) {
157 existing = it->second;
158 } else if (const char* p = std::getenv("PATH"); p && *p) {
159 existing = p;
160 }
161 out["PATH"] = existing.empty() ? joined : (joined + kPathSep + existing);
162 }
163 for (auto& [k, v] : env_dict()) out[k] = v;
164 return out;
165}
