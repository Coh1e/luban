23// 找仓里的 manifests_seed 目录。Python 退场后，seed 在仓根 manifests_seed/。
24// 候选（按优先级）：
25// 1) <exe_dir>/manifests_seed （安装后并排，未来 luban-init.exe）
26// 2) <exe_dir>/../manifests_seed （build/luban.exe → 仓根）
27// 3) <exe_dir>/../../manifests_seed （build/default/luban.exe → 仓根）
28fs::path seed_root() {
29#ifdef _WIN32
30 wchar_t buf[MAX_PATH * 4];
31 DWORD got = GetModuleFileNameW(nullptr, buf, static_cast<DWORD>(std::size(buf)));
32 fs::path exe = (got == 0) ? fs::current_path() / "luban.exe" : fs::path(std::wstring(buf, got));
33#else
34 fs::path exe = fs::current_path() / "luban";
35#endif
36 fs::path d = exe.parent_path();
37 std::vector<fs::path> candidates = {
38 d / "manifests_seed",
39 d.parent_path() / "manifests_seed",
40 d.parent_path().parent_path() / "manifests_seed",
41 };
42 std::error_code ec;
43 for (auto& c : candidates) {
44 if (fs::is_directory(c, ec)) return c;
45 }
46 return {};
47}

49std::vector<Entry> coerce(const json& arr) {
50 std::vector<Entry> out;
51 if (!arr.is_array()) return out;
52 for (auto& item : arr) {
53 if (!item.is_object()) continue;
54 if (!item.contains("name") || !item["name"].is_string()) continue;
55 std::string name = item["name"].get<std::string>();
56 if (name.empty()) continue;
57 bool enabled = true;
58 if (item.contains("enabled") && item["enabled"].is_boolean()) {
59 enabled = item["enabled"].get<bool>();
60 }
61 std::string note;
62 if (item.contains("_why") && item["_why"].is_string()) {
63 note = item["_why"].get<std::string>();
64 }
65 out.push_back({name, enabled, note});
66 }
67 return out;
68}

72Selection load(bool deploy_seed) {
73 fs::path target = paths::selection_json_path();
74 fs::path seed = seed_root() / "selection.json";
75
76 std::error_code ec;
77 if (!fs::exists(target, ec)) {
78 if (deploy_seed && fs::exists(seed, ec)) {
79 // Copy the in-tree seed to the user's config dir on first
80 // run. file_util::write_text_atomic handles parent-dir
81 // creation + tmp+rename — atomic write is overkill for a
82 // first-time copy but harmless and dedups the I/O surface.
83 std::string text = file_util::read_text_no_bom(seed);
84 file_util::write_text_atomic(target, text);
85 } else {
86 // 不部署 seed 也不存在用户文件——直接读 seed 当源
87 target = seed;
88 }
89 }
90
91 Selection sel;
92 if (!fs::exists(target, ec)) return sel;
93
94 std::string text = file_util::read_text_no_bom(target);
95 json doc;
96 try { doc = json::parse(text); } catch (...) { return sel; }
97 if (!doc.is_object()) return sel;
98 if (doc.contains("components")) sel.components = coerce(doc["components"]);
99 if (doc.contains("extras")) sel.extras = coerce(doc["extras"]);
100 return sel;
101}

115void save(const Selection& sel) {
116 json doc = json::object();

118 doc["schema"] = 1;
119 doc["components"] = json::array();
120 for (auto& e : sel.components) doc["components"].push_back(entry_to_json(e));

124 fs::path target = paths::selection_json_path();

158std::vector<fs::path> deploy_overlays() {
159 fs::path seed = seed_root();
160 fs::path overlay = paths::overlay_dir();
161 std::vector<fs::path> deployed;
162
163 std::error_code ec;
164 fs::create_directories(overlay, ec);
165 if (seed.empty() || !fs::is_directory(seed, ec)) return deployed;
166
167 for (auto& entry : fs::directory_iterator(seed, ec)) {
168 if (!entry.is_regular_file()) continue;
169 const auto& src = entry.path();
170 if (src.extension() != ".json") continue;
171 if (src.filename() == "selection.json") continue;
172
173 fs::path dst = overlay / src.filename();
174 if (!fs::exists(dst, ec)) {
175 std::string text = file_util::read_text_no_bom(src);
176 file_util::write_text_atomic(dst, text);
177 }
178 deployed.push_back(dst);
179 }
180 return deployed;
181}
